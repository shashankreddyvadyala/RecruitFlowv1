# =============================================================================
# AP TAX CLASSIFICATION  —  CLUSTER PREDICTION
# =============================================================================
#
# PURPOSE
#   Assign every AP invoice line to a tax cluster name (e.g. "Utilities",
#   "Outside Plant Construction", "Freight - Common Carrier Charges").
#   The cluster determines how the invoice is taxed.
#
# HOW IT WORKS  (three layers, in priority order)
#   1. Parker Volpe Rules  — company-specific vendor / description / EXTC combos
#   2. Generic Rules       — field-based: MIC code, EXTC code, Account code, Vendor
#   3. Keyword Rules       — description text search (least specific, fires last)
#   If no rule matches → XGBoost ML predicts the cluster.
#
# OUTPUT COLUMN  "decision"
#   "rule"        → matched a business rule   (confidence = 1.0)
#   "ml"          → ML prediction ≥ 0.60 confidence
#   "need_review" → ML confidence < 0.60  → send to AP team for manual check
#
# SPARK CONNECT COMPATIBLE
#   No sparkContext / broadcast used.  Model is passed via Python closure.
#
# RULES SOURCE
#   Cluster_Rules_2_21_24.xlsx  (3 sheets)
#     Sheet 1: Keyword Rules  → _KEYWORD_RULES_BASE  (CELL 8)
#     Sheet 2: Generic Rules  → PySpark _assign() calls  (CELL 4, Priority 2)
#     Sheet 3: Parker Volpe   → PySpark _assign() calls  (CELL 4, Priority 1)
# =============================================================================

# =============================================================================
# CELL 4 — RULES ENGINE
# =============================================================================
# All rules are pure PySpark — evaluated lazily in a single Spark job.
# Rule priority (first match wins, later rules never overwrite):
#   Priority 1 — Parker Volpe  (most specific: vendor + description + EXTC)
#   Priority 2 — Generic Rules (field-based: MIC, EXTC, Account, Vendor)
#   Priority 3 — Keyword Rules (description text search, least specific)
# =============================================================================

def _assign(df, condition, cluster_name, rule_label):
    """
    Apply one classification rule.

    Sets 'cluster' and 'rule_matched' only when they are still NULL —
    so the first matching rule always wins and later rules are ignored.
    """
    return (
        df.withColumn(
            "cluster",
            when(col("cluster").isNull() & condition, lit(cluster_name))
            .otherwise(col("cluster"))
        ).withColumn(
            "rule_matched",
            when(col("rule_matched").isNull() & condition, lit(rule_label))
            .otherwise(col("rule_matched"))
        )
    )


# All three description columns are searched by _contains() so a match
# in any of them fires the rule — no invoice text is missed.
_DESC_COLS = [
    "final_description",    # LINE_DESCRIPTION + DESCRIPTION combined (built below)
    "DESCRIPTION",          # raw AP invoice description
    "DIST_ALL_DESCRIPTION", # distribution-level description
]


def _contains(*keywords):
    """
    Returns True if ANY description column contains ANY of the given keywords.
    Case-insensitive.  Pass multiple keywords for OR logic.
    Example: _contains("battery", "enersys")  fires if either word appears.
    """
    if not keywords:
        return lit(False)
    expr = lit(False)
    for kw in keywords:
        for col_name in _DESC_COLS:
            expr = expr | lower(col(col_name)).like(f"%{kw.lower()}%")
    return expr


def _not_contains(*keywords):
    """Returns True if NONE of the description columns contain any keyword."""
    return ~_contains(*keywords)

            trim(col(col_name)).isin(FILLER_VALUES) | col(col_name).isNull(),

    df = _assign(df, col(A).like("6998.%") | col(A).like("6999.%"),

    df = _assign(df, col(A).like("6362.%"),

    df = _assign(df, col(A).like("1439.%"),
        "Inventory",                   "acct=1439.x")
    df = _assign(df, col(A).like("651%.%"),
        "Maintenance",                 "acct=651x.x")

    # =========================================================================
    # PRIORITY 3 — KEYWORD RULES  (description text search, least specific)
    # Source: Cluster_Rules.xlsx → Sheet "Keyword Rules"
    # keyword_rules list is built in CELL 8 (hardcoded + optional Excel merge).

            when(col("cluster").isNotNull(), col("cluster"))

    total  = int(counts["n"].sum())

    if total:

              f"→ ML needed: {total - ruled:,}")
    return df

# =============================================================================
# CELL 5 — FEATURE ENGINEERING
# =============================================================================
# Builds a sparse feature matrix from a Pandas DataFrame.
# Called from both train() and the scoring closure in score().
# =============================================================================

def _to_str(value):
    """
    Safely convert any value to a plain string.
    Returns "UNKNOWN" for None, NaN, or empty arrays.
    Used to normalise values before LabelEncoder.transform().
    """
    if value is None:                                    return "UNKNOWN"
    if isinstance(value, float) and np.isnan(value):    return "UNKNOWN"
    if isinstance(value, (list, np.ndarray)):
        return str(value[0]).strip() if len(value) > 0 else "UNKNOWN"
    return str(value).strip()


def _build_features(pdf, tfidf, tfidf_vendor, cat_enc):
    """
    Build a sparse feature matrix for a Pandas DataFrame.

    Features (concatenated horizontally):
      1. Description TF-IDF      — word 1–3-grams from final_description
      2. Vendor name TF-IDF      — char 3–5-grams from VENDOR_NAME
      3. Categorical columns     — label-encoded EXTC, MIC, COMPANY, STATE, SOURCE
      4. Vendor hash             — MD5(VENDOR_NAME) % 2000  (handles unseen vendors)
      5. Transaction amount      — log1p(|TRAN_AMOUNT|)
      6. Account code hash       — MD5(ACCOUNT_CODE) % 500  (handles 120+ unique values)

    Args:
        pdf:          Pandas DataFrame with invoice rows.
        tfidf:        Fitted TfidfVectorizer for descriptions.
        tfidf_vendor: Fitted TfidfVectorizer for vendor names.
        cat_enc:      Dict of {column_name: fitted LabelEncoder}.

    Returns:
        scipy sparse matrix, shape (n_rows, n_features)
    """
    # Text features
    X_text   = tfidf.transform(pdf["final_description"].fillna("").astype(str).tolist())
    X_vendor = tfidf_vendor.transform(pdf[COLS["vendor_name"]].fillna("").astype(str).tolist())

    # Categorical features — unseen values mapped to "UNKNOWN"
    cat_parts = []
    for col_name in CAT_COLS:
        enc   = cat_enc[col_name]
        known = set(enc.classes_)
        vals  = np.array([
            v if (v := _to_str(u)) in known else "UNKNOWN"
            for u in pdf[col_name].values
        ])
        cat_parts.append(enc.transform(vals).reshape(-1, 1).astype(np.float32))

    # Numeric/hash features
    vendor_hash = (
        pdf[COLS["vendor_name"]].fillna("")
        .apply(lambda n: int(hashlib.md5(str(n).encode("utf-8")).hexdigest(), 16) % 2000)
        .values.reshape(-1, 1).astype(np.float32)
    )
    amount = (
        np.log1p(np.abs(pd.to_numeric(pdf[COLS["tran_amount"]], errors="coerce").fillna(0).values))
        .reshape(-1, 1).astype(np.float32)
    )
    acct_hash = (
        pdf[COLS["account_code"]].fillna("")
        .apply(lambda a: int(hashlib.md5(str(a).encode("utf-8")).hexdigest(), 16) % 500)
        .values.reshape(-1, 1).astype(np.float32)
    )

    dense_block = csr_matrix(np.hstack(cat_parts + [vendor_hash, amount, acct_hash]))
    return sp_hstack([X_text, X_vendor, dense_block], format="csr")


print("Features OK")


# =============================================================================
# CELL 6 — TRAIN
# =============================================================================
# Uses Spark only for sampling (fast, no shuffle).
# Pulls the sample to the driver and trains XGBoost there.
# =============================================================================

def train(df_hist, df_curr_eval=None):
    """
    Train an XGBoost multi-class classifier on rule-labelled invoice rows.

    Steps:
      1. Filter to rows where label_source="rule" (known clusters).
      2. Stratified sample via sampleBy() — no full shuffle, very fast.
      3. Pull sample to driver with toPandas().
      4. Fit TF-IDF vectorizers and LabelEncoders on the sample.
      5. Train XGBoost with early stopping.

    Args:
        df_hist:       Spark DataFrame — historical invoices (already labelled).
        df_curr_eval:  Spark DataFrame — current month invoices (used as out-of-time
                       eval set if provided; otherwise 10% holdout is used).

    Returns:
        (model, tfidf, tfidf_vendor, cat_enc, label_enc)
        — all fitted objects, ready to be passed to score().
    """
    TRAIN_COLS = (
        ["final_description", "label_cluster"]
        + CAT_COLS
        + [COLS["vendor_name"], COLS["tran_amount"], COLS["account_code"]]
    )
    avail = [c for c in TRAIN_COLS if c in df_hist.columns]

    # Filter to rows with a known cluster
    df_labelled = (
        df_hist
        .filter(col("label_source") == "rule")
        .filter(col("label_cluster").isNotNull())
        .select(avail)
    )

    # Count rows per cluster; drop clusters with too few rows
    cluster_counts = (
        df_labelled.groupBy("label_cluster")
        .agg(count("*").alias("n"))
        .toPandas()
    )
    valid_clusters = cluster_counts.loc[
        cluster_counts["n"] >= MIN_CLASS_SAMPLES, "label_cluster"
    ].tolist()
    print(f"  {int(cluster_counts['n'].sum()):,} labelled rows  |  "
          f"{len(valid_clusters)} valid clusters (≥{MIN_CLASS_SAMPLES} rows each)")

    if not valid_clusters:
        raise ValueError("No clusters meet MIN_CLASS_SAMPLES — check rules output.")

    # Stratified sample — sampleBy does not shuffle, so it is fast
    n_by_cluster = cluster_counts.set_index("label_cluster")["n"].to_dict()
    fractions    = {
        c: min(1.0, MAX_ROWS_PER_CLASS / max(1, n_by_cluster[c]))
        for c in valid_clusters
    }
    pdf = (
        df_labelled
        .filter(col("label_cluster").isin(valid_clusters))
        .sampleBy("label_cluster", fractions=fractions, seed=42)
        .toPandas()
    )
    print(f"  Training sample: {len(pdf):,} rows pulled to driver")

    # Fit description TF-IDF (word 1–3-grams, larger vocab = better phrase capture)
    min_df = 2 if len(pdf) > 500 else 1
    tfidf = TfidfVectorizer(
        max_features=8_000, ngram_range=(1, 3),
        sublinear_tf=True, min_df=min_df,
        analyzer="word", token_pattern=r"[a-zA-Z0-9]{2,}",
    )
    X_text = tfidf.fit_transform(pdf["final_description"].fillna("").astype(str).tolist())

    # Fit vendor name TF-IDF (character n-grams capture partial vendor names)
    min_df_v = 2 if len(pdf) > 300 else 1
    tfidf_vendor = TfidfVectorizer(
        max_features=1_000, analyzer="char_wb",
        ngram_range=(3, 5), sublinear_tf=True, min_df=min_df_v,
    )
    X_vendor = tfidf_vendor.fit_transform(
        pdf[COLS["vendor_name"]].fillna("").astype(str).tolist()
    )

    # Fit one LabelEncoder per categorical column (include "UNKNOWN" for unseen values)
    cat_enc  = {}
    cat_parts = []
    for col_name in CAT_COLS:
        vals = np.array([_to_str(v) for v in pdf[col_name].values])
        enc  = LabelEncoder()
        enc.fit(np.append(vals, "UNKNOWN"))
        cat_enc[col_name] = enc
        cat_parts.append(enc.transform(vals).reshape(-1, 1).astype(np.float32))

    vendor_hash = (
        pdf[COLS["vendor_name"]].fillna("")
        .apply(lambda n: int(hashlib.md5(str(n).encode("utf-8")).hexdigest(), 16) % 2000)
        .values.reshape(-1, 1).astype(np.float32)
    )
    amount = (
        np.log1p(np.abs(pd.to_numeric(pdf[COLS["tran_amount"]], errors="coerce").fillna(0).values))
        .reshape(-1, 1).astype(np.float32)
    )
    acct_hash = (
        pdf[COLS["account_code"]].fillna("")
        .apply(lambda a: int(hashlib.md5(str(a).encode("utf-8")).hexdigest(), 16) % 500)
        .values.reshape(-1, 1).astype(np.float32)
    )

    X = sp_hstack(
        [X_text, X_vendor, csr_matrix(np.hstack(cat_parts + [vendor_hash, amount, acct_hash]))],
        format="csr",
    )
    print(f"  Feature matrix: {X.shape[0]:,} rows × {X.shape[1]:,} features")

    label_enc = LabelEncoder()
    y = label_enc.fit_transform(pdf["label_cluster"].astype(str))
    del pdf  # free driver memory

    # Build evaluation set (out-of-time = current month; fallback = 10% holdout)
    pdf_eval = pd.DataFrame()
    if df_curr_eval is not None:
        avail_ev = [c for c in TRAIN_COLS if c in df_curr_eval.columns]
        pdf_eval = (
            df_curr_eval
            .filter(col("label_source") == "rule")
            .filter(col("label_cluster").isNotNull())
            .select(avail_ev).dropna(subset=["label_cluster"])
            .limit(500 * len(valid_clusters))
            .toPandas()
        )
        known    = set(label_enc.classes_)
        pdf_eval = pdf_eval[pdf_eval["label_cluster"].astype(str).isin(known)]

    if len(pdf_eval) == 0:
        n   = X.shape[0]
        if n < 10:
            raise ValueError(
                f"Only {n} training rows — need at least 10. "
                "Lower MIN_CLASS_SAMPLES or check rules output."
            )
        idx  = np.random.RandomState(42).permutation(n)
        cut  = max(1, int(n * 0.9))
        X, X_eval = X[idx[:cut]], X[idx[cut:]]
        y, y_eval = y[idx[:cut]], y[idx[cut:]]
        eval_label = "10% historical holdout"
    else:
        X_eval    = _build_features(pdf_eval, tfidf, tfidf_vendor, cat_enc)
        y_eval    = label_enc.transform(pdf_eval["label_cluster"].astype(str).values)
        eval_label = f"current-month out-of-time ({len(y_eval):,} rows)"
        del pdf_eval

    n_classes = len(label_enc.classes_)
    print(f"  Eval set: {eval_label}  |  {n_classes} clusters")
    print("  Training XGBoost …")

    model = XGBClassifier(
        n_estimators=500,
        max_depth=8,
        learning_rate=0.03,
        subsample=0.85,
        colsample_bytree=0.7,
        colsample_bylevel=0.8,
        min_child_weight=3,
        gamma=0.1,
        reg_alpha=0.1,
        reg_lambda=1.5,
        eval_metric="mlogloss",
        early_stopping_rounds=30,
        tree_method="hist",
        grow_policy="lossguide",
        max_leaves=64,
        objective="multi:softprob",
        num_class=n_classes,
        random_state=42,
        n_jobs=-1,
    )
    model.fit(X, y, eval_set=[(X_eval, y_eval)], verbose=False)

    best_n    = getattr(model, "best_iteration", model.n_estimators)
    y_pred    = np.argmax(model.predict_proba(X_eval), axis=1)
    acc       = accuracy_score(y_eval, y_pred)
    f1        = f1_score(y_eval, y_pred, average="weighted", zero_division=0)
    print(f"  Best iteration : {best_n} / {model.n_estimators} trees")
    print(f"  Eval accuracy  : {acc:.4f}")
    print(f"  Weighted F1    : {f1:.4f}")

    return model, tfidf, tfidf_vendor, cat_enc, label_enc


print("Train OK")


# =============================================================================
# CELL 7 — SCORE  (fully distributed — never pulls the full dataset to driver)
# =============================================================================
# Uses mapInPandas so XGBoost scoring runs on each Spark executor partition.
# The model and encoders are captured in the Python closure — no sparkContext
# or JVM broadcast required (compatible with Spark Connect).
# =============================================================================

def score(df, model, tfidf, tfidf_vendor, cat_enc, label_enc):
    """
    Classify every invoice row using mapInPandas (fully distributed).

    Rule rows     → cluster already known; confidence = 1.0, decision = "rule".
    No-rule rows  → XGBoost predicts the cluster.
                    confidence ≥ CONFIDENCE_THRESHOLD → decision = "ml"
                    confidence <  CONFIDENCE_THRESHOLD → decision = "need_review"

    Returns a Spark DataFrame (stays distributed throughout).

    SERIALISATION STRATEGY
    ----------------------
    mapInPandas closures are serialised by cloudpickle and shipped to executor
    processes. Passing live Python objects (XGBoost model, sklearn vectorizers)
    directly in a closure is unreliable — their internal C++ state does not always
    survive cloudpickle.

    Solution: convert every ML artifact to plain bytes on the driver, pass the
    bytes in the closure (bytes always serialise cleanly), and reconstruct the
    objects once per partition on the executor.
    """
    import pickle as _pickle

    # ── Serialise every ML artifact to plain bytes on the driver ──────────────
    # Passing live Python objects directly in a mapInPandas closure is unreliable
    # because cloudpickle re-serialises them.  Explicitly pickling to bytes first
    # means the closure only captures raw bytes — always safe to serialise.
    # The worker reconstructs each object once at partition start (not per row).
    _model_bytes        = _pickle.dumps(model)
    _tfidf_bytes        = _pickle.dumps(tfidf)
    _tfidf_vendor_bytes = _pickle.dumps(tfidf_vendor)
    _cat_enc_bytes      = _pickle.dumps(cat_enc)
    _label_enc_bytes    = _pickle.dumps(label_enc)

    # ── Capture all other worker dependencies as plain local values ───────────
    # These are small (dicts, lists, scalars) so they are safe to pass directly.
    import datetime as _dt_local
    import hashlib   as _hashlib
    import numpy     as _np
    import pandas    as _pd
    from scipy.sparse import hstack as _sp_hstack, csr_matrix as _csr_matrix

    _cols      = COLS
    _cat_cols  = CAT_COLS
    _threshold = CONFIDENCE_THRESHOLD
    _n_parts   = SCORE_PARTITIONS

    # ── Build the column selection list on the driver ─────────────────────────
    keep = list(dict.fromkeys([
        COLS["invoice_id"],
        COLS["accounting_date"],
        "INV_LINE_NUMBER",
        "INVOICE_DIST_LINE_NUM",
        "INVOICE_NUM",
        COLS["vendor_name"],
        COLS["account_code"],
        COLS["tran_amount"],
        "final_description",
        "label_source",
        "label_cluster",
        "cluster",
        "rule_matched",
    ] + CAT_COLS))
    avail = [c for c in keep if c in df.columns]

    # ── Output schema ─────────────────────────────────────────────────────────
    output_schema = StructType(
        list(df.select(avail).schema.fields) + [
            StructField("predicted_cluster", StringType(), True),
            StructField("confidence",        FloatType(),  True),
            StructField("decision",          StringType(), True),
            StructField("reported_date",     DateType(),   True),
        ]
    )

    # ── Partition scoring function ────────────────────────────────────────────
    def _score_partition(iterator):
        """
        Runs on each executor partition.
        ML artifacts are reconstructed from bytes ONCE per partition (not per row).
        All other dependencies come from the closure as plain local values.
        """
        # Reconstruct ML objects from bytes — happens once when this partition starts
        _model     = _pickle.loads(_model_bytes)
        _tfidf     = _pickle.loads(_tfidf_bytes)
        _tv        = _pickle.loads(_tfidf_vendor_bytes)
        _cat_enc   = _pickle.loads(_cat_enc_bytes)
        _label_enc = _pickle.loads(_label_enc_bytes)

        def _to_str_local(value):
            if value is None:                                  return "UNKNOWN"
            if isinstance(value, float) and _np.isnan(value): return "UNKNOWN"
            if isinstance(value, (_np.ndarray, list)):
                return str(value[0]).strip() if len(value) > 0 else "UNKNOWN"
            return str(value).strip()

        def _build_features_local(pdf):
            X_text   = _tfidf.transform(
                pdf["final_description"].fillna("").astype(str).tolist())
            X_vendor = _tv.transform(
                pdf[_cols["vendor_name"]].fillna("").astype(str).tolist())

            cat_parts = []
            for col_name in _cat_cols:
                enc   = _cat_enc[col_name]
                known = set(enc.classes_)
                vals  = _np.array([
                    v if (v := _to_str_local(u)) in known else "UNKNOWN"
                    for u in pdf[col_name].values
                ])
                cat_parts.append(enc.transform(vals).reshape(-1, 1).astype(_np.float32))

            vendor_hash = (
                pdf[_cols["vendor_name"]].fillna("")
                .apply(lambda n: int(_hashlib.md5(str(n).encode("utf-8")).hexdigest(), 16) % 2000)
                .values.reshape(-1, 1).astype(_np.float32)
            )
            amount = (
                _np.log1p(_np.abs(
                    _pd.to_numeric(pdf[_cols["tran_amount"]], errors="coerce").fillna(0).values
                )).reshape(-1, 1).astype(_np.float32)
            )
            acct_hash = (
                pdf[_cols["account_code"]].fillna("")
                .apply(lambda a: int(_hashlib.md5(str(a).encode("utf-8")).hexdigest(), 16) % 500)
                .values.reshape(-1, 1).astype(_np.float32)
            )
            return _sp_hstack(
                [X_text, X_vendor,
                 _csr_matrix(_np.hstack(cat_parts + [vendor_hash, amount, acct_hash]))],
                format="csr",
            )

        for pdf in iterator:
            if pdf.empty:
                yield pdf.assign(
                    predicted_cluster=_pd.Series(dtype="object"),
                    confidence       =_pd.Series(dtype="float32"),
                    decision         =_pd.Series(dtype="object"),
                    reported_date    =_pd.Series(dtype="object"),
                )
                continue

            rule_mask = pdf["label_source"] == "rule"
            ml_mask   = ~rule_mask

            pdf = pdf.copy()
            pdf["predicted_cluster"] = None
            pdf["confidence"]        = None
            pdf["decision"]          = None
            pdf["reported_date"]     = None

            # Rule rows — cluster already known, confidence = 1.0
            if rule_mask.any():
                pdf.loc[rule_mask, "predicted_cluster"] = pdf.loc[rule_mask, "cluster"]
                pdf.loc[rule_mask, "confidence"]        = _np.float32(1.0)
                pdf.loc[rule_mask, "decision"]          = "rule"

            # ML rows — score with XGBoost
            if ml_mask.any():
                X    = _build_features_local(pdf.loc[ml_mask].copy())
                prob = _model.predict_proba(X)
                idx  = _np.argmax(prob, axis=1)
                conf = prob[_np.arange(len(prob)), idx].astype(_np.float32)

                pdf.loc[ml_mask, "predicted_cluster"] = list(_label_enc.inverse_transform(idx))
                pdf.loc[ml_mask, "confidence"]        = list(conf)
                pdf.loc[ml_mask, "decision"] = _np.where(
                    conf >= _threshold, "ml", "need_review"
                ).tolist()

            pdf["reported_date"] = _dt_local.date.today()
            yield pdf

    return (
        df.select(avail)
          .repartition(_n_parts)
          .mapInPandas(_score_partition, schema=output_schema)
    )


print("Score OK")

def _load_keyword_rules_from_excel(excel_path):
    """
    Load keyword rules from Cluster_Rules.xlsx and merge them with
    _KEYWORD_RULES_BASE using the following strategy:

      - Hardcoded base rules are the foundation (always present).
      - Excel rule with the same first keyword → OVERRIDES the hardcoded version.
      - Excel rule with a new keyword          → APPENDED after the base list.
      - Hardcoded rules not in the Excel       → KEPT unchanged.

    This lets the AP team update rules in the Excel without touching the code,
    while ensuring developer-added rules are never silently lost.

    Reads two sheets:
      "Keyword Rules" — direct keyword → cluster mappings.
      "Generic Rules" — extracts description keyword patterns (e.g. "%ceeot%").

    Returns:
        Merged list of (keywords_tuple, cluster_name, label) tuples.
    """
    import subprocess, os, re as _re

    # Copy the Excel from ADLS to the driver's local /tmp/
    local_path = "/tmp/_cluster_rules_tmp.xlsx"
    try:
        dbutils.fs.cp(excel_path, f"file:{local_path}", recurse=False)
    except NameError:
        subprocess.run(
            ["hdfs", "dfs", "-get", "-f", excel_path, local_path], check=True
        )

    # Normalise cluster names that differ between Excel and canonical code names
    CLUSTER_FIXES = {
        "cable-metallic":              "Cable - Metallic",
        "cable - nonmetalic":          "Cable - Nonmetallic",
        "cable nonmetalic":            "Cable - Nonmetallic",
        "security service":            "Security Service",
        "tools and testing equipment": "Tools & Testing Equipment",
    }
    def fix_cluster(name):
        return CLUSTER_FIXES.get(str(name).strip().lower(), str(name).strip())

    # Fix known typos in Excel rule text before parsing
    RULE_FIXES = {"pannel": "panel", "cabel": "cable", "ectc": "extc"}
    def fix_rule(text):
        for wrong, right in RULE_FIXES.items():
            text = text.replace(wrong, right)
        return text

    # Keywords that are too vague or are parsing artifacts — skip them
    SKIP_KEYWORDS = {"data", "cabel", "cabel but", "cable but", "fiber and",
                     "but", "or", "and", "not"}

    try:
        xl = pd.ExcelFile(local_path)
    except Exception as ex:
        os.remove(local_path)
        raise RuntimeError(f"Cannot open Excel at {excel_path}: {ex}")

    excel_rules = []   # collected (kws_tuple, cluster) pairs
    seen_kw0    = set()  # deduplicate by first keyword

    # -- Sheet 1: Keyword Rules -----------------------------------------------
    try:
        df_kw = xl.parse("Keyword Rules")
    except Exception:
        df_kw = xl.parse(0)
        print("  Note: 'Keyword Rules' sheet not found — using first sheet")

    for _, row in df_kw.dropna(subset=["RULE", "CLUSTER"]).iterrows():
        raw     = fix_rule(str(row["RULE"]).strip().lower())
        cluster = fix_cluster(row["CLUSTER"])

        # Strip "but not …" qualifiers — negative conditions are handled by CELL 4
        if "but not"    in raw: raw = raw.split("but not")[0].strip().strip("()")
        if "should not" in raw: raw = raw.split("should not")[0].strip().strip("()")

        # Long battery-style rows: extract the first meaningful short phrase
        if len(raw) > 80:
            parts = [p.strip() for p in raw.replace(";", ",").split(",")
                     if 2 < len(p.strip()) < 30]
            raw = parts[0] if parts else ""

        if not raw or raw in SKIP_KEYWORDS:
            continue

        kws = tuple(k.strip() for k in raw.replace("/", ",").split(",")
                    if k.strip() and len(k.strip()) > 1)
        if not kws or kws[0] in seen_kw0:
            continue
        seen_kw0.add(kws[0])
        excel_rules.append((kws, cluster))

    # -- Sheet 2: Generic Rules — extract description keyword patterns only ---
    try:
        df_gen = xl.parse("Generic Rules")
        BAD_KW = {"base service", "landscape gravel", "fiber", "fiber and",
                  "account_code", "extc", "mic", "vendor_name"}
        for _, row in df_gen.dropna(subset=["RULE", "CLUSTER"]).iterrows():
            raw     = str(row["RULE"]).strip().lower()
            cluster = fix_cluster(row["CLUSTER"])
            match   = _re.search(r"description\s*=\s*%?([a-zA-Z0-9 _-]{3,30})%?", raw)
            if match:
                kw = match.group(1).strip()
                if kw and kw not in BAD_KW and "account" not in kw and len(kw) > 3 and kw not in seen_kw0:
                    seen_kw0.add(kw)
                    excel_rules.append(((kw,), cluster))
    except Exception as ex:
        print(f"  Note: Generic Rules sheet issue: {ex}")

    os.remove(local_path)

    # -- Merge Excel rules into the hardcoded base list -----------------------
    merged   = list(_KEYWORD_RULES_BASE)
    hc_index = {rule[0][0]: i for i, rule in enumerate(merged)}

    overrides = []
    additions = []
    for ex_kws, ex_cluster in excel_rules:
        kw0 = ex_kws[0]
        lbl = f"kw_xl={kw0[:20]}"
        if kw0 in hc_index:
            old_cluster = merged[hc_index[kw0]][1]
            merged[hc_index[kw0]] = (ex_kws, ex_cluster, lbl)
            if old_cluster != ex_cluster:
                overrides.append(f"  '{kw0}': '{old_cluster}' → '{ex_cluster}'")
        else:
            merged.append((ex_kws, ex_cluster, lbl))
            hc_index[kw0] = len(merged) - 1
            additions.append(f"  '{kw0}' → '{ex_cluster}'")

    # Print a clear summary of what changed
    print(f"  Excel merge summary:")
    print(f"    Hardcoded base : {len(_KEYWORD_RULES_BASE)} rules")
    print(f"    From Excel     : {len(excel_rules)} rules")
    print(f"    Overrides      : {len(overrides)}")
    print(f"    New additions  : {len(additions)}")
    print(f"    Final total    : {len(merged)} rules")
    for line in overrides:        print(f"    OVERRIDE {line}")
    for line in additions[:5]:    print(f"    NEW     {line}")
    if len(additions) > 5:        print(f"    ... and {len(additions)-5} more")

    return merged

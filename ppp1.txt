def _assign(df, condition, cluster_name, rule_label):
    """
    Apply one classification rule.

    Sets 'cluster' and 'rule_matched' only when they are still NULL —
    so the first matching rule always wins and later rules are ignored.
    """
    return (
        df.withColumn(
            "cluster",
            when(col("cluster").isNull() & condition, lit(cluster_name))
            .otherwise(col("cluster"))
        ).withColumn(
            "rule_matched",
            when(col("rule_matched").isNull() & condition, lit(rule_label))
            .otherwise(col("rule_matched"))
        )
    )


# All three description columns are searched by _contains() so a match
# in any of them fires the rule — no invoice text is missed.
_DESC_COLS = [
    "final_description",    # LINE_DESCRIPTION + DESCRIPTION combined (built below)
    "DESCRIPTION",          # raw AP invoice description
    "DIST_ALL_DESCRIPTION", # distribution-level description
]


def _contains(*keywords):
    """
    Returns True if ANY description column contains ANY of the given keywords.
    Case-insensitive.  Pass multiple keywords for OR logic.
    Example: _contains("battery", "enersys")  fires if either word appears.
    """
    if not keywords:
        return lit(False)
    expr = lit(False)
    for kw in keywords:
        for col_name in _DESC_COLS:
            expr = expr | lower(col(col_name)).like(f"%{kw.lower()}%")
    return expr


def _not_contains(*keywords):
    """Returns True if NONE of the description columns contain any keyword."""
    return ~_contains(*keywords)

    ruled  = int(counts.loc[counts["label_source"] == "rule", "n"].sum()) if total else 0
    if total:
        print(f"  Rule coverage: {ruled:,} / {total:,} ({ruled/total*100:.1f}%)  "
              f"→ ML needed: {total - ruled:,}")
    return df


print("Rules engine OK")

# CELL 5 — FEATURE ENGINEERING
# =============================================================================
# Builds a sparse feature matrix from a Pandas DataFrame.
# Called from both train() and the scoring closure in score().
# =============================================================================

def _to_str(value):
    """
    Safely convert any value to a plain string.
    Returns "UNKNOWN" for None, NaN, or empty arrays.
    Used to normalise values before LabelEncoder.transform().
    """
    if value is None:                                    return "UNKNOWN"
    if isinstance(value, float) and np.isnan(value):    return "UNKNOWN"
    if isinstance(value, (list, np.ndarray)):
        return str(value[0]).strip() if len(value) > 0 else "UNKNOWN"
    return str(value).strip()


def _build_features(pdf, tfidf, tfidf_vendor, tfidf_combo, cat_enc):

    # Text features
    X_text   = tfidf.transform(pdf["final_description"].fillna("").astype(str).tolist())
    X_vendor = tfidf_vendor.transform(pdf[COLS["vendor_name"]].fillna("").astype(str).tolist())

    # Combo feature: EXTC + account prefix + MIC
    # 90% of these combos map to exactly 1 cluster in real AT&T data
    combos   = (
        pdf[COLS["extc"]].fillna("") + " " +
        pdf[COLS["account_code"]].fillna("").str[:6] + " " +
        pdf[COLS["mic"]].fillna("")
    ).astype(str).tolist()
    X_combo  = tfidf_combo.transform(combos)

    # Categorical features — unseen values mapped to "UNKNOWN"
    cat_parts = []
    for col_name in CAT_COLS:
        enc   = cat_enc[col_name]
        known = set(enc.classes_)
        vals  = np.array([
            v if (v := _to_str(u)) in known else "UNKNOWN"
            for u in pdf[col_name].values
        ])
        cat_parts.append(enc.transform(vals).reshape(-1, 1).astype(np.float32))

    # Numeric / hash features
    vendor_hash = (
        pdf[COLS["vendor_name"]].fillna("")
        .apply(lambda n: int(hashlib.md5(str(n).encode("utf-8")).hexdigest(), 16) % 2000)
        .values.reshape(-1, 1).astype(np.float32)
    )
    amount = (
        np.log1p(np.abs(pd.to_numeric(pdf[COLS["tran_amount"]], errors="coerce").fillna(0).values))
        .reshape(-1, 1).astype(np.float32)
    )
    acct_hash = (
        pdf[COLS["account_code"]].fillna("")
        .apply(lambda a: int(hashlib.md5(str(a).encode("utf-8")).hexdigest(), 16) % 500)
        .values.reshape(-1, 1).astype(np.float32)
    )
    # Account prefix hash captures the account family (e.g. 6121.x = Operating Expense)
    acct_prefix_hash = (
        pdf[COLS["account_code"]].fillna("").str[:6]
        .apply(lambda a: int(hashlib.md5(str(a).encode("utf-8")).hexdigest(), 16) % 200)
        .values.reshape(-1, 1).astype(np.float32)
    )

    dense_block = csr_matrix(np.hstack(
        cat_parts + [vendor_hash, amount, acct_hash, acct_prefix_hash]
    ))
    return sp_hstack([X_text, X_vendor, X_combo, dense_block], format="csr")


print("Features OK")

# CELL 6 — TRAIN
# =============================================================================
# Uses Spark only for sampling (fast, no shuffle).
# Pulls the sample to the driver and trains XGBoost there.
# =============================================================================

def train(df_hist, df_curr_eval=None):

    TRAIN_COLS = (
        ["final_description", "label_cluster"]
        + CAT_COLS
        + [COLS["vendor_name"], COLS["tran_amount"], COLS["account_code"]]
        # Note: EXTC_CODE and MIC are already in CAT_COLS — not listed again here.
        # combo strings are built from those columns inside _build_features.
    )
    avail = [c for c in TRAIN_COLS if c in df_hist.columns]

    # Filter to rows with a known cluster
    df_labelled = (
        df_hist
        .filter(col("label_source") == "rule")
        .filter(col("label_cluster").isNotNull())
        .select(avail)
    )

    # Count rows per cluster; drop clusters with too few rows
    cluster_counts = (
        df_labelled.groupBy("label_cluster")
        .agg(count("*").alias("n"))
        .toPandas()
    )
    valid_clusters = cluster_counts.loc[
        cluster_counts["n"] >= MIN_CLASS_SAMPLES, "label_cluster"
    ].tolist()
    print(f"  {int(cluster_counts['n'].sum()):,} labelled rows  |  "
          f"{len(valid_clusters)} valid clusters (≥{MIN_CLASS_SAMPLES} rows each)")

    if not valid_clusters:
        raise ValueError("No clusters meet MIN_CLASS_SAMPLES — check rules output.")

    # Stratified sample — sampleBy does not shuffle, so it is fast
    n_by_cluster = cluster_counts.set_index("label_cluster")["n"].to_dict()
    fractions    = {
        c: min(1.0, MAX_ROWS_PER_CLASS / max(1, n_by_cluster[c]))
        for c in valid_clusters
    }
    pdf = (
        df_labelled
        .filter(col("label_cluster").isin(valid_clusters))
        .sampleBy("label_cluster", fractions=fractions, seed=42)
        .toPandas()
    )
    print(f"  Training sample: {len(pdf):,} rows pulled to driver")

    # Description TF-IDF — expanded to 12k features for better phrase coverage
    min_df = 2 if len(pdf) > 500 else 1
    tfidf = TfidfVectorizer(
        max_features=12_000, ngram_range=(1, 3),
        sublinear_tf=True, min_df=min_df,
        analyzer="word", token_pattern=r"[a-zA-Z0-9]{2,}",
    )
    X_text = tfidf.fit_transform(pdf["final_description"].fillna("").astype(str).tolist())

    # Vendor TF-IDF — expanded to 2k features, wider char n-gram range (2-5)
    # captures partial vendor names and handles company suffixes (LLC, INC, CORP)
    min_df_v = 2 if len(pdf) > 300 else 1
    tfidf_vendor = TfidfVectorizer(
        max_features=2_000, analyzer="char_wb",
        ngram_range=(2, 5), sublinear_tf=True, min_df=min_df_v,
    )
    X_vendor = tfidf_vendor.fit_transform(
        pdf[COLS["vendor_name"]].fillna("").astype(str).tolist()
    )

    # Combo TF-IDF — "EXTC + ACCOUNT_PREFIX + MIC" concatenated
    # Data analysis: 90% of these combos uniquely identify one cluster
    combo_texts = (
        pdf[COLS["extc"]].fillna("") + " " +
        pdf[COLS["account_code"]].fillna("").str[:6] + " " +
        pdf[COLS["mic"]].fillna("")
    ).astype(str).tolist()
    tfidf_combo = TfidfVectorizer(
        max_features=3_000, ngram_range=(1, 2),
        sublinear_tf=True, min_df=1,
    )
    X_combo = tfidf_combo.fit_transform(combo_texts)

    # Fit one LabelEncoder per categorical column (include "UNKNOWN" for unseen values)
    cat_enc  = {}
    cat_parts = []
    for col_name in CAT_COLS:
        vals = np.array([_to_str(v) for v in pdf[col_name].values])
        enc  = LabelEncoder()
        enc.fit(np.append(vals, "UNKNOWN"))
        cat_enc[col_name] = enc
        cat_parts.append(enc.transform(vals).reshape(-1, 1).astype(np.float32))

    vendor_hash = (
        pdf[COLS["vendor_name"]].fillna("")
        .apply(lambda n: int(hashlib.md5(str(n).encode("utf-8")).hexdigest(), 16) % 2000)
        .values.reshape(-1, 1).astype(np.float32)
    )
    amount = (
        np.log1p(np.abs(pd.to_numeric(pdf[COLS["tran_amount"]], errors="coerce").fillna(0).values))
        .reshape(-1, 1).astype(np.float32)
    )
    acct_hash = (
        pdf[COLS["account_code"]].fillna("")
        .apply(lambda a: int(hashlib.md5(str(a).encode("utf-8")).hexdigest(), 16) % 500)
        .values.reshape(-1, 1).astype(np.float32)
    )
    acct_prefix_hash = (
        pdf[COLS["account_code"]].fillna("").str[:6]
        .apply(lambda a: int(hashlib.md5(str(a).encode("utf-8")).hexdigest(), 16) % 200)
        .values.reshape(-1, 1).astype(np.float32)
    )

    X = sp_hstack(
        [X_text, X_vendor, X_combo,
         csr_matrix(np.hstack(cat_parts + [vendor_hash, amount, acct_hash, acct_prefix_hash]))],
        format="csr",
    )
    print(f"  Feature matrix: {X.shape[0]:,} rows × {X.shape[1]:,} features")

    label_enc = LabelEncoder()
    y = label_enc.fit_transform(pdf["label_cluster"].astype(str))
    del pdf  # free driver memory

    # Build evaluation set (out-of-time = current month; fallback = 10% holdout)
    pdf_eval = pd.DataFrame()
    if df_curr_eval is not None:
        avail_ev = [c for c in TRAIN_COLS if c in df_curr_eval.columns]
        pdf_eval = (
            df_curr_eval
            .filter(col("label_source") == "rule")
            .filter(col("label_cluster").isNotNull())
            .select(avail_ev).dropna(subset=["label_cluster"])
            .limit(500 * len(valid_clusters))
            .toPandas()
        )
        known    = set(label_enc.classes_)
        pdf_eval = pdf_eval[pdf_eval["label_cluster"].astype(str).isin(known)]

    if len(pdf_eval) == 0:
        n   = X.shape[0]
        if n < 10:
            raise ValueError(
                f"Only {n} training rows — need at least 10. "
                "Lower MIN_CLASS_SAMPLES or check rules output."
            )
        idx  = np.random.RandomState(42).permutation(n)
        cut  = max(1, int(n * 0.9))
        X, X_eval = X[idx[:cut]], X[idx[cut:]]
        y, y_eval = y[idx[:cut]], y[idx[cut:]]
        eval_label = "10% historical holdout"
    else:
        X_eval    = _build_features(pdf_eval, tfidf, tfidf_vendor, tfidf_combo, cat_enc)
        y_eval    = label_enc.transform(pdf_eval["label_cluster"].astype(str).values)
        eval_label = f"current-month out-of-time ({len(y_eval):,} rows)"
        del pdf_eval

    n_classes = len(label_enc.classes_)
    print(f"  Eval set: {eval_label}  |  {n_classes} clusters")
    print("  Training XGBoost …")

    model = XGBClassifier(
        n_estimators=1000,          # more trees — early stopping finds optimal
        max_depth=9,                # slightly deeper for the richer feature set
        learning_rate=0.02,         # lower LR + more trees = better generalisation
        subsample=0.85,
        colsample_bytree=0.65,      # more aggressive feature subsampling
        colsample_bylevel=0.75,
        min_child_weight=2,         # allow splits on smaller clusters
        gamma=0.05,                 # lower split threshold for finer decisions
        reg_alpha=0.05,
        reg_lambda=1.0,
        eval_metric="mlogloss",
        early_stopping_rounds=40,   # more patience with lower LR
        tree_method="hist",
        grow_policy="lossguide",
        max_leaves=96,              # more leaves for the richer feature set
        objective="multi:softprob",
        num_class=n_classes,
        random_state=42,
        n_jobs=-1,
    )
    model.fit(X, y, eval_set=[(X_eval, y_eval)], verbose=False)

    best_n    = getattr(model, "best_iteration", model.n_estimators)
    y_pred    = np.argmax(model.predict_proba(X_eval), axis=1)
    acc       = accuracy_score(y_eval, y_pred)
    f1        = f1_score(y_eval, y_pred, average="weighted", zero_division=0)
    print(f"  Best iteration : {best_n} / {model.n_estimators} trees")
    print(f"  Eval accuracy  : {acc:.4f}")
    print(f"  Weighted F1    : {f1:.4f}")

    return model, tfidf, tfidf_vendor, tfidf_combo, cat_enc, label_enc


print("Train OK")

def score(df, model, tfidf, tfidf_vendor, tfidf_combo, cat_enc, label_enc):
    import datetime

    # Columns to carry through to the output
    keep = list(dict.fromkeys([
        COLS["invoice_id"],
        COLS["accounting_date"],
        "INV_LINE_NUMBER",
        "INVOICE_DIST_LINE_NUM",
        "INVOICE_NUM",
        COLS["vendor_name"],
        COLS["account_code"],
        COLS["tran_amount"],
        "final_description",
        "label_source",
        "label_cluster",
        "cluster",
        "rule_matched",
    ] + CAT_COLS))
    avail = [c for c in keep if c in df.columns]
    df    = df.select(avail)

    today_str  = str(datetime.date.today())    # "YYYY-MM-DD" — used in Spark lit()
    today_date = datetime.date.today()         # date object — used in pandas column

    # ── Rule rows: assign predictions directly in Spark ───────────────────────
    # These rows already have a cluster — just copy it and mark confidence = 1.0.
    df_rules = (
        df.filter(col("label_source") == "rule")
          .withColumn("predicted_cluster", col("cluster"))
          .withColumn("confidence",        lit(1.0).cast(FloatType()))
          .withColumn("decision",          lit("rule"))
          .withColumn("reported_date",     lit(today_str).cast(DateType()))
    )

    # ── No-rule rows: score on the driver ─────────────────────────────────────
    # Pull only the unmatched rows (always a tiny fraction — ~1-2%).
    df_norule = df.filter(col("label_source") == "no_rule")
    n_norule  = df_norule.count()
    print(f"  ML rows to score on driver: {n_norule:,}")

    if n_norule == 0:
        # Nothing to predict — union an empty placeholder so schema is consistent
        df_ml = df_rules.limit(0)
    else:
        pdf_norule = df_norule.toPandas()

        # Build features and score
        X    = _build_features(pdf_norule, tfidf, tfidf_vendor, tfidf_combo, cat_enc)
        prob = model.predict_proba(X)
        idx  = np.argmax(prob, axis=1)
        conf = prob[np.arange(len(prob)), idx].astype(np.float32)

        pdf_norule["predicted_cluster"] = label_enc.inverse_transform(idx)
        pdf_norule["confidence"]        = conf
        pdf_norule["decision"]          = np.where(
            conf >= CONFIDENCE_THRESHOLD, "ml", "need_review"
        )
        pdf_norule["reported_date"]     = today_date  # datetime.date, not string — Arrow requires date objects for DateType

        # Define the output schema matching df_rules
        ml_schema = df_rules.schema
        df_ml = spark.createDataFrame(pdf_norule, schema=ml_schema)

    # ── Combine and return as a single Spark DataFrame ────────────────────────
    return df_rules.union(df_ml)


print("Score OK")

def _load_keyword_rules_from_excel(excel_path):
    import subprocess, os, re as _re

    # Copy the Excel from ADLS to the driver's local /tmp/
    local_path = "/tmp/_cluster_rules_tmp.xlsx"
    try:
        dbutils.fs.cp(excel_path, f"file:{local_path}", recurse=False)
    except NameError:
        subprocess.run(
            ["hdfs", "dfs", "-get", "-f", excel_path, local_path], check=True
        )

    # Normalise cluster names that differ between Excel and canonical code names
    CLUSTER_FIXES = {
        "cable-metallic":              "Cable - Metallic",
        "cable - nonmetalic":          "Cable - Nonmetallic",
        "cable nonmetalic":            "Cable - Nonmetallic",
        "security service":            "Security Service",
        "tools and testing equipment": "Tools & Testing Equipment",
    }
    def fix_cluster(name):
        return CLUSTER_FIXES.get(str(name).strip().lower(), str(name).strip())

    # Fix known typos in Excel rule text before parsing
    RULE_FIXES = {"pannel": "panel", "cabel": "cable", "ectc": "extc"}
    def fix_rule(text):
        for wrong, right in RULE_FIXES.items():
            text = text.replace(wrong, right)
        return text

    # Keywords that are too vague or are parsing artifacts — skip them
    SKIP_KEYWORDS = {"data", "cabel", "cabel but", "cable but", "fiber and",
                     "but", "or", "and", "not"}

    try:
        xl = pd.ExcelFile(local_path)
    except Exception as ex:
        os.remove(local_path)
        raise RuntimeError(f"Cannot open Excel at {excel_path}: {ex}")

    excel_rules = []   # collected (kws_tuple, cluster) pairs
    seen_kw0    = set()  # deduplicate by first keyword

    # -- Sheet 1: Keyword Rules -----------------------------------------------
    try:
        df_kw = xl.parse("Keyword Rules")
    except Exception:
        df_kw = xl.parse(0)
        print("  Note: 'Keyword Rules' sheet not found — using first sheet")

    for _, row in df_kw.dropna(subset=["RULE", "CLUSTER"]).iterrows():
        raw     = fix_rule(str(row["RULE"]).strip().lower())
        cluster = fix_cluster(row["CLUSTER"])

        # Strip "but not …" qualifiers — negative conditions are handled by CELL 4
        if "but not"    in raw: raw = raw.split("but not")[0].strip().strip("()")
        if "should not" in raw: raw = raw.split("should not")[0].strip().strip("()")

        # Long battery-style rows: extract the first meaningful short phrase
        if len(raw) > 80:
            parts = [p.strip() for p in raw.replace(";", ",").split(",")
                     if 2 < len(p.strip()) < 30]
            raw = parts[0] if parts else ""

        if not raw or raw in SKIP_KEYWORDS:
            continue

        kws = tuple(k.strip() for k in raw.replace("/", ",").split(",")
                    if k.strip() and len(k.strip()) > 1)
        if not kws or kws[0] in seen_kw0:
            continue
        seen_kw0.add(kws[0])
        excel_rules.append((kws, cluster))

    # -- Sheet 2: Generic Rules — extract description keyword patterns only ---
    try:
        df_gen = xl.parse("Generic Rules")
        BAD_KW = {"base service", "landscape gravel", "fiber", "fiber and",
                  "account_code", "extc", "mic", "vendor_name"}
        for _, row in df_gen.dropna(subset=["RULE", "CLUSTER"]).iterrows():
            raw     = str(row["RULE"]).strip().lower()
            cluster = fix_cluster(row["CLUSTER"])
            match   = _re.search(r"description\s*=\s*%?([a-zA-Z0-9 _-]{3,30})%?", raw)
            if match:
                kw = match.group(1).strip()
                if kw and kw not in BAD_KW and "account" not in kw and len(kw) > 3 and kw not in seen_kw0:
                    seen_kw0.add(kw)
                    excel_rules.append(((kw,), cluster))
    except Exception as ex:
        print(f"  Note: Generic Rules sheet issue: {ex}")

    os.remove(local_path)

    # -- Merge Excel rules into the hardcoded base list -----------------------
    merged   = list(_KEYWORD_RULES_BASE)
    hc_index = {rule[0][0]: i for i, rule in enumerate(merged)}

    overrides = []
    additions = []
    for ex_kws, ex_cluster in excel_rules:
        kw0 = ex_kws[0]
        lbl = f"kw_xl={kw0[:20]}"
        if kw0 in hc_index:
            old_cluster = merged[hc_index[kw0]][1]
            merged[hc_index[kw0]] = (ex_kws, ex_cluster, lbl)
            if old_cluster != ex_cluster:
                overrides.append(f"  '{kw0}': '{old_cluster}' → '{ex_cluster}'")
        else:
            merged.append((ex_kws, ex_cluster, lbl))
            hc_index[kw0] = len(merged) - 1
            additions.append(f"  '{kw0}' → '{ex_cluster}'")

    # Print a clear summary of what changed
    print(f"  Excel merge summary:")
    print(f"    Hardcoded base : {len(_KEYWORD_RULES_BASE)} rules")
    print(f"    From Excel     : {len(excel_rules)} rules")
    print(f"    Overrides      : {len(overrides)}")
    print(f"    New additions  : {len(additions)}")
    print(f"    Final total    : {len(merged)} rules")
    for line in overrides:        print(f"    OVERRIDE {line}")
    for line in additions[:5]:    print(f"    NEW     {line}")
    if len(additions) > 5:        print(f"    ... and {len(additions)-5} more")

    return merged

    KEYWORD_RULES = _KEYWORD_RULES_BASE

print(f"  Keyword rules ready: {len(KEYWORD_RULES)} rules ✓")
print("Keyword rules ready ✓")

# =============================================================================
# CELL 9 — RUN
# =============================================================================

print("=" * 60)
print("  AP TAX PIPELINE — CLUSTER PREDICTION")
print("=" * 60)
_run_start = _dt.datetime.now()

# -- Validate config -----------------------------------------------------------
if "daily_invoices" not in PATHS:
    raise ValueError("PATHS['daily_invoices'] is required but not set.")

_mo_parts = INVOICE_MONTH.split("-")
if (len(_mo_parts) != 2
        or not _mo_parts[0].isdigit()
        or not _mo_parts[1].isdigit()
        or not 1 <= int(_mo_parts[1]) <= 12):
    raise ValueError(f"INVOICE_MONTH must be 'YYYY-MM', got: {INVOICE_MONTH!r}")

DAILY_PATH = PATHS["daily_invoices"]
HIST_PATH  = PATHS.get("historical_labeled")
OUT_PATH   = PATHS.get("reports_output")

# Date range: score current month, train on everything before it
_yr, _mo = int(_mo_parts[0]), int(_mo_parts[1])
_start   = f"{INVOICE_MONTH}-01"
_end     = f"{(_dt.date(_yr, _mo, 1) + _dt.timedelta(days=32)).replace(day=1).strftime('%Y-%m')}-01"

print(f"  Scoring month  : {INVOICE_MONTH}  ({_start} to {_end})")
print(f"  Training data  : {'historical_labeled + daily_invoices' if HIST_PATH else 'daily_invoices (months before ' + INVOICE_MONTH + ')'}")
print(f"  Rules source   : {'Excel: ' + str(_excel_path) if _excel_path else 'hardcoded'}")
print(f"  Output path    : {OUT_PATH or '/tmp/ (default)'}")

# Helper: wraps build_labels so KEYWORD_RULES is always passed in
def _label(df):
    return build_labels(df, keyword_rules=KEYWORD_RULES)

# -- [1/4] Load and label historical (training) data --------------------------
print(f"\n[1/4] Loading historical training data …")

df_daily_hist = (
    spark.read.format("delta").load(DAILY_PATH)
    .filter(col(COLS["accounting_date"]) < _start)
)
if HIST_PATH:
    df_extra   = spark.read.format("delta").load(HIST_PATH)
    df_hist_raw = df_daily_hist.unionByName(df_extra, allowMissingColumns=True)
    print("  Source: daily_invoices + historical_labeled")
else:
    df_hist_raw = df_daily_hist
    print(f"  Source: daily_invoices (all months before {INVOICE_MONTH})")

if SAMPLE_LIMIT:
    df_hist_raw = df_hist_raw.limit(SAMPLE_LIMIT)
    print(f"  SAMPLE MODE: capped at {SAMPLE_LIMIT:,} rows")

df_hist    = df_hist_raw.transform(_label).cache()
hist_count = df_hist.count()
print(f"  Labelled + cached ✓  ({hist_count:,} rows)")
_rule_hist = df_hist.filter(col("label_source") == "rule").count()
_rule_pct  = _rule_hist / hist_count * 100 if hist_count else 0
print(f"  Rule coverage  : {_rule_hist:,} ({_rule_pct:.1f}%)")
if _rule_pct < 20:
    print("  ⚠  WARNING: rule coverage < 20% — check rules or data quality")

# -- [2/4] Load and label current month (to score) ----------------------------
print(f"\n[2/4] Loading current month ({INVOICE_MONTH}) …")

df_curr_raw = (
    spark.read.format("delta").load(DAILY_PATH)
    .filter((col(COLS["accounting_date"]) >= _start) &
            (col(COLS["accounting_date"]) <  _end))
)
if SAMPLE_LIMIT:
    df_curr_raw = df_curr_raw.limit(SAMPLE_LIMIT)

df_curr    = df_curr_raw.transform(_label).cache()
curr_count = df_curr.count()
print(f"  Labelled + cached ✓  ({curr_count:,} rows)")

if curr_count == 0:
    raise ValueError(
        f"No invoices found for INVOICE_MONTH={INVOICE_MONTH!r}. "
        "Check the date range and DAILY_PATH."
    )

# -- [3/4] Train --------------------------------------------------------------
print("\n[3/4] Training …")
model, tfidf, tfidf_vendor, tfidf_combo, cat_enc, label_enc = train(df_hist, df_curr_eval=df_curr)
df_hist.unpersist()
print("  Historical cache released ✓")
print("  Model and encoders ready for scoring ✓")

# -- [4/4] Score (distributed) ------------------------------------------------
print("\n[4/4] Scoring …")
# Note: on small sample data, need_review will be high because the model trains
# on fewer examples and assigns lower confidence. In production (millions of rows)
# expect need_review < 30%.
df_predictions = score(df_curr, model, tfidf, tfidf_vendor, tfidf_combo, cat_enc, label_enc).cache()

summary = (
    df_predictions.groupBy("decision")
    .agg(count("*").alias("n"))
    .toPandas()
    .sort_values("n", ascending=False)
)
total = int(summary["n"].sum())

print("\n  Results:")
for _, row in summary.iterrows():
    pct = row["n"] / total * 100
    bar = chr(9608) * int(pct / 2)
    print(f"    {row['decision']:<15}  {int(row['n']):>8,}  {pct:>5.1f}%  {bar}")
print(f"    {'─'*50}")
print(f"    {'TOTAL':<15}  {total:>8,}")

df_curr.unpersist()

# -- [5/4] Save output --------------------------------------------------------
_out_file = f"ap_tax_predictions_{INVOICE_MONTH}"

if OUT_PATH:
    _full_out = f"{OUT_PATH.rstrip('/')}/{_out_file}.parquet"
    df_predictions.write.mode("overwrite").parquet(_full_out)
    print(f"\n  Saved to ADLS: {_full_out}")
else:
    _full_out = f"/tmp/{_out_file}.csv"
    df_predictions.toPandas().to_csv(_full_out, index=False)
    print(f"\n  Saved locally: {_full_out}")

_elapsed = _dt.datetime.now() - _run_start
print(f"\n✓  DONE  ({_elapsed.total_seconds():.0f}s elapsed)")
print(f"   df_predictions  —  Spark DataFrame, {total:,} rows")
print(f"   Columns: {df_predictions.columns}")
print()

print("     df_predictions.filter(col('decision') == 'ml').show()")
print("     df_predictions.filter(col('decision') == 'rule').show()")

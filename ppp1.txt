def validate_rules_table(rules_df):
    """Returns (is_valid: bool, errors: list[str])."""
    errors = []
    missing_cols = [c for c in REQUIRED_RULE_COLS if c not in rules_df.columns]
    if missing_cols:
        return False, [f"Missing columns: {missing_cols}"]

    enabled = rules_df[rules_df["enabled"].astype(str).str.lower().isin(["true","1","yes"])]

    dupes = enabled[enabled["rule_id"].duplicated(keep=False)]["rule_id"].unique()
    if len(dupes):
        errors.append(f"Duplicate rule_ids: {list(dupes)}")

    invalid_ct = enabled[~enabled["condition_type"].isin(VALID_CONDITION_TYPES)]["condition_type"].unique()
    if len(invalid_ct):
        errors.append(f"Invalid condition_type values: {list(invalid_ct)}")

    blank_cl = enabled[enabled["cluster"].isna() | (enabled["cluster"].astype(str).str.strip()=="")]
    if len(blank_cl):
        errors.append(f"Rules with blank cluster: {list(blank_cl['rule_id'])}")

    req_map = {
        "VENDOR_EXTC":["vendor_like","extc_values"], "VENDOR_DESC":["vendor_like","desc_keywords"],
        "VENDOR_EXTC_DESC":["vendor_like","extc_values","desc_keywords"], "VENDOR_ONLY":["vendor_like"],
        "EXTC_ONLY":["extc_values"], "EXTC_PATTERN":["extc_values"],
        "MIC_EXACT":["mic_values"], "MIC_PATTERN":["mic_values"],
        "MIC_EXTC":["mic_values","extc_values"], "ACCT_EXTC":["extc_values","account_values"],
        "ACCT_PATTERN":["account_values"], "ACCT_DESC":["account_values","desc_keywords"],
        "COMPANY_CODE":["company_code"], "DESC_EXTC":["desc_keywords","extc_values"],
        "DESC_ONLY":["desc_keywords"],
    }
    for _, row in enabled.iterrows():
        ct = row.get("condition_type","")
        if ct not in req_map: continue
        for field in req_map[ct]:
            val = str(row.get(field,"")).strip()
            if not val or val.lower() in ["nan","none",""]:
                errors.append(f"Rule '{row['rule_id']}' ({ct}) missing: {field}")

    return len(errors) == 0, errors

def load_rules(source):
    """
    Universal loader — behaviour depends on what is passed:

      None              → returns the HARDCODED rules table built into this file.
                          Safe default: always works even without any external file.

      DataFrame         → used as-is (e.g. from load_rules_from_delta()).

      ".xlsx" / ".xls"  → reads the 3-sheet workbook:
                          'Parker Volpe Rules', 'Generic Rules', 'Keyword Rules'.

      ".csv"            → flat CSV with all required columns.

    The external source COMPLETELY REPLACES the hardcoded rules — it is not merged.
    """
    if source is None:
        df = _get_hardcoded_rules()
        df["priority"] = pd.to_numeric(df["priority"], errors="coerce").fillna(999).astype(int)
        return df

    if isinstance(source, pd.DataFrame):
        df = source.copy()
        df["priority"] = pd.to_numeric(df["priority"], errors="coerce").fillna(999).astype(int)
        return df

    if str(source).endswith(".csv"):
        df = pd.read_csv(source, dtype=str)
        df["priority"] = pd.to_numeric(df["priority"], errors="coerce").fillna(999).astype(int)
        return df

    if str(source).endswith((".xlsx", ".xls")):
        return _load_rules_from_3sheet_excel(source)

    raise ValueError(f"Unsupported rules source: {source}. Pass None, .csv, .xlsx, or a DataFrame.")


def _load_rules_from_3sheet_excel(path):
    """
    Reads the 3-sheet AP Tax Rules workbook.
    Sheets: 'Parker Volpe Rules', 'Generic Rules', 'Keyword Rules'
    Row 1 = title, Row 2 = subtitle, Row 3 = headers  (header=2 in pandas = 0-indexed row 2)
    """
    SHEET_CATEGORY = {
        'Parker Volpe Rules': 'Parker Volpe',
        'Generic Rules':      'Generic',
        'Keyword Rules':      'Keyword',
    }
    # Friendly column names (as shown in Excel) → internal column names
    FRIENDLY_TO_COL = {
        'Priority':             'priority',
        'Rule ID':              'rule_id',
        'Cluster (Target)':     'cluster',
        'Condition Type':       'condition_type',
        'Vendor Contains':      'vendor_like',
        'EXTC Code(s)':         'extc_values',
        'MIC Code(s)':          'mic_values',
        'Account Code(s)':      'account_values',
        'Description Keywords': 'desc_keywords',
        'Exclude Keywords':     'desc_exclude',
        'Company Code':         'company_code',
        'Enabled':              'enabled',
        'Notes':                'notes',
    }

    xl = pd.ExcelFile(path)
    sheets = []
    for sheet_name, category in SHEET_CATEGORY.items():
        if sheet_name not in xl.sheet_names:
            print(f"  ⚠  Sheet '{sheet_name}' not found in {path} — skipped")
            continue
        df = pd.read_excel(path, sheet_name=sheet_name, dtype=str, header=2)
        df = df.rename(columns=FRIENDLY_TO_COL)
        # Drop summary footer row (non-numeric priority)
        df = df[pd.to_numeric(df['priority'], errors='coerce').notna()].copy()
        df['priority']      = pd.to_numeric(df['priority']).astype(int)
        df['rule_category'] = category
        # Drop any completely empty rows
        df = df.dropna(subset=['rule_id', 'cluster'], how='any')
        df = df[df['rule_id'].astype(str).str.strip() != '']
        sheets.append(df)
        print(f"  ✓  {sheet_name:<25} {len(df):>3} rules")

    if not sheets:
        raise ValueError(f"No valid rule sheets found in {path}")

    combined = pd.concat(sheets, ignore_index=True).sort_values('priority', kind='stable')
    return combined

def load_rules_from_delta(spark, delta_path):
    """Databricks / PySpark: load rules from a Delta table."""
    return spark.read.format("delta").load(delta_path).toPandas()

def _rules_version_hash(rules_df):
    """Short MD5 of enabled rules content — changes whenever any rule changes."""
    enabled = rules_df[rules_df["enabled"].astype(str).str.lower().isin(["true","1","yes"])]
    return hashlib.md5(enabled.sort_values("rule_id").to_csv(index=False).encode()).hexdigest()[:12]

def _parse_vals(v):
    if not v or str(v).strip().lower() in ["nan","none",""]: return []
    return [x.strip() for x in str(v).split(",") if x.strip()]

def _field_mask(series, values):
    mask = pd.Series(False, index=series.index)
    for v in values:
        if v.startswith("*") and v.endswith("*"):
            mask |= series.str.contains(v[1:-1], case=False, regex=False, na=False)
        elif v.endswith("*"):
            mask |= series.str.startswith(v[:-1], na=False)
        elif v.startswith("*"):
            mask |= series.str.endswith(v[1:], na=False)
        else:
            mask |= (series == v)
    return mask

def _desc_mask(desc_lower, keywords):
    mask = pd.Series(False, index=desc_lower.index)
    for kw in keywords:
        mask |= desc_lower.str.contains(kw.lower(), regex=False, na=False)
    return mask

def _compile_rule(row, df):
    ct   = str(row["condition_type"]).strip().upper()
    desc = df["final_description"].str.lower().fillna("")
    vn   = df[COLS["vendor_name"]].fillna("").str.upper()
    ec   = df[COLS["extc"]].fillna("").astype(str)
    mc   = df[COLS["mic"]].fillna("").astype(str)
    ac   = df[COLS["account_code"]].fillna("").astype(str)
    co   = df[COLS["company_code"]].fillna("").astype(str)

    T = pd.Series(True,  index=df.index)
    F = pd.Series(False, index=df.index)

    ev = _parse_vals(row.get("extc_values",""));   mv = _parse_vals(row.get("mic_values",""))
    av = _parse_vals(row.get("account_values","")); dv = _parse_vals(row.get("desc_keywords",""))
    xv = _parse_vals(row.get("desc_exclude",""))
    vs = str(row.get("vendor_like","")).strip();    cs = str(row.get("company_code","")).strip()

    def _ok(s): return s and s.lower() not in ["nan","none",""]

    vnd_m  = vn.str.contains(vs.upper(), regex=False, na=False) if _ok(vs) else T
    extc_m = _field_mask(ec, ev) if ev else T
    mic_m  = _field_mask(mc, mv) if mv else T
    acct_m = _field_mask(ac, av) if av else T
    desc_m = _desc_mask(desc, dv) if dv else T
    excl_m = ~_desc_mask(desc, xv) if xv else T
    co_m   = (co == cs) if _ok(cs) else T

    # MIC_PATTERN AND logic: "FBR*,*K" = startswith FBR AND endswith K
    if ct == "MIC_PATTERN" and len(mv) == 2 and mv[0].endswith("*") and mv[1].startswith("*"):
        mic_m = mc.str.startswith(mv[0][:-1], na=False) & mc.str.endswith(mv[1][1:], na=False)

    return {
        "VENDOR_EXTC":      vnd_m & extc_m,
        "VENDOR_DESC":      vnd_m & desc_m & excl_m,
        "VENDOR_EXTC_DESC": vnd_m & extc_m & desc_m & excl_m,
        "VENDOR_ONLY":      vnd_m,
        "EXTC_ONLY":        extc_m,
        "EXTC_PATTERN":     extc_m,
        "MIC_EXACT":        mic_m,
        "MIC_PATTERN":      mic_m,
        "MIC_EXTC":         mic_m & extc_m,
        "ACCT_EXTC":        acct_m & extc_m,
        "ACCT_PATTERN":     acct_m & excl_m,
        "ACCT_DESC":        acct_m & desc_m & excl_m,
        "COMPANY_CODE":     co_m,
        "DESC_EXTC":        desc_m & extc_m & excl_m,
        "DESC_ONLY":        desc_m & excl_m,
    }.get(ct, F)

def build_labels(df, rules_df):
    df = df.copy()
    df["final_description"] = df[COLS["final_description"]].fillna("").astype(str)
    df["cluster"]      = pd.Series(pd.NA, index=df.index, dtype="object")
    df["rule_matched"] = pd.Series(pd.NA, index=df.index, dtype="object")

    active = (
        rules_df[rules_df["enabled"].astype(str).str.lower().isin(["true","1","yes"])]
        .copy().reset_index(drop=True).sort_values("priority", kind="stable")
    )

    matched_counts, skipped = {}, []
    for _, row in active.iterrows():
        try:
            mask = _compile_rule(row, df)
        except Exception as e:
            skipped.append((row["rule_id"], str(e))); continue
        unfilled = df["cluster"].isna() & mask
        n_new = int(unfilled.sum())
        if n_new > 0:
            df.loc[unfilled, "cluster"]      = row["cluster"]
            df.loc[unfilled, "rule_matched"] = row["rule_id"]
            matched_counts[row["rule_id"]]   = n_new

    if skipped:
        print(f"  ⚠  {len(skipped)} rule(s) skipped: {[r for r,_ in skipped]}")

    df["label_cluster"] = df["cluster"]
    df["label_source"]  = np.where(df["cluster"].notna(), "rule", "no_rule")
    return df, matched_counts

def _to_str(v):
    if v is None: return "UNKNOWN"
    if isinstance(v, float) and np.isnan(v): return "UNKNOWN"
    return str(v).strip() or "UNKNOWN"

def _build_features(pdf, tfidf, tfidf_vendor, tfidf_combo, cat_enc):
    X_text   = tfidf.transform(pdf["final_description"].fillna("").astype(str).tolist())
    X_vendor = tfidf_vendor.transform(pdf[COLS["vendor_name"]].fillna("").astype(str).tolist())
    combos   = (pdf[COLS["extc"]].fillna("") + " " +
                pdf[COLS["account_code"]].fillna("").str[:6] + " " +
                pdf[COLS["mic"]].fillna("")).astype(str).tolist()
    X_combo  = tfidf_combo.transform(combos)
    cat_parts = []
    for cn in CAT_COLS:
        enc = cat_enc[cn]; known = set(enc.classes_)
        vals = np.array([v if (v := _to_str(u)) in known else "UNKNOWN" for u in pdf[cn].values])
        cat_parts.append(enc.transform(vals).reshape(-1,1).astype(np.float32))
    v_h = pdf[COLS["vendor_name"]].fillna("").apply(lambda n: int(hashlib.md5(str(n).encode()).hexdigest(),16)%2000).values.reshape(-1,1).astype(np.float32)
    amt = np.log1p(np.abs(pd.to_numeric(pdf[COLS["tran_amount"]],errors="coerce").fillna(0).values)).reshape(-1,1).astype(np.float32)
    a_h = pdf[COLS["account_code"]].fillna("").apply(lambda a: int(hashlib.md5(str(a).encode()).hexdigest(),16)%500).values.reshape(-1,1).astype(np.float32)
    a_p = pdf[COLS["account_code"]].fillna("").str[:6].apply(lambda a: int(hashlib.md5(str(a).encode()).hexdigest(),16)%200).values.reshape(-1,1).astype(np.float32)
    dense = csr_matrix(np.hstack(cat_parts + [v_h, amt, a_h, a_p]))
    return sp_hstack([X_text, X_vendor, X_combo, dense], format="csr")

def train(df_hist, df_curr_eval=None):
    TC    = ["final_description","label_cluster"] + CAT_COLS + [COLS["vendor_name"], COLS["tran_amount"], COLS["account_code"]]
    avail = [c for c in TC if c in df_hist.columns]
    pdf   = df_hist[df_hist["label_source"]=="rule"][avail].dropna(subset=["label_cluster"])

    counts = pdf["label_cluster"].value_counts()
    valid  = counts[counts >= MIN_CLASS_SAMPLES].index.tolist()
    print(f"  {len(pdf):,} labelled rows  |  {len(valid)} valid clusters")
    if not valid: raise ValueError("No clusters meet MIN_CLASS_SAMPLES.")

    sampled = []
    for _, grp in pdf[pdf["label_cluster"].isin(valid)].groupby("label_cluster"):
        sampled.append(grp.sample(min(len(grp), MAX_ROWS_PER_CLASS), random_state=42))
    pdf = pd.concat(sampled, ignore_index=True)
    print(f"  Training sample: {len(pdf):,} rows")

    min_df = 2 if len(pdf) > 500 else 1
    tfidf = TfidfVectorizer(max_features=12_000, ngram_range=(1,3), sublinear_tf=True,
                             min_df=min_df, analyzer="word", token_pattern=r"[a-zA-Z0-9]{2,}")
    X_text = tfidf.fit_transform(pdf["final_description"].fillna("").astype(str).tolist())

    tfidf_vendor = TfidfVectorizer(max_features=2_000, analyzer="char_wb", ngram_range=(2,5),
                                    sublinear_tf=True, min_df=2 if len(pdf)>300 else 1)
    X_vendor = tfidf_vendor.fit_transform(pdf[COLS["vendor_name"]].fillna("").astype(str).tolist())

    combos = (pdf[COLS["extc"]].fillna("") + " " +
              pdf[COLS["account_code"]].fillna("").str[:6] + " " +
              pdf[COLS["mic"]].fillna("")).astype(str).tolist()
    tfidf_combo = TfidfVectorizer(max_features=3_000, ngram_range=(1,2), sublinear_tf=True, min_df=1)
    X_combo = tfidf_combo.fit_transform(combos)

    cat_enc, cat_parts = {}, []
    for cn in CAT_COLS:
        vals = np.array([_to_str(v) for v in pdf[cn].values])
        enc  = LabelEncoder(); enc.fit(np.append(vals, "UNKNOWN"))
        cat_enc[cn] = enc
        cat_parts.append(enc.transform(vals).reshape(-1,1).astype(np.float32))

    v_h = pdf[COLS["vendor_name"]].fillna("").apply(lambda n: int(hashlib.md5(str(n).encode()).hexdigest(),16)%2000).values.reshape(-1,1).astype(np.float32)
    amt = np.log1p(np.abs(pd.to_numeric(pdf[COLS["tran_amount"]],errors="coerce").fillna(0).values)).reshape(-1,1).astype(np.float32)
    a_h = pdf[COLS["account_code"]].fillna("").apply(lambda a: int(hashlib.md5(str(a).encode()).hexdigest(),16)%500).values.reshape(-1,1).astype(np.float32)
    a_p = pdf[COLS["account_code"]].fillna("").str[:6].apply(lambda a: int(hashlib.md5(str(a).encode()).hexdigest(),16)%200).values.reshape(-1,1).astype(np.float32)

    X = sp_hstack([X_text, X_vendor, X_combo, csr_matrix(np.hstack(cat_parts + [v_h, amt, a_h, a_p]))], format="csr")
    print(f"  Feature matrix: {X.shape[0]:,} × {X.shape[1]:,}")

    label_enc = LabelEncoder()
    y = label_enc.fit_transform(pdf["label_cluster"].astype(str))

    pdf_eval = pd.DataFrame()
    if df_curr_eval is not None:
        avail_ev = [c for c in TC if c in df_curr_eval.columns]
        pdf_eval = df_curr_eval[df_curr_eval["label_source"]=="rule"][avail_ev].dropna(subset=["label_cluster"])
        known    = set(label_enc.classes_)
        pdf_eval = pdf_eval[pdf_eval["label_cluster"].astype(str).isin(known)].sample(min(len(pdf_eval), 500*len(valid)), random_state=42)

    if len(pdf_eval) == 0:
        n = X.shape[0]; idx = np.random.RandomState(42).permutation(n); cut = max(1, int(n*0.9))
        X, X_eval = X[idx[:cut]], X[idx[cut:]]
        y, y_eval = y[idx[:cut]], y[idx[cut:]]
        eval_label = "10% holdout"
    else:
        X_eval     = _build_features(pdf_eval, tfidf, tfidf_vendor, tfidf_combo, cat_enc)
        y_eval     = label_enc.transform(pdf_eval["label_cluster"].astype(str).values)
        eval_label = f"out-of-time ({len(y_eval):,} rows)"

    n_classes = len(label_enc.classes_)
    print(f"  Eval: {eval_label}  |  {n_classes} clusters")
    print("  Training XGBoost …")

    model = XGBClassifier(
        n_estimators=N_ESTIMATORS, max_depth=MAX_DEPTH, learning_rate=LEARNING_RATE,
        subsample=0.85, colsample_bytree=0.65, colsample_bylevel=0.75,
        min_child_weight=2, gamma=0.05, reg_alpha=0.05, reg_lambda=1.0,
        eval_metric="mlogloss", early_stopping_rounds=EARLY_STOPPING,
        tree_method="hist", grow_policy="lossguide", max_leaves=64,
        objective="multi:softprob", num_class=n_classes,
        random_state=42, n_jobs=-1, verbosity=0,
    )
    model.fit(X, y, eval_set=[(X_eval, y_eval)], verbose=False)

    best_n = getattr(model, "best_iteration", N_ESTIMATORS)
    y_pred = np.argmax(model.predict_proba(X_eval), axis=1)
    acc    = accuracy_score(y_eval, y_pred)
    f1_w   = f1_score(y_eval, y_pred, average="weighted", zero_division=0)
    f1_mac = f1_score(y_eval, y_pred, average="macro",    zero_division=0)
    print(f"  Best iteration : {best_n}  |  Accuracy: {acc:.4f}  |  F1-weighted: {f1_w:.4f}")
    return model, tfidf, tfidf_vendor, tfidf_combo, cat_enc, label_enc, X_eval, y_eval, y_pred, acc, f1_w, f1_mac, best_n, n_classes, eval_label

def score(df, model, tfidf, tfidf_vendor, tfidf_combo, cat_enc, label_enc):
    today = _dt.date.today(); df = df.copy()
    df_rules = df[df["label_source"]=="rule"].copy()
    df_rules["predicted_cluster"] = df_rules["cluster"]
    df_rules["confidence"]        = 1.0
    df_rules["decision"]          = "rule"
    df_rules["reported_date"]     = today
    df_norule = df[df["label_source"]=="no_rule"].copy()
    print(f"  ML rows to score: {len(df_norule):,}")
    if len(df_norule) > 0:
        X    = _build_features(df_norule, tfidf, tfidf_vendor, tfidf_combo, cat_enc)
        prob = model.predict_proba(X)
        idx  = np.argmax(prob, axis=1)
        conf = prob[np.arange(len(prob)), idx].astype(np.float32)
        df_norule["predicted_cluster"] = label_enc.inverse_transform(idx)
        df_norule["confidence"]        = conf
        df_norule["decision"]          = np.where(conf >= CONFIDENCE_THRESHOLD, "ml", "need_review")
        df_norule["reported_date"]     = today
    return pd.concat([df_rules, df_norule], ignore_index=True)

def _plot_confusion_matrix(y_true, y_pred, label_enc, tmpdir, top_n=15):
    classes = label_enc.classes_; counts = np.bincount(y_true)
    top_idx = np.argsort(counts)[-top_n:][::-1]
    mask    = np.isin(y_true, top_idx); yt, yp = y_true[mask], y_pred[mask]
    remap   = {old: new for new, old in enumerate(top_idx)}
    yt2 = np.array([remap[v] for v in yt]); yp2 = np.array([remap.get(v,-1) for v in yp])
    valid = yp2 >= 0; yt2, yp2 = yt2[valid], yp2[valid]
    top_labels = [classes[i] for i in top_idx]
    cm = confusion_matrix(yt2, yp2, labels=list(range(len(top_labels))))
    fig, ax = plt.subplots(figsize=(14, 11))
    sns.heatmap(cm, annot=True, fmt="d", cmap="Blues", xticklabels=top_labels, yticklabels=top_labels, ax=ax, linewidths=0.4, annot_kws={"size":7})
    ax.set_xlabel("Predicted"); ax.set_ylabel("Actual"); ax.set_title(f"Confusion Matrix (top {top_n})")
    plt.xticks(rotation=45, ha="right", fontsize=7); plt.yticks(fontsize=7); plt.tight_layout()
    p = os.path.join(tmpdir,"confusion_matrix.png"); fig.savefig(p,dpi=120); plt.close(fig); return p

    report  = classification_report(y_true, y_pred, labels=present, target_names=names, zero_division=0)
    p = os.path.join(tmpdir,"classification_report.txt")
    with open(p,"w") as f: f.write(report)
    return p

def _save_rule_coverage(df_labelled, matched_counts, rules_df, tmpdir):
    total = len(df_labelled)
    rc = pd.Series(matched_counts, name="rows_matched").reset_index()
    rc.columns = ["rule_id","rows_matched"]
    rc = rc.merge(rules_df[["rule_id","priority","cluster","condition_type","enabled"]], on="rule_id", how="left")
    rc["pct_of_total"] = (rc["rows_matched"] / total * 100).round(2)
    p = os.path.join(tmpdir,"rule_coverage.csv"); rc.sort_values("rows_matched",ascending=False).to_csv(p,index=False); return p

def _save_need_review(df_out, tmpdir):
    nr = (df_out[df_out["decision"]=="need_review"]
          [[COLS["invoice_id"], COLS["vendor_name"], "final_description",
            "predicted_cluster", "confidence", COLS["account_code"], COLS["extc"]]]
          .sort_values("confidence"))
    p = os.path.join(tmpdir,"need_review_invoices.csv"); nr.to_csv(p,index=False); return p

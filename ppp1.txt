def load_rules(excel_path=None):
    """
    Load rules from Excel. Returns a dict with keys: keyword, generic, parker_volpe.
    If no path is provided, returns empty rules — all invoices will go to ML.
    """
    if not excel_path:
        print("   [RULES] No Excel path set — skipping rules. All invoices go to ML.")
        empty = pd.DataFrame(columns=["rule", "cluster"])
        return {"keyword": empty, "generic": empty, "parker_volpe": pd.DataFrame()}

    print(f"   Loading rules from: {excel_path}")

    # pd.read_excel cannot read abfss:// paths directly.
    # Copy from ADLS to driver-local /tmp first.
    local_path = "/tmp/cluster_rules.xlsx"
    dbutils.fs.cp(excel_path, f"file:{local_path}", recurse=False)

    keyword_df = pd.read_excel(local_path, sheet_name="Keyword Rules")
    generic_df = pd.read_excel(local_path, sheet_name="Generic Rules")
    pv_raw     = pd.read_excel(local_path, sheet_name="Parker Volpe - Rules")

    keyword_df.columns = ["rule", "cluster", "generic_satisfied"]
    generic_df.columns = ["rule", "cluster", "generic_satisfied", "notes"]

    rules = {
        "keyword":      keyword_df.dropna(subset=["rule", "cluster"]),
        "generic":      generic_df.dropna(subset=["rule", "cluster"]),
        "parker_volpe": _parse_parker_volpe(pv_raw),
    }
    print(f"   Keyword: {len(rules['keyword'])}  Generic: {len(rules['generic'])}  "
          f"Parker Volpe: {len(rules['parker_volpe'])}")
    return rules


def _parse_parker_volpe(raw):
    """
    Parse the Parker Volpe sheet into structured records.
    The sheet has 5 sections of increasing specificity:
      1. EXTC only
      2. Vendor only
      3. Vendor + EXTC
      4. Vendor + Description keyword
      5. Vendor + Description keyword + EXTC  (most specific — fires last, always wins)
    """
    records = []

    # Section 1: EXTC only
    records.append({"extc": "69A", "cluster": "Operating Lease", "rule_type": "extc_only"})

    # Section 2: Vendor only (rows 6–17)
    for _, row in raw.iloc[6:18, :2].dropna(how="all").iterrows():
        if pd.notna(row.iloc[0]) and pd.notna(row.iloc[1]):
            records.append({
                "vendor_name": str(row.iloc[0]).strip(),
                "cluster":     str(row.iloc[1]).strip(),
                "rule_type":   "vendor_only",
            })

    # Section 3: Vendor + EXTC (rows 21–28)
    for _, row in raw.iloc[21:29, :3].dropna(how="all").iterrows():
        if pd.notna(row.iloc[0]) and pd.notna(row.iloc[2]):
            records.append({
                "vendor_name": str(row.iloc[0]).strip(),
                "extc":        str(row.iloc[1]).strip(),
                "cluster":     str(row.iloc[2]).strip(),
                "rule_type":   "vendor_extc",
            })

    # Section 4: Vendor + Description keyword (rows 32–41)
    for _, row in raw.iloc[32:42, :3].dropna(how="all").iterrows():
        if pd.notna(row.iloc[0]) and pd.notna(row.iloc[2]):
            records.append({
                "vendor_name":         str(row.iloc[0]).strip(),
                "description_keyword": str(row.iloc[1]).strip() if pd.notna(row.iloc[1]) else "",
                "cluster":             str(row.iloc[2]).strip(),
                "rule_type":           "vendor_desc",
            })

    # Section 5: Vendor + Description keyword + EXTC (rows 45–49)
    for _, row in raw.iloc[45:50, :4].dropna(how="all").iterrows():
        if pd.notna(row.iloc[0]) and pd.notna(row.iloc[3]):
            records.append({
                "vendor_name":         str(row.iloc[0]).strip(),
                "description_keyword": str(row.iloc[1]).strip() if pd.notna(row.iloc[1]) else "",
                "extc":                str(row.iloc[2]).strip(),
                "cluster":             str(row.iloc[3]).strip(),
                "rule_type":           "vendor_desc_extc",
            })

    return pd.DataFrame(records)


# ── Rule application helpers ──────────────────────────────────────────────────────

def _assign(df, condition, cluster_name, rule_label):
    """
    Overwrite cluster and rule_matched columns where condition is True.
    Called repeatedly — later calls overwrite earlier ones (= higher priority wins).
    """
    return (
        df
        .withColumn("cluster",      when(condition, lit(cluster_name)).otherwise(col("cluster")))
        .withColumn("rule_matched", when(condition, lit(rule_label)).otherwise(col("rule_matched")))
    )


def _desc_contains(*keywords):
    """
    Returns a Spark condition: final_description contains ANY of the given keywords (case-insensitive).
    """
    if not keywords:
        return lit(False)
    cond = lower(col("final_description")).like(f"%{keywords[0].lower()}%")
    for kw in keywords[1:]:
        cond = cond | lower(col("final_description")).like(f"%{kw.lower()}%")
    return cond


def build_final_description(df):
    """
    Builds 'final_description' by combining available text fields.
    Priority: LINE_DESCRIPTION + DESCRIPTION > LINE_DESCRIPTION > DESCRIPTION > DIST_ALL_DESCRIPTION
    Strips known filler values that carry no tax classification signal.
    """
    filler_values = ["SCM Purchase", "SCM PURCHASE", "scm purchase"]

    def _null_if_filler(col_name):
        return when(col(col_name).isin(filler_values) | col(col_name).isNull(), lit(None)) \
               .otherwise(col(col_name))

    df = (df
          .withColumn("_line", _null_if_filler(COLS["line_description"]))
          .withColumn("_desc", _null_if_filler(COLS["description"]))
          .withColumn("_dist", _null_if_filler(COLS["dist_all_description"]))
    )

    df = df.withColumn(
        "final_description",
        coalesce(
            # Best case: both line + description available — concatenate them
            when(col("_line").isNotNull() & col("_desc").isNotNull(),
                 concat_ws(" ", col("_line"), col("_desc"))),
            col("_line"),   # Only line description
            col("_desc"),   # Only description
            col("_dist"),   # Fallback: distribution description
            lit(""),        # Last resort: empty string (never null)
        )
    ).drop("_line", "_desc", "_dist")

    return df


def _apply_keyword_rules(df, keyword_rules):
    """
    Match keywords in final_description.
    Supports: single keyword, comma-separated OR list, "X but not Y" exclusion.
    """
    for _, rule in keyword_rules.iterrows():
        raw_rule = str(rule["rule"]).strip()
        cluster  = rule["cluster"]

        if "but not" in raw_rule.lower():
            # e.g. "cable but not cable-metallic"
            base, exclusion_str = raw_rule.lower().split("but not", 1)
            exclusions = [e.strip() for e in exclusion_str.replace("(", "").replace(")", "").split(",")]
            cond = lower(col("final_description")).like(f"%{base.strip()}%")
            for excl in exclusions:
                cond = cond & ~lower(col("final_description")).like(f"%{excl}%")
        elif "," in raw_rule:
            # e.g. "ladder, scaffolding, rigging" → match any
            cond = _desc_contains(*[k.strip() for k in raw_rule.split(",")])
        else:
            # Simple single keyword
            cond = lower(col("final_description")).like(f"%{raw_rule.lower()}%")

        df = _assign(df, cond, cluster, f"kw:{raw_rule[:40]}")

    return df

def _apply_parker_volpe_rules(df, pv_rules):
    """
    Apply Parker Volpe (AT&T-specific) override rules.
    5 levels of specificity — applied in order so most specific always wins.
    """
    EXTC   = COLS["extc"]
    VENDOR = COLS["vendor_name"]

    # Level 1: EXTC code only
    for _, r in pv_rules[pv_rules["rule_type"] == "extc_only"].iterrows():
        df = _assign(df, col(EXTC) == r["extc"], r["cluster"], f"pv:extc={r['extc']}")

    # Level 2: Vendor name only
    for _, r in pv_rules[pv_rules["rule_type"] == "vendor_only"].iterrows():
        df = _assign(df, col(VENDOR) == r["vendor_name"], r["cluster"],
                     f"pv:vendor={r['vendor_name']}")

    # Level 3: Vendor + EXTC (supports OR-separated EXTC list)
    for _, r in pv_rules[pv_rules["rule_type"] == "vendor_extc"].iterrows():
        extc_values = [e.strip() for e in str(r.get("extc", "")).split("OR")]
        df = _assign(df,
                     (col(VENDOR) == r["vendor_name"]) & col(EXTC).isin(extc_values),
                     r["cluster"], f"pv:vendor+extc")

    # Level 4: Vendor + description keyword
    for _, r in pv_rules[pv_rules["rule_type"] == "vendor_desc"].iterrows():
        keywords = [k.strip() for k in str(r.get("description_keyword", "")).split("OR")]
        df = _assign(df,
                     (col(VENDOR) == r["vendor_name"]) & _desc_contains(*keywords),
                     r["cluster"], f"pv:vendor+desc")

    # Level 5: Vendor + description keyword + EXTC (most specific — fires last, always wins)
    for _, r in pv_rules[pv_rules["rule_type"] == "vendor_desc_extc"].iterrows():
        extc_values = [e.strip() for e in str(r.get("extc", "")).split("OR")]
        keywords    = [k.strip() for k in str(r.get("description_keyword", "")).split("OR")]
        df = _assign(df,
                     (col(VENDOR) == r["vendor_name"]) &
                     _desc_contains(*keywords) &
                     col(EXTC).isin(extc_values),
                     r["cluster"], f"pv:vendor+desc+extc")

    return df


def _map_cluster_to_account_code(df, spark):
    """
    Join cluster name → account code from the mapping CSV.
    If no mapping CSV is set, falls back to the raw ACCOUNT_CODE on the invoice.
    """
    map_path = PATHS.get("cluster_account_map")

    if map_path:
        # CSV must have columns: cluster, account_code_mapped
        cluster_map = (
            spark.read.csv(map_path, header=True, inferSchema=True)
            .withColumnRenamed("cluster", "_cm_cluster")
        )
        df = (
            df.join(F.broadcast(cluster_map), col("cluster") == col("_cm_cluster"), how="left")
              .drop("_cm_cluster")
        )
        df = df.withColumn("label_account_code",
                   when(col("cluster").isNotNull() & col("account_code_mapped").isNotNull(),
                        col("account_code_mapped"))
                   .otherwise(col(COLS["account_code"])))
    else:
        # No mapping CSV yet — use the raw ACCOUNT_CODE from the invoice as the training label
        df = df.withColumn("account_code_mapped", lit(None).cast("string"))
        df = df.withColumn("label_account_code", col(COLS["account_code"]))

    # label_source: "rule" if a rule fired, "no_rule" if nothing matched → goes to ML
    df = df.withColumn("label_source",
               when(col("cluster").isNotNull(), lit("rule"))
               .otherwise(lit("no_rule")))
    return df


def build_labels(df, rules, spark):
    """
    Master label builder.
    1. Build final_description from available text fields
    2. Apply all three rule sets in priority order (keyword → generic → Parker Volpe)
    3. Map cluster names to account codes
    4. Tag each invoice as label_source = "rule" or "no_rule"
    """
    df = build_final_description(df)

    # Initialize rule output columns as null (not yet matched)
    df = df.withColumn("cluster",      lit(None).cast("string"))
    df = df.withColumn("rule_matched", lit(None).cast("string"))

    # Apply rules — each pass can overwrite previous, so highest priority fires last
    df = _apply_keyword_rules(df,      rules["keyword"])      # lowest priority
    df = _apply_generic_rules(df)                             # medium priority
    df = _apply_parker_volpe_rules(df, rules["parker_volpe"]) # highest priority

    df = _map_cluster_to_account_code(df, spark)

    rule_ct    = df.filter(col("label_source") == "rule").count()
    no_rule_ct = df.filter(col("label_source") == "no_rule").count()
    total      = rule_ct + no_rule_ct
    print(f"   Rule-matched:  {rule_ct:,}  ({rule_ct/total*100:.1f}%)")
    print(f"   No rule match: {no_rule_ct:,}  ({no_rule_ct/total*100:.1f}%)  → goes to ML")

    return df

print("✓ Rules engine OK")


# ================================================================================
# CELL 5 — FEATURE ENGINEERING
#
# Converts raw invoice fields into a numeric feature vector for ML:
#   • GTE Large model  → 1024-dim semantic text embedding from final_description
#   • StringIndexer    → numeric index for high-cardinality fields (EXTC, MIC, etc.)
#   • OneHotEncoder    → binary vector for low-cardinality fields (STATE, COMPANY)
#   • VectorAssembler  → combine everything into one 'combined_vector' column
#
# Train mode: fits and saves the categorical encoding pipeline
# Inference mode: loads the saved pipeline (so encoding is identical to training)
# ================================================================================

def _build_text_embeddings(df, spark):
    """
    Run GTE Large on every row's final_description.
    Uses a Pandas UDF so the model runs distributed across Spark workers.
    Adds 'text_embedding' column: array of 1024 floats, L2-normalized.
    """
    model_name = FEATURE_CONFIG["gte_model_name"]
    batch_size = FEATURE_CONFIG["embedding_batch_size"]
    # Broadcast just the model name string — each worker loads the model locally
    model_name_bc = spark.sparkContext.broadcast(model_name)

    @F.pandas_udf(ArrayType(FloatType()))
    def _embed_udf(texts: pd.Series) -> pd.Series:
        model = SentenceTransformer(model_name_bc.value)
        vecs  = model.encode(
            texts.fillna("").tolist(),
            batch_size=batch_size,
            normalize_embeddings=True,
            show_progress_bar=False,
        )
        return pd.Series(vecs.tolist())

    return df.withColumn("text_embedding", _embed_udf(col("final_description")))


def _fit_categorical_pipeline(df):
    """
    Fit StringIndexer + OneHotEncoder on training data.
    Returns: (fitted PipelineModel, transformed df, list of output column names)
    Save the returned pipeline and reuse it at inference time.
    """
    null_fill  = FEATURE_CONFIG["null_fill_value"]
    high_card  = FEATURE_CONFIG["high_cardinality_cols"]
    low_card   = FEATURE_CONFIG["low_cardinality_cols"]

    df     = df.fillna({c: null_fill for c in high_card + low_card})
    stages = []
    out_cols = []

    # High-cardinality fields: just index (no one-hot — too many categories)
    for c in high_card:
        out = f"{c}_idx"
        stages.append(StringIndexer(inputCol=c, outputCol=out, handleInvalid="keep"))
        out_cols.append(out)

    # Low-cardinality fields: index then one-hot encode
    for c in low_card:
        idx_col = f"{c}_idx"
        ohe_col = f"{c}_ohe"
        stages.append(StringIndexer(inputCol=c, outputCol=idx_col, handleInvalid="keep"))
        stages.append(OneHotEncoder(inputCol=idx_col, outputCol=ohe_col, dropLast=True))
        out_cols.append(ohe_col)

    pipeline_model = Pipeline(stages=stages).fit(df)
    return pipeline_model, pipeline_model.transform(df), out_cols


def _apply_categorical_pipeline(df, pipeline_model):
    """Apply a previously fitted categorical pipeline to new data."""
    null_fill = FEATURE_CONFIG["null_fill_value"]
    high_card = FEATURE_CONFIG["high_cardinality_cols"]
    low_card  = FEATURE_CONFIG["low_cardinality_cols"]
    df = df.fillna({c: null_fill for c in high_card + low_card})
    out_cols = (
        [f"{c}_idx" for c in high_card] +
        [f"{c}_ohe" for c in low_card]
    )
    return pipeline_model.transform(df), out_cols


def _assemble_combined_vector(df, categorical_output_cols):
    """
    Merge text embedding (1024-dim array) + categorical features into
    one Spark ML dense vector called 'combined_vector'.
    """
    @F.udf(VectorUDT())
    def _array_to_vector(arr):
        if arr is None:
            return Vectors.dense([0.0] * FEATURE_CONFIG["embedding_dim"])
        return Vectors.dense([float(x) for x in arr])

    df = df.withColumn("text_embedding_vec", _array_to_vector(col("text_embedding")))

    assembler = VectorAssembler(
        inputCols=["text_embedding_vec"] + categorical_output_cols,
        outputCol="combined_vector",
        handleInvalid="keep",
    )
    return assembler.transform(df)


def run_feature_engineering_train(df, spark):
    """Full feature engineering in train mode: fits and returns the pipeline."""
    print("   Building text embeddings (GTE Large)...")
    df = _build_text_embeddings(df, spark)
    print("   Fitting categorical encoding pipeline...")
    pipeline_model, df, cat_cols = _fit_categorical_pipeline(df)
    print("   Assembling combined feature vector...")
    df = _assemble_combined_vector(df, cat_cols)
    total_dims = FEATURE_CONFIG["embedding_dim"] + len(cat_cols)
    print(f"   Feature vector dimensions: ~{total_dims}")
    return df, pipeline_model


def run_feature_engineering_inference(df, spark, pipeline_model):
    """Feature engineering in inference mode: applies a saved pipeline."""
    print("   Building text embeddings (GTE Large)...")
    df = _build_text_embeddings(df, spark)
    print("   Applying saved categorical encoding pipeline...")
    df, cat_cols = _apply_categorical_pipeline(df, pipeline_model)
    print("   Assembling combined feature vector...")
    df = _assemble_combined_vector(df, cat_cols)
    return df

print("✓ Feature engineering OK")


# ================================================================================
# CELL 6 — MODEL TRAINING
#
# Trains XGBoost and LightGBM classifiers on rule-labeled invoices.
# Also builds an embedding index for the similarity fallback at inference time.
#
# Only rule-labeled invoices are used for training — their labels are reliable
# (assigned by deterministic rules). "no_rule" invoices are excluded.
# ================================================================================

def _prepare_training_arrays(df):
    """
    Pull rule-labeled invoices from Spark into numpy arrays for sklearn/xgboost.
    Returns: X (features), y (encoded labels), text_embeddings, label_encoder
    """
    print("   Pulling rule-labeled records to driver...")
    pdf = (
        df.filter(col("label_source") == "rule")
          .withColumn("vec_array", vector_to_array(col("combined_vector")))
          .select("vec_array", "text_embedding", "label_account_code", COLS["invoice_id"])
          .dropna(subset=["vec_array", "label_account_code"])
          .toPandas()
    )

    X          = np.array(pdf["vec_array"].tolist(),      dtype=np.float32)
    embeddings = np.array(pdf["text_embedding"].tolist(), dtype=np.float32)

    label_enc = LabelEncoder()
    y         = label_enc.fit_transform(pdf["label_account_code"].astype(str))

    print(f"   Training records: {len(X):,}  |  Dims: {X.shape[1]}  |  Classes: {len(label_enc.classes_)}")
    return X, y, embeddings, label_enc


def _build_embedding_index(text_embeddings, y_encoded, label_encoder):
    """
    Build the k-NN similarity index from training embeddings.
    Stores L2-normalized text embeddings so cosine similarity = dot product at query time.
    """
    norms = np.linalg.norm(text_embeddings, axis=1, keepdims=True)
    norms = np.where(norms == 0, 1.0, norms)  # Avoid divide-by-zero
    normalized = (text_embeddings / norms).astype(np.float32)
    return {
        "embeddings": normalized,
        "labels":     label_encoder.inverse_transform(y_encoded),
    }


def train_models(df):
    """
    Train XGBoost + LightGBM. Evaluate both individually and as a weighted ensemble.
    Logs metrics to MLflow if tracking URI is set, otherwise prints to console.
    Returns: xgb_model, lgb_model, label_encoder, embedding_index
    """
    cfg = MODEL_CONFIG
    X, y, embeddings, label_enc = _prepare_training_arrays(df)

    X_train, X_test, y_train, y_test, emb_train, _ = train_test_split(
        X, y, embeddings,
        test_size=cfg["test_size"],
        random_state=cfg["random_seed"],
        stratify=y,
    )

    # Set up MLflow if available, otherwise use a no-op context
    use_mlflow = bool(PATHS.get("mlflow_tracking_uri"))
    if use_mlflow:
        mlflow.set_tracking_uri(PATHS["mlflow_tracking_uri"])
        mlflow.set_experiment(cfg["mlflow_experiment"])
        run_ctx = mlflow.start_run(run_name="xgb_lgb_ensemble")
        run_ctx.__enter__()
    else:
        print("   [MLflow] No tracking URI — metrics printed to console only.")
        run_ctx = None

    def _log_metrics(metrics):
        if use_mlflow:
            mlflow.log_metrics(metrics)

    try:
        # ── Train XGBoost ────────────────────────────────────────────────────────
        print("   Training XGBoost...")
        xgb_model = XGBClassifier(**cfg["xgb_params"], objective="multi:softprob",
                                  num_class=len(label_enc.classes_))
        xgb_model.fit(X_train, y_train, eval_set=[(X_test, y_test)], verbose=100)

        xgb_preds = xgb_model.predict(X_test)
        xgb_acc   = accuracy_score(y_test, xgb_preds)
        xgb_f1    = f1_score(y_test, xgb_preds, average="weighted", zero_division=0)
        _log_metrics({"xgb_accuracy": xgb_acc, "xgb_f1_weighted": xgb_f1})
        print(f"   XGBoost  → accuracy={xgb_acc:.4f}  f1_weighted={xgb_f1:.4f}")

        # ── Train LightGBM ───────────────────────────────────────────────────────
        print("   Training LightGBM...")
        lgb_model = LGBMClassifier(**cfg["lgb_params"], objective="multiclass",
                                   num_class=len(label_enc.classes_))
        lgb_model.fit(X_train, y_train, eval_set=[(X_test, y_test)])

        lgb_preds = lgb_model.predict(X_test)
        lgb_acc   = accuracy_score(y_test, lgb_preds)
        lgb_f1    = f1_score(y_test, lgb_preds, average="weighted", zero_division=0)
        _log_metrics({"lgb_accuracy": lgb_acc, "lgb_f1_weighted": lgb_f1})
        print(f"   LightGBM → accuracy={lgb_acc:.4f}  f1_weighted={lgb_f1:.4f}")

        # ── Evaluate ensemble ────────────────────────────────────────────────────
        w_xgb     = cfg["ensemble_weights"]["xgb"]
        w_lgb     = cfg["ensemble_weights"]["lgb"]
        ens_proba = w_xgb * xgb_model.predict_proba(X_test) + w_lgb * lgb_model.predict_proba(X_test)
        ens_preds = np.argmax(ens_proba, axis=1)
        ens_acc   = accuracy_score(y_test, ens_preds)
        ens_f1    = f1_score(y_test, ens_preds, average="weighted", zero_division=0)
        _log_metrics({"ensemble_accuracy": ens_acc, "ensemble_f1_weighted": ens_f1})
        print(f"   Ensemble → accuracy={ens_acc:.4f}  f1_weighted={ens_f1:.4f}")

        if use_mlflow:
            mlflow.xgboost.log_model(xgb_model, "xgboost_model",
                                     registered_model_name="ap_tax_xgboost")
            mlflow.lightgbm.log_model(lgb_model, "lightgbm_model",
                                      registered_model_name="ap_tax_lightgbm")

    finally:
        if run_ctx:
            run_ctx.__exit__(None, None, None)

    # Build the k-NN index using training text embeddings (not combined features)
    # Important: at inference we query with text embeddings, so index must also be text embeddings
    emb_index = _build_embedding_index(emb_train, y_train, label_enc)

    print("   Training complete.")
    return xgb_model, lgb_model, label_enc, emb_index

# ================================================================================
# CELL 7 — INFERENCE ENGINE
#
# Classifies invoices that were NOT matched by any rule.
# Uses a 4-gate fallback chain — first gate that fires wins:
#
#   Gate 1 → Both models agree AND ensemble confidence >= 0.60   → label_source = "ml"
#   Gate 2 → One model is very confident >= 0.80                 → label_source = "ml"
#   Gate 3 → k-NN similarity search: >= 70% of neighbors agree  → label_source = "ml"
#   Gate 4 → All gates failed                                    → label_source = "need_to_review"
# ================================================================================

def _similarity_lookup(query_embedding, emb_index, k, threshold):
    """
    Find the k most similar training records using cosine similarity.
    Returns: (predicted_label, vote_fraction, accepted: bool)
    """
    norm = np.linalg.norm(query_embedding)
    if norm == 0:
        return None, 0.0, False

    # Dot product with L2-normalized index = cosine similarity
    similarities  = emb_index["embeddings"] @ (query_embedding / norm)
    top_k_indices = np.argpartition(similarities, -k)[-k:]
    top_k_labels  = emb_index["labels"][top_k_indices]

    unique_labels, vote_counts = np.unique(top_k_labels, return_counts=True)
    best_idx   = np.argmax(vote_counts)
    best_label = unique_labels[best_idx]
    vote_frac  = vote_counts[best_idx] / k

    return best_label, float(vote_frac), (vote_frac >= threshold)


def _run_fallback_chain(xgb_preds, xgb_confs, lgb_preds, lgb_confs,
                        ens_preds, ens_confs, models_agree, text_embeddings, emb_index):
    """
    Walk through the 4 decision gates for every invoice row.
    Returns dicts of: final_pred, final_conf, decision_path, label_source
    """
    t = THRESHOLDS
    final_preds, final_confs, decision_paths, label_sources = [], [], [], []

    for i in range(len(xgb_preds)):

        # ── Gate 1: Both models agree + ensemble is confident ───────────────────
        if models_agree[i] and ens_confs[i] >= t["ensemble_accept"]:
            final_preds.append(ens_preds[i])
            final_confs.append(float(ens_confs[i]))
            decision_paths.append("ensemble")
            label_sources.append("ml")
            continue

        # ── Gate 2: One model is highly dominant ─────────────────────────────────
        xgb_dominant = xgb_confs[i] >= t["dominant_model"]
        lgb_dominant = lgb_confs[i] >= t["dominant_model"]

        if xgb_dominant and not lgb_dominant:
            final_preds.append(xgb_preds[i]); final_confs.append(float(xgb_confs[i]))
            decision_paths.append("xgb_dominant"); label_sources.append("ml"); continue

        if lgb_dominant and not xgb_dominant:
            final_preds.append(lgb_preds[i]); final_confs.append(float(lgb_confs[i]))
            decision_paths.append("lgb_dominant"); label_sources.append("ml"); continue

        if xgb_dominant and lgb_dominant:
            # Both dominant but disagree — take whichever is more confident
            if xgb_confs[i] >= lgb_confs[i]:
                final_preds.append(xgb_preds[i]); final_confs.append(float(xgb_confs[i]))
                decision_paths.append("xgb_dominant_tiebreak")
            else:
                final_preds.append(lgb_preds[i]); final_confs.append(float(lgb_confs[i]))
                decision_paths.append("lgb_dominant_tiebreak")
            label_sources.append("ml"); continue

        # ── Gate 3: k-NN similarity search ───────────────────────────────────────
        sim_label, sim_conf, sim_accepted = _similarity_lookup(
            text_embeddings[i], emb_index,
            k=t["similarity_k"], threshold=t["similarity_accept"],
        )
        if sim_accepted:
            final_preds.append(sim_label); final_confs.append(sim_conf)
            decision_paths.append("similarity"); label_sources.append("ml"); continue

        # ── Gate 4: All gates failed → needs human review ────────────────────────
        # Store best ML guess as a reference (not used as final classification)
        final_preds.append(ens_preds[i]); final_confs.append(float(ens_confs[i]))
        decision_paths.append("need_to_review"); label_sources.append("need_to_review")

    return {
        "final_pred":    final_preds,
        "final_conf":    final_confs,
        "decision_path": decision_paths,
        "label_source":  label_sources,
    }


def classify_invoices(df, xgb_model, lgb_booster, label_enc, emb_index, spark):
    """
    Classify all invoices:
    - Rule-matched invoices:   pass through with final_pred = rule label, conf = 1.0
    - No-rule invoices:        run through ML ensemble + fallback chain
    """
    id_col = COLS["invoice_id"]

    df_rule    = df.filter(col("label_source") == "rule")
    df_no_rule = df.filter(col("label_source") == "no_rule")
    print(f"   Rule-matched: {df_rule.count():,}  |  Going to ML: {df_no_rule.count():,}")

    # ── Tag rule-matched rows with pass-through prediction columns ───────────────
    df_rule = (
        df_rule
        .withColumn("xgb_pred",     lit(None).cast(StringType()))
        .withColumn("xgb_conf",     lit(None).cast(FloatType()))
        .withColumn("lgb_pred",     lit(None).cast(StringType()))
        .withColumn("lgb_conf",     lit(None).cast(FloatType()))
        .withColumn("ensemble_pred",lit(None).cast(StringType()))
        .withColumn("ensemble_conf",lit(None).cast(FloatType()))
        .withColumn("models_agree", lit(None).cast(BooleanType()))
        .withColumn("final_pred",   col("label_account_code"))  # Already labeled by rule
        .withColumn("final_conf",   lit(1.0).cast(FloatType())) # Rules are 100% confident
        .withColumn("decision_path",lit("rule"))
    )

    if df_no_rule.count() == 0:
        print("   All invoices matched by rules — skipping ML.")
        return df_rule

    # ── Pull no-rule invoices to Pandas for scoring ──────────────────────────────
    pdf = (
        df_no_rule
        .withColumn("vec_array", vector_to_array(col("combined_vector")))
        .select(id_col, "vec_array", "text_embedding")
        .toPandas()
    )
    # Replace any nulls with zero vectors so no record is silently dropped
    dim = FEATURE_CONFIG["embedding_dim"]
    pdf["vec_array"]      = pdf["vec_array"].apply(
        lambda v: v if v is not None else [0.0] * dim)
    pdf["text_embedding"] = pdf["text_embedding"].apply(
        lambda v: v if v is not None else [0.0] * dim)

    X          = np.array(pdf["vec_array"].tolist(),      dtype=np.float32)
    embeddings = np.array(pdf["text_embedding"].tolist(), dtype=np.float32)

    # ── Score with both models ───────────────────────────────────────────────────
    xgb_proba  = xgb_model.predict_proba(X)
    xgb_idx    = np.argmax(xgb_proba, axis=1)
    xgb_conf   = xgb_proba[np.arange(len(xgb_proba)), xgb_idx]
    xgb_pred   = label_enc.inverse_transform(xgb_idx)

    lgb_proba  = lgb_booster.predict(X)
    lgb_idx    = np.argmax(lgb_proba, axis=1)
    lgb_conf   = lgb_proba[np.arange(len(lgb_proba)), lgb_idx]
    lgb_pred   = label_enc.inverse_transform(lgb_idx)

    w_xgb      = MODEL_CONFIG["ensemble_weights"]["xgb"]
    w_lgb      = MODEL_CONFIG["ensemble_weights"]["lgb"]
    ens_proba  = w_xgb * xgb_proba + w_lgb * lgb_proba
    ens_idx    = np.argmax(ens_proba, axis=1)
    ens_conf   = ens_proba[np.arange(len(ens_proba)), ens_idx]
    ens_pred   = label_enc.inverse_transform(ens_idx)
    models_agree = (xgb_pred == lgb_pred)

    # ── Run 4-gate fallback chain ────────────────────────────────────────────────
    results = _run_fallback_chain(
        xgb_pred, xgb_conf, lgb_pred, lgb_conf,
        ens_pred, ens_conf, models_agree, embeddings, emb_index,
    )

    # ── Push predictions back into Spark ────────────────────────────────────────
    pdf_out = pd.DataFrame({
        id_col:           pdf[id_col].values,
        "xgb_pred":       xgb_pred,
        "xgb_conf":       xgb_conf.astype(float),
        "lgb_pred":       lgb_pred,
        "lgb_conf":       lgb_conf.astype(float),
        "ensemble_pred":  ens_pred,
        "ensemble_conf":  ens_conf.astype(float),
        "models_agree":   models_agree,
        "final_pred":     results["final_pred"],
        "final_conf":     results["final_conf"],
        "decision_path":  results["decision_path"],
        "label_source":   results["label_source"],
    })

    schema = StructType([
        StructField(id_col,           df.schema[id_col].dataType, True),
        StructField("xgb_pred",       StringType(),  True),
        StructField("xgb_conf",       FloatType(),   True),
        StructField("lgb_pred",       StringType(),  True),
        StructField("lgb_conf",       FloatType(),   True),
        StructField("ensemble_pred",  StringType(),  True),
        StructField("ensemble_conf",  FloatType(),   True),
        StructField("models_agree",   BooleanType(), True),
        StructField("final_pred",     StringType(),  True),
        StructField("final_conf",     FloatType(),   True),
        StructField("decision_path",  StringType(),  True),
        StructField("label_source",   StringType(),  True),
    ])

    preds_spark = spark.createDataFrame(pdf_out, schema=schema)

    # Drop original label_source ("no_rule") from df_no_rule before joining,
    # because preds_spark has the correct updated value from the fallback chain
    df_no_rule  = df_no_rule.drop("label_source").join(preds_spark, on=id_col, how="left")
    df_final    = df_rule.unionByName(df_no_rule, allowMissingColumns=True)

    # ── Print classification summary ─────────────────────────────────────────────
    summary = (
        df_final
        .groupBy("label_source", "decision_path")
        .agg(count("*").alias("n"))
        .orderBy("n", ascending=False)
        .toPandas()
    )
    total = summary["n"].sum()
    print(f"\n   {'Source':<22} {'Decision Path':<25} {'Count':>8}  {'Pct':>6}")
    print(f"   {'-'*22} {'-'*25} {'-'*8}  {'-'*6}")
    for _, r in summary.iterrows():
        pct = r["n"] / total * 100
        print(f"   {str(r['label_source']):<22} {str(r['decision_path']):<25} {r['n']:>8,}  {pct:>5.1f}%")
    print(f"   {'TOTAL':<22} {'':<25} {total:>8,}  100.0%")

    return df_final

print("✓ Inference engine OK")


# ================================================================================
# CELL 8 — CLUSTER DISCOVERY
#
# Groups "need_to_review" invoices by semantic similarity using KMeans.
# Helps identify new invoice patterns that don't match any existing rule or class.
#
# Output: discovery_cluster_id column added to df
#         + a cluster_summary table for the Excel report
# ================================================================================

def discover_new_clusters(df, spark):
    """
    Run KMeans on GTE embeddings of need_to_review invoices.
    Auto-selects best k via silhouette score (tries k=2 up to max_clusters).
    """
    id_col = COLS["invoice_id"]
    df_ntr = df.filter(col("label_source") == "need_to_review")
    ntr_n  = df_ntr.count()
    print(f"   Need-to-review records: {ntr_n:,}")

    # Need at least min_cluster_size records to be worth clustering
    if ntr_n < DISCOVERY_CONFIG["min_cluster_size"]:
        print("   Too few records to cluster — skipping discovery.")
        return df.withColumn("discovery_cluster_id", lit(None).cast("int")), pd.DataFrame()

    # Pull to Pandas for sklearn KMeans
    ctx_cols = [id_col, "final_description",
                COLS["vendor_name"], COLS["extc"], COLS["mic"],
                COLS["state"], COLS["tran_amount"], "text_embedding"]
    available_cols = [c for c in ctx_cols if c in df_ntr.columns]
    pdf = df_ntr.select(available_cols).dropna(subset=["text_embedding"]).toPandas()
    X   = np.array(pdf["text_embedding"].tolist(), dtype=np.float32)

    # ── Auto-select best k ───────────────────────────────────────────────────────
    max_k      = min(DISCOVERY_CONFIG["max_clusters"], len(X) // 4, len(X) - 1)
    max_k      = max(max_k, 2)
    best_k     = 2
    best_score = -1.0

    for k in range(2, max_k + 1):
        labels = KMeans(n_clusters=k, random_state=42, n_init=5).fit_predict(X)
        score  = silhouette_score(X, labels, sample_size=min(500, len(X)))
        if score > best_score:
            best_k, best_score = k, score

    print(f"   Best k = {best_k}  (silhouette score = {best_score:.3f})")

    # ── Final KMeans with best k ─────────────────────────────────────────────────
    pdf["discovery_cluster_id"] = KMeans(
        n_clusters=best_k,
        random_state=DISCOVERY_CONFIG["random_seed"],
        n_init=10,
    ).fit_predict(X)

    # Discard clusters that are too small (noise)
    cluster_sizes  = pdf["discovery_cluster_id"].value_counts()
    valid_clusters = cluster_sizes[cluster_sizes >= DISCOVERY_CONFIG["min_cluster_size"]].index
    pdf["discovery_cluster_id"] = pdf["discovery_cluster_id"].apply(
        lambda c: int(c) if c in valid_clusters else -1
    )

    # ── Build cluster summary for the report ─────────────────────────────────────
    n_samples = DISCOVERY_CONFIG["samples_per_cluster"]
    rows = []
    for cid in sorted(valid_clusters):
        sub = pdf[pdf["discovery_cluster_id"] == cid]
        rows.append({
            "cluster_id":            int(cid),
            "record_count":          len(sub),
            "proposed_cluster_name": "",   # ← User fills this in the Excel report
            "proposed_account_code": "",   # ← User fills this in the Excel report
            "top_vendors":           ", ".join(sub[COLS["vendor_name"]].value_counts().head(3).index.tolist()),
            "top_extc_codes":        ", ".join(sub[COLS["extc"]].value_counts().head(3).index.tolist()),
            "top_states":            ", ".join(sub[COLS["state"]].value_counts().head(3).index.tolist()),
            "avg_transaction_amount": round(sub[COLS["tran_amount"]].mean(), 2),
            "sample_descriptions":   " | ".join(sub["final_description"].dropna().head(n_samples).tolist()),
        })
    cluster_summary = pd.DataFrame(rows)

    # ── Push cluster IDs back to Spark and join to main df ──────────────────────
    cluster_id_spark = spark.createDataFrame(pdf[[id_col, "discovery_cluster_id"]])
    df = df.join(cluster_id_spark, on=id_col, how="left")

    print(f"   Valid clusters discovered: {len(valid_clusters)}")
    return df, cluster_summary

print("✓ Cluster discovery OK")


# ================================================================================
# CELL 9 — REPORT GENERATOR
#
# Generates an Excel report for the business team with 3 sheets:
#   Sheet 1 — Executive Summary    : high-level classification counts
#   Sheet 2 — Discovered Clusters  : new patterns found (user fills in names)
#   Sheet 3 — Raw NTR Records      : every need_to_review invoice with ML scores
#
# Saves to driver /tmp (always), and also copies to ADLS if reports_output is set.
# ================================================================================

def generate_report(df, cluster_summary, report_month):
    timestamp    = datetime.now().strftime("%Y%m%d_%H%M")
    local_file   = f"/tmp/NeedToReview_{report_month}_{timestamp}.xlsx"

    # ── Build Sheet 1: Executive Summary ────────────────────────────────────────
    total  = df.count()
    counts = df.groupBy("label_source").agg(count("*").alias("n")).toPandas()

    def _n(src):
        rows = counts[counts["label_source"] == src]["n"]
        return int(rows.sum()) if len(rows) > 0 else 0

    rule_n  = _n("rule")
    ml_n    = _n("ml")
    ntr_n   = _n("need_to_review")
    avg_conf_row = df.filter(col("label_source") == "ml").agg(avg("final_conf")).collect()[0][0]
    avg_conf = round(float(avg_conf_row), 4) if avg_conf_row else 0.0

    summary_df = pd.DataFrame([
        {"Metric": "Total Invoices Processed",          "Value": total,          "Pct": "100%"},
        {"Metric": "Classified by Rules",               "Value": rule_n,         "Pct": f"{rule_n/total*100:.1f}%"},
        {"Metric": "Classified by ML",                  "Value": ml_n,           "Pct": f"{ml_n/total*100:.1f}%"},
        {"Metric": "Need Human Review",                 "Value": ntr_n,          "Pct": f"{ntr_n/total*100:.1f}%"},
        {"Metric": "Total Auto-classified",             "Value": rule_n + ml_n,  "Pct": f"{(rule_n+ml_n)/total*100:.1f}%"},
        {"Metric": "Average ML Confidence (ML records)","Value": avg_conf,       "Pct": ""},
    ])

    # ── Build Sheet 3: Raw NTR invoice records ───────────────────────────────────
    desired_cols = [
        COLS["invoice_id"], COLS["invoice_num"], "final_description",
        COLS["vendor_name"], COLS["extc"], COLS["mic"], COLS["state"],
        COLS["tran_amount"], COLS["account_code"],
        "xgb_pred", "xgb_conf", "lgb_pred", "lgb_conf",
        "ensemble_pred", "ensemble_conf",
        "final_pred", "final_conf", "decision_path", "discovery_cluster_id",
    ]
    existing_cols = [c for c in desired_cols if c in df.columns]
    raw_pdf = (
        df.filter(col("label_source") == "need_to_review")
          .select(existing_cols)
          .orderBy("discovery_cluster_id")
          .toPandas()
    )

    # ── Write Excel ──────────────────────────────────────────────────────────────
    with pd.ExcelWriter(local_file, engine="openpyxl") as writer:

        summary_df.to_excel(writer, sheet_name="Executive Summary", index=False)

        if not cluster_summary.empty:
            cluster_summary.to_excel(writer, sheet_name="Discovered Clusters", index=False)
            # Highlight the user-fill columns in yellow
            try:
                from openpyxl.styles import PatternFill
                ws      = writer.sheets["Discovered Clusters"]
                yellow  = PatternFill(start_color="FFFF00", end_color="FFFF00", fill_type="solid")
                headers = [cell.value for cell in ws[1]]
                for fill_col in ["proposed_cluster_name", "proposed_account_code"]:
                    if fill_col in headers:
                        col_idx = headers.index(fill_col) + 1
                        for row in ws.iter_rows(min_row=2, min_col=col_idx, max_col=col_idx):
                            for cell in row:
                                cell.fill = yellow
            except Exception:
                pass  # Yellow highlighting is cosmetic — don't fail the report for it
        else:
            pd.DataFrame({"message": ["No new clusters discovered this month."]}) \
              .to_excel(writer, sheet_name="Discovered Clusters", index=False)

        raw_pdf.to_excel(writer, sheet_name="Raw NTR Records", index=False)

    # ── Copy to ADLS if path is configured ──────────────────────────────────────
    if PATHS.get("reports_output"):
        adls_path = f"{PATHS['reports_output']}NeedToReview_{report_month}_{timestamp}.xlsx"
        dbutils.fs.cp(f"file:{local_file}", adls_path, recurse=False)
        print(f"   Report saved to ADLS:  {adls_path}")
        final_path = adls_path
    else:
        print(f"   Report saved to driver: {local_file}")
        print(f"   To download: Databricks UI → Compute → your cluster → Driver Logs")
        final_path = local_file

    print(f"   Auto-classified: {rule_n + ml_n:,} / {total:,} ({(rule_n+ml_n)/total*100:.1f}%)")
    print(f"   Need review:     {ntr_n:,} / {total:,} ({ntr_n/total*100:.1f}%)")
    return final_path

print("✓ Report generator OK")

def write_staging(df: DataFrame, table: str, label: str) -> int:
    """Write a DataFrame to a staging Delta table if WRITE_STAGING_TABLES is True.
    Always returns the row count (triggers an action regardless)."""
    count = df.count()
    if WRITE_STAGING_TABLES:
        (
            df.write
            .format("delta")
            .mode("overwrite")
            .option("overwriteSchema", "true")
            .saveAsTable(table)
        )
        logger.info(f"[{label}] Written → {table}  ({count:,} rows)")
    else:
        logger.info(f"[{label}] Staging write skipped ({count:,} rows in memory)")
    return count

def get_vsc() -> VectorSearchClient:
    global _vsc_client

def vs_batch_search(
    keys: list[tuple],          # list of (invoice_id, inv_line_number)
    embeddings: list[list[float]],
    index_name: str,
    columns: list[str],
    num_results: int,
    threshold: float = 0.0,
) -> dict[tuple, list[dict]]:
    """
    Run batch similarity search against a Databricks Vector Search index.
    Returns {(invoice_id, inv_line_number): [hit_dict, ...]} for hits above threshold.
    Each hit_dict maps column name → value using the `columns` list.
    """
    index = get_vsc().get_index(endpoint_name=VS_ENDPOINT, index_name=index_name)
    results: dict[tuple, list[dict]] = {}

    for start in range(0, len(embeddings), VS_BATCH_SIZE):
        batch_keys = keys[start : start + VS_BATCH_SIZE]
        batch_embs = embeddings[start : start + VS_BATCH_SIZE]

        for key, emb in zip(batch_keys, batch_embs):
            try:
                resp = index.similarity_search(
                    query_vector=emb,
                    columns=columns,
                    num_results=num_results,
                )
                hits = resp.get("result", {}).get("data_array", [])
                parsed = []
                for hit in hits:
                    score = float(hit[-1]) if hit else 0.0
                    if score >= threshold:
                        parsed.append(dict(zip(columns, hit)))
                if parsed:
                    results[key] = parsed
            except Exception as exc:
                logger.warning(f"VS query error for key {key}: {exc}")

    return results

# 1.3  Attach VS results to a DataFrame via broadcast + UDFs
# ─────────────────────────────────────────────────────────────────────────────
def attach_vs_results(
    df: DataFrame,
    vs_results: dict,
    new_cols: dict[str, tuple],   # {output_col_name: (lookup_key, dtype, default)}
) -> tuple[DataFrame, object]:
    """
    Attach values from a VS result lookup dict to a DataFrame using broadcast UDFs.
    `new_cols` maps output column name → (result_dict_key, Spark type, default_value).

    Returns (df, broadcast_handle). Caller MUST call broadcast_handle.unpersist()
    AFTER the returned DataFrame has been fully materialised (i.e. after .count(),
    .cache(), or a Delta write). Unpersisting here would destroy the broadcast
    before Spark's lazy plan executes the UDFs.
    """
    bc = sc.broadcast(vs_results)

    for col_name, (dict_key, dtype, default) in new_cols.items():
        def _make_udf(dk, dv, broadcast_ref=bc):
            def _udf(iid, ln):
                hits = broadcast_ref.value.get((iid, ln))
                if hits:
                    return hits[0].get(dk, dv)
                return dv
            return F.udf(_udf, dtype)

        df = df.withColumn(col_name, _make_udf(dict_key, default)(
            F.col("invoice_id"), F.col("inv_line_number")
        ))

    # Do NOT unpersist here — return the handle so the caller can do it post-materialise
    return df, bc

# ─────────────────────────────────────────────────────────────────────────────
_deploy_client = None

def get_deploy_client_cached():
    global _deploy_client
    if _deploy_client is None:
        _deploy_client = get_deploy_client("databricks")
    return _deploy_client


def xgboost_batch_predict(
    df: DataFrame,
    endpoint: str = XGBOOST_ENDPOINT,
    batch_size: int = XGBOOST_BATCH_SIZE,
) -> tuple[dict, dict]:
    """
    Run batch inference against the XGBoost model serving endpoint.
    Returns (preds_dict, confs_dict) keyed by (invoice_id, inv_line_number).
    """
    client = get_deploy_client_cached()
    rows   = df.select("invoice_id", "inv_line_number", "gte_embedding").collect()
    preds, confs = {}, {}

    for start in range(0, len(rows), batch_size):
        batch = rows[start : start + batch_size]
        embs  = [r["gte_embedding"] for r in batch]
        try:
            resp        = client.predict(endpoint=endpoint, inputs={"inputs": embs})
            predictions = resp.get("predictions", [])
            for row, pred in zip(batch, predictions):
                key = (row["invoice_id"], row["inv_line_number"])
                # Expected response: {"category": "...", "probability": 0.xx}
                # ↑ Adjust if your endpoint returns raw indices or a different schema
                if isinstance(pred, dict):
                    preds[key] = pred.get("category", "Unknown")
                    confs[key] = float(pred.get("probability", 0.0))
                elif isinstance(pred, (list, tuple)) and len(pred) >= 2:
                    preds[key] = str(pred[0])
                    confs[key] = float(pred[1])
                else:
                    preds[key] = str(pred)
                    confs[key] = 0.5
        except Exception as exc:
            logger.error(f"XGBoost batch [{start}:{start+batch_size}] failed: {exc}")
            for row in batch:
                key = (row["invoice_id"], row["inv_line_number"])
                preds[key] = "Unknown"
                confs[key] = 0.0

    return preds, confs


def attach_xgboost_preds(df: DataFrame) -> DataFrame:
    preds, confs = xgboost_batch_predict(df)
    preds_bc = sc.broadcast(preds)
    confs_bc = sc.broadcast(confs)

    udf_pred = F.udf(lambda iid, ln: preds_bc.value.get((iid, ln), "Unknown"), StringType())
    udf_conf = F.udf(lambda iid, ln: float(confs_bc.value.get((iid, ln), 0.0)), FloatType())

    result = (
        df
        .withColumn("xgboost_pred", udf_pred(F.col("invoice_id"), F.col("inv_line_number")))
        .withColumn("xgboost_conf", udf_conf(F.col("invoice_id"), F.col("inv_line_number")))
    )
    preds_bc.unpersist()
    confs_bc.unpersist()
    return result

# 1.5  Tax lookup + Use Tax (shared between Step 3 and Re-route)
# ─────────────────────────────────────────────────────────────────────────────
# ─────────────────────────────────────────────────────────────────────────────
# 1.5  Tax matrix — pre-loaded once, reused by apply_tax_lookup (Steps 3.3 + 3.6)
# ─────────────────────────────────────────────────────────────────────────────
# Loaded here (not inside the function) so it is read from disk exactly once.
# dropDuplicates on (cluster, state) is critical: any duplicate rows would cause
# a LEFT JOIN fan-out that increases row count and breaks the Step 3.7 quality gate.
_TAX_MATRIX_BC = F.broadcast(
    spark.table(TAX_MATRIX_TABLE)
    .select(
        F.col("cluster").alias("_tm_cluster"),
        F.col("state").alias("_tm_state"),
        "taxability",
    )
    .dropDuplicates(["_tm_cluster", "_tm_state"])
)


def apply_tax_lookup(df: DataFrame) -> DataFrame:
    """LEFT JOIN taxability matrix, then derive Reverse_UseTax.
    Uses _TAX_MATRIX_BC which is loaded, deduplicated, and broadcast once at startup."""
    df = (
        df
        .join(
            _TAX_MATRIX_BC,
            (df["final_category"] == _TAX_MATRIX_BC["_tm_cluster"]) &
            (df["state"]          == _TAX_MATRIX_BC["_tm_state"]),
            how="left",
        )
        .drop("_tm_cluster", "_tm_state")
        .withColumn("tax_source",
            F.when(F.col("taxability").isNotNull(), F.lit("matrix"))
             .otherwise(F.lit("no_match"))
        )
    )
    # ── Task 3.3b: Use Tax reversal logic ─────────────────────────────────────
    # ⚠️  OPEN QUESTION: NULL use_tax_amount — confirm with client whether
    #     NULL means "no use tax assessed" or "data gap". Current treatment: NULL → deferred (null output).
    df = df.withColumn("Reverse_UseTax",
        F.when(
            (F.col("use_tax_amount") > 0) & (F.col("taxability") == "Exempt"),
            F.lit("Yes"),    # use tax was paid but item is exempt → reversal required
        ).when(
            (F.col("use_tax_amount") > 0) & (F.col("taxability") == "Taxable"),
            F.lit("No"),     # use tax paid on taxable item → correct, no reversal
        ).when(
            F.col("use_tax_amount") == 0,
            F.lit("No"),     # no use tax → no reversal
        ).when(
            F.col("taxability").isNull(),
            F.lit(None).cast(StringType()),   # taxability unknown → cannot determine
        ).otherwise(
            F.lit(None).cast(StringType()),   # use_tax_amount IS NULL → deferred
        )
    )
    return df

def parse_short_justification(raw: str) -> str:
    if not raw:
        return ""
    for line in raw.splitlines():
        if line.startswith("SHORT_JUSTIFICATION:"):
            return line.replace("SHORT_JUSTIFICATION:", "").strip()
    return raw.splitlines()[0].strip()[:300]


def parse_full_writeup(raw: str) -> str:
    if not raw:
        return ""
    if "FULL_WRITEUP:" in raw:
        return raw.split("FULL_WRITEUP:", 1)[1].strip()
    lines = raw.splitlines()
    return "\n".join(lines[1:]).strip() if len(lines) > 1 else raw


udf_parse_short = F.udf(parse_short_justification, StringType())
udf_parse_full  = F.udf(parse_full_writeup,        StringType())


def apply_tax_writer(df: DataFrame, temp_view_suffix: str = "") -> DataFrame:
    """Generate audit write-ups via ai_query(). RAG examples fetched when called from Step 4."""
    view = f"writeup_input{temp_view_suffix}"
    df.createOrReplaceTempView(view)

    df = spark.sql(f"""
        SELECT
            *,
            ai_query(
                '{LLM_ENDPOINT}',
                CONCAT(
                    'You are an AT&T AP Tax Analyst producing AUDIT-READY taxability justifications.',
                    ' Be specific, cite the category and state, reference applicable tax rule.',
                    ' Do NOT use generic language.',
                    CHR(10), CHR(10),
                    '=== INVOICE ===', CHR(10),
                    'Description : ', final_description, CHR(10),
                    'Vendor       : ', vendor_name, CHR(10),
                    'Category     : ', COALESCE(final_category, 'UNKNOWN'), CHR(10),
                    'State        : ', COALESCE(state, 'UNKNOWN'), CHR(10),
                    'Amount       : $', CAST(ROUND(amount, 2) AS STRING), CHR(10),
                    'Taxability   : ', COALESCE(taxability, 'UNKNOWN'), CHR(10),
                    'Reverse UTax : ', COALESCE(Reverse_UseTax, 'N/A'), CHR(10),
                    IF(Reverse_UseTax = 'Yes',
                       CONCAT('Use Tax Amt  : $', CAST(ROUND(use_tax_amount, 2) AS STRING), CHR(10)),
                       ''
                    ),
                    COALESCE(
                        IF(prior_writeup_examples IS NOT NULL AND LENGTH(prior_writeup_examples) > 0,
                           CONCAT(CHR(10), '=== SIMILAR PRIOR WRITE-UPS ===', CHR(10), prior_writeup_examples, CHR(10)),
                           ''
                        ),
                        ''
                    ),
                    CHR(10),
                    '=== INSTRUCTIONS ===', CHR(10),
                    'Respond in EXACTLY this format (no preamble):',   CHR(10),
                    'SHORT_JUSTIFICATION: <one sentence, ≤35 words>',   CHR(10),
                    'FULL_WRITEUP: <2-3 paragraphs, ≤200 words.',
                    IF(Reverse_UseTax = 'Yes',
                       CONCAT(' State use tax reversal of $', CAST(ROUND(use_tax_amount,2) AS STRING), ' is required.'),
                       ''
                    ),
                    '>'
                )
            ) AS _writeup_raw
        FROM {view}
    """)

    df = (
        df
        .withColumn("short_justification", udf_parse_short(F.col("_writeup_raw")))
        .withColumn("full_writeup",        udf_parse_full(F.col("_writeup_raw")))
        .withColumn("reviewer_note",
            F.when(
                F.col("taxability").isNull(),
                F.lit("⚠️ No taxability matrix match — manual determination required."),
            ).when(
                F.col("Reverse_UseTax") == "Yes",
                F.concat(
                    F.lit("⚠️ Use tax reversal required: $"),
                    F.round(F.col("use_tax_amount"), 2).cast(StringType()),
                ),
            ).otherwise(F.lit(None).cast(StringType()))
        )
        .withColumn("audit_ready",
            (
                F.col("taxability").isNotNull()
                & F.col("short_justification").isNotNull()
                & (F.length(F.col("short_justification")) > 5)
            ).cast(BooleanType())
        )
        .drop("_writeup_raw")
    )
    return df

# 1.7  QA scoring (shared between Step 5 and Re-route lite)
# ─────────────────────────────────────────────────────────────────────────────
def assign_priority_and_flags(df: DataFrame) -> DataFrame:
    """
    Given flag columns (flag_*), compute anomaly_score, review_priority,
    override_suggested, anomalies array, and recommendation.
    Requires flag_vendor_consistency, flag_amount_zscore, flag_state_history,
    flag_rule_conflict, flag_taxability_sanity, flag_batch_outlier to exist.
    """
    w = QA_WEIGHTS
    df = df.withColumn("anomaly_score",
        (F.col("flag_vendor_consistency").cast(FloatType())    * w["vendor_consistency"])
        + (F.col("flag_amount_zscore").cast(FloatType())       * w["amount_zscore"])
        + (F.col("flag_state_history").cast(FloatType())       * w["state_history"])
        + (F.col("flag_rule_conflict").cast(FloatType())       * w["rule_classifier_conflict"])
        + (F.col("flag_taxability_sanity").cast(FloatType())   * w["taxability_sanity"])
        + (F.col("flag_batch_outlier").cast(FloatType())       * w["batch_outlier"])
    )
    # Null taxability always escalates to at least HIGH
    df = df.withColumn("anomaly_score",
        F.when(
            F.col("taxability").isNull(),
            F.greatest(F.col("anomaly_score"), F.lit(PRIORITY_HIGH_THRESHOLD))
        ).otherwise(F.col("anomaly_score"))
    )
    df = (
        df
        .withColumn("review_priority",
            F.when(F.col("anomaly_score") >= PRIORITY_CRITICAL_THRESHOLD, F.lit("CRITICAL"))
             .when(F.col("anomaly_score") >= PRIORITY_HIGH_THRESHOLD,     F.lit("HIGH"))
             .when(F.col("anomaly_score") >= PRIORITY_MED_THRESHOLD,      F.lit("MED"))
             .otherwise(F.lit("LOW"))
        )
        .withColumn("override_suggested",
            F.col("review_priority").isin("HIGH", "CRITICAL")
        )
        .withColumn("anomalies",
            F.array_compact(F.array(
                F.when(F.col("flag_vendor_consistency"),   F.lit("vendor_category_mismatch")),
                F.when(F.col("flag_amount_zscore"),        F.lit("amount_outlier")),
                F.when(F.col("flag_state_history"),        F.lit("new_state_category_combo")),
                F.when(F.col("flag_rule_conflict"),        F.lit("rule_classifier_conflict")),
                F.when(F.col("flag_taxability_sanity"),    F.lit("taxability_sanity_fail")),
                F.when(F.col("flag_batch_outlier"),        F.lit("batch_vendor_outlier")),
                F.when(F.col("taxability").isNull(),       F.lit("no_taxability_match")),
            ))
        )
        .withColumn("recommendation",
            F.when(F.col("flag_rule_conflict"),
                   F.lit("Review: rule and ML model disagree on category."))
             .when(F.col("flag_taxability_sanity"),
                   F.lit("Review: taxability assignment contradicts state tax rule."))
             .when(F.col("taxability").isNull(),
                   F.lit("Escalate: no matrix match. Manual determination required."))
             .when(F.col("flag_amount_zscore"),
                   F.lit("Review: invoice amount is a statistical outlier for this category."))
             .when(F.col("flag_vendor_consistency"),
                   F.lit("Review: category differs from this vendor's historical pattern."))
             .when(F.col("flag_batch_outlier"),
                   F.lit("Review: rare category assignment for this vendor this month."))
             .when(F.col("flag_state_history"),
                   F.lit("Review: category-state combination not seen in prior 12 months."))
             .otherwise(F.lit(None).cast(StringType()))
        )
    )
    return df

    qa_df
    .withColumn("_cat_mean", F.mean("amount").over(cat_win))
    .withColumn("_cat_std",  F.stddev("amount").over(cat_win))
    .withColumn("_zscore",
        F.when(
            F.col("_cat_std").isNotNull() & (F.col("_cat_std") > 0),
            F.abs((F.col("amount") - F.col("_cat_mean")) / F.col("_cat_std"))
        ).otherwise(F.lit(0.0))
    )
    .withColumn("flag_amount_zscore", F.col("_zscore") > AMOUNT_ZSCORE_THRESHOLD)
    .drop("_cat_mean", "_cat_std", "_zscore")
)
c2 = qa_df.filter(F.col("flag_amount_zscore")).count()
logger.info(f"  Check 2 flags: {c2:,}")

# ── Check 3: State-category history ──────────────────────────────────────────
logger.info("QA Check 3: State-category history...")
try:
    state_cat_seen = F.broadcast(
        spark.table(HISTORICAL_TABLE)
        .select(
            F.col("category").alias("_hcat"),
            F.col("state").alias("_hstate"),
        )
        .distinct()
    )
    qa_df = (
        qa_df
        .join(
            state_cat_seen,
            (qa_df["final_category"] == state_cat_seen["_hcat"]) &
            (qa_df["state"]          == state_cat_seen["_hstate"]),
            how="left",
        )
        .withColumn("flag_state_history", F.col("_hcat").isNull())
        .drop("_hcat", "_hstate")
    )
except Exception as e:
    logger.warning(f"QA Check 3 skipped: {e}")
    qa_df = qa_df.withColumn("flag_state_history", F.lit(False).cast(BooleanType()))
c3 = qa_df.filter(F.col("flag_state_history")).count()
logger.info(f"  Check 3 flags: {c3:,}")

# ── Check 4: Rule vs classifier conflict ──────────────────────────────────────
logger.info("QA Check 4: Rule vs classifier conflict...")
# Rules-path records have xgboost_pred = NULL (classifier never ran on them).
# Without the isNotNull guard, NULL != final_category → NULL, not False,
# so the flag would silently be NULL instead of False for those records.
qa_df = qa_df.withColumn("flag_rule_conflict",
    (F.col("rule_match") == True) &
    F.col("xgboost_pred").isNotNull() &
    (F.col("final_category") != F.col("xgboost_pred"))
)
c4 = qa_df.filter(F.col("flag_rule_conflict")).count()
logger.info(f"  Check 4 flags: {c4:,}")

# ── Check 5: Taxability sanity ────────────────────────────────────────────────
logger.info("QA Check 5: Taxability sanity vs state tax rules...")
try:
    state_rules = F.broadcast(
        spark.table(STATE_TAX_TABLE)
        .select(
            F.col("state").alias("_sr_state"),
            F.col("taxable").alias("_sr_taxable"),
        )
        .distinct()
    )
    qa_df = (
        qa_df
        .join(state_rules, qa_df["state"] == state_rules["_sr_state"], how="left")
        .withColumn("flag_taxability_sanity",
            (F.col("taxability") == "Taxable") &
            (F.col("_sr_taxable") == False) &
            F.col("_sr_state").isNotNull()
        )
        .drop("_sr_state", "_sr_taxable")
    )
except Exception as e:
    logger.warning(f"QA Check 5 skipped: {e}")
    qa_df = qa_df.withColumn("flag_taxability_sanity", F.lit(False).cast(BooleanType()))
c5 = qa_df.filter(F.col("flag_taxability_sanity")).count()
logger.info(f"  Check 5 flags: {c5:,}")

# ── Check 6: Batch-level outlier ──────────────────────────────────────────────
logger.info("QA Check 6: Batch-level vendor-category outlier...")
vendor_batch_win   = Window.partitionBy("vendor_id")
vendor_cat_bat_win = Window.partitionBy("vendor_id", "final_category")

qa_df = (
    qa_df
    .withColumn("_vtotal", F.count("*").over(vendor_batch_win))
    .withColumn("_vcat",   F.count("*").over(vendor_cat_bat_win))
    .withColumn("_vshare", F.col("_vcat") / F.col("_vtotal"))
    .withColumn("flag_batch_outlier",
        (F.col("_vtotal") >= MIN_VENDOR_INVOICES_FOR_FLAG) &
        (F.col("_vshare") < BATCH_OUTLIER_THRESHOLD)
    )
    .drop("_vtotal", "_vcat", "_vshare")
)
c6 = qa_df.filter(F.col("flag_batch_outlier")).count()
logger.info(f"  Check 6 flags: {c6:,}")

# ── Composite score + priority ────────────────────────────────────────────────
qa_df = assign_priority_and_flags(qa_df)
qa_df.cache()

priority_dist = {
    r["review_priority"]: r["count"]
    for r in qa_df.groupBy("review_priority").count().collect()
}
override_count = qa_df.filter(F.col("override_suggested")).count()

record_metric("step3_5_priority_dist",  priority_dist)
record_metric("step3_5_override_count", override_count)
logger.info(f"Priority distribution: {priority_dist}")
logger.info(f"Records flagged for re-route: {override_count:,}")

write_staging(qa_df, T_QA, "3.5-qa")
written_df.unpersist()

# MAGIC %md
# MAGIC ---
# MAGIC ## SECTION 8 — Step 3.6: Re-route
# MAGIC > Flagged records are re-classified with QA hints injected into the LLM prompt.  
# MAGIC > Maximum **2 iterations**. Records still anomalous after 2 passes are escalated.

# COMMAND ----------

logger.info("=" * 60)
logger.info("STEP 3.6  Re-route")
logger.info("=" * 60)

# Split pass vs flagged
pass_df_final     = qa_df.filter(~F.col("override_suggested")).withColumn("reroute_iteration", F.lit(0).cast(IntegerType()))
flagged_to_reroute = qa_df.filter(F.col("override_suggested")).withColumn("reroute_iteration", F.lit(0).cast(IntegerType()))

pass_count_initial    = pass_df_final.count()
flagged_count_initial = flagged_to_reroute.count()

record_metric("step3_6_initial_pass",    pass_count_initial)
record_metric("step3_6_initial_flagged", flagged_count_initial)
logger.info(f"QA passed: {pass_count_initial:,}  |  Flagged: {flagged_count_initial:,}")

resolved_batches: list[DataFrame] = [pass_df_final]
still_flagged_df: DataFrame       = flagged_to_reroute

for iteration in range(1, MAX_REROUTE_ITERATIONS + 1):
    iter_count = still_flagged_df.count()
    if iter_count == 0:
        logger.info(f"Re-route iteration {iteration}: no records remain — exiting loop.")
        break

    logger.info(f"Re-route iteration {iteration}: {iter_count:,} records")

    # ── Re-classify with QA hint ──────────────────────────────────────────────
    still_flagged_df = attach_xgboost_preds(still_flagged_df)

    still_flagged_df.createOrReplaceTempView(f"reroute_llm_iter{iteration}")
    cat_list_str = ", ".join(CATEGORY_LIST)

    still_flagged_df = spark.sql(f"""
        SELECT *,
            ai_classify(
                CONCAT(final_description, ' [QA flags: ', array_join(anomalies, ', '), ']'),
                {CATEGORY_SQL_ARRAY}
            ) AS _llm_raw
        FROM reroute_llm_iter{iteration}
    """)
    still_flagged_df = (
        still_flagged_df
        .withColumn("llm_pred",
            F.when(F.col("_llm_raw").isNotNull(), F.col("_llm_raw.label"))
             .otherwise(F.lit("Unknown")).cast(StringType())
        )
        .withColumn("llm_conf",
            F.when(F.col("_llm_raw").isNotNull(), F.col("_llm_raw.score").cast(FloatType()))
             .otherwise(F.lit(0.0))
        )
        .drop("_llm_raw")
        .withColumn("agreement", F.col("xgboost_pred") == F.col("llm_pred"))
        .withColumn("_min_conf", F.least(F.col("xgboost_conf"), F.col("llm_conf")))
    )

    agree_df    = still_flagged_df.filter(F.col("agreement")).cache()
    disagree_df = still_flagged_df.filter(~F.col("agreement")).cache()

    agree_count    = agree_df.count()
    disagree_count = disagree_df.count()
    logger.info(f"  Iter {iteration}: agree={agree_count:,}  disagree={disagree_count:,}")

    agree_tagged = (
        agree_df
        .withColumn("final_category",      F.col("xgboost_pred"))
        .withColumn("ensemble_confidence", (F.col("xgboost_conf") + F.col("llm_conf")) / 2)
        .withColumn("resolution_method",   F.lit(f"reroute_accept_iter{iteration}"))
    )

    if disagree_count > 0:
        disagree_df.drop("_min_conf").createOrReplaceTempView(f"reroute_resolve_iter{iteration}")
        resolved_llm = spark.sql(f"""
            SELECT *,
                ai_query(
                    '{LLM_ENDPOINT}',
                    CONCAT(
                        'REROUTE PASS {iteration}. QA flags: ', array_join(anomalies, ', '), '.',
                        CHR(10),
                        'XGBoost=', xgboost_pred, ' (', ROUND(xgboost_conf,3), ')',
                        '  LLM=',  llm_pred,     ' (', ROUND(llm_conf,3), ')',
                        CHR(10),
                        'Description: ', final_description,
                        '  Vendor: ', vendor_name,
                        '  extc=', extc, ' mic=', mic_cln,
                        CHR(10),
                        'Pick the best category from: {cat_list_str}',
                        CHR(10),
                        'Reply with ONLY the category name.'
                    )
                ) AS _res_raw
            FROM reroute_resolve_iter{iteration}
        """)
        disagree_tagged = (
            resolved_llm
            .withColumn("final_category",      F.trim(F.col("_res_raw")).cast(StringType()))
            .withColumn("ensemble_confidence",  F.greatest(F.col("xgboost_conf"), F.col("llm_conf")))
            .withColumn("resolution_method",    F.lit(f"reroute_resolve_iter{iteration}"))
            .drop("_res_raw", "_min_conf")
        )
    else:
        disagree_tagged = (
            disagree_df.drop("_min_conf")
            .withColumn("final_category",      F.col("xgboost_pred"))
            .withColumn("ensemble_confidence", F.col("xgboost_conf"))
            .withColumn("resolution_method",   F.lit(f"reroute_resolve_iter{iteration}"))
        )

    iter_reclassified = agree_tagged.drop("_min_conf").unionByName(
        disagree_tagged, allowMissingColumns=True
    )
    agree_df.unpersist()
    disagree_df.unpersist()

    # ── Re-apply Tax Lookup + Use Tax ─────────────────────────────────────────
    iter_reclassified = apply_tax_lookup(iter_reclassified)

    # ── Re-apply Tax Writer (no RAG for speed on re-route) ───────────────────
    iter_reclassified = iter_reclassified.withColumn(
        "prior_writeup_examples", F.lit("").cast(StringType())
    )
    iter_reclassified = apply_tax_writer(iter_reclassified, temp_view_suffix=f"_reroute_iter{iteration}")
    iter_reclassified = iter_reclassified.drop("prior_writeup_examples")

    # ── Re-apply QA scoring ───────────────────────────────────────────────────
    iter_reclassified = assign_priority_and_flags(iter_reclassified)
    iter_reclassified = iter_reclassified.withColumn(
        "reroute_iteration", F.lit(iteration).cast(IntegerType())
    )

    # Split: resolved vs still anomalous
    now_resolved  = iter_reclassified.filter(~F.col("override_suggested"))
    still_flagged_df = iter_reclassified.filter(F.col("override_suggested"))

    resolved_this_iter = now_resolved.count()
    still_count        = still_flagged_df.count()
    logger.info(f"  Iteration {iteration}: {resolved_this_iter:,} resolved  |  {still_count:,} still flagged")

    resolved_batches.append(now_resolved)

# ── Escalate remaining flagged records ────────────────────────────────────────
escalated_df: DataFrame = still_flagged_df.withColumn(
    "recommendation",
    F.lit(f"ESCALATED: Unresolved after {MAX_REROUTE_ITERATIONS} re-route passes. Manual review required.")
)
escalated_count = escalated_df.count()
record_metric("step3_6_escalated", escalated_count)
logger.info(f"Escalated (unresolved): {escalated_count:,}")
resolved_batches.append(escalated_df)

# ── Union all into final rerouted ─────────────────────────────────────────────
rerouted_df: DataFrame = resolved_batches[0]
for part in resolved_batches[1:]:
    rerouted_df = rerouted_df.unionByName(part, allowMissingColumns=True)

rerouted_df.cache()
write_staging(rerouted_df, T_REROUTED, "3.6-rerouted")
qa_df.unpersist()

# MAGIC ## SECTION 9 — Step 3.7: Write Final Output & Notify

# ── Derive processing_path ────────────────────────────────────────────────────
final_df: DataFrame = rerouted_df.withColumn("processing_path",
    F.when(F.col("match_type") == "deterministic",                     F.lit("FAST_DETERMINISTIC"))
     .when(F.col("match_type") == "rag",                               F.lit("FAST_RAG"))
     .when(F.col("resolution_method").startswith("reroute"),           F.lit("ESCALATION_REROUTED"))
     .when(F.col("resolution_method") == "llm_resolution",            F.lit("COMPLEX_LLM_RESOLVED"))
     .when(F.col("resolution_method").isin("ACCEPT", "MEDIUM"),       F.lit("STANDARD_CLASSIFIER"))
     .otherwise(F.lit("UNKNOWN"))
)

# ── Inject pipeline metadata ──────────────────────────────────────────────────
final_df = (
    final_df
    .withColumn("reviewed_by",        F.lit(None).cast(StringType()))
    .withColumn("pipeline_run_month", F.lit(RUN_MONTH))
    .withColumn("pipeline_run_ts",    F.lit(RUN_TS).cast(TimestampType()))
)

# ── Rename columns to canonical output names ─────────────────────────────────
# These renames must happen BEFORE the column selection below.
# Order matters: rule_match → rule_applied, final_category → category
# ── Rename to canonical output names (must precede column selection) ──────────
if "rule_match" in final_df.columns and "rule_applied" not in final_df.columns:
    final_df = final_df.withColumnRenamed("rule_match", "rule_applied")

if "final_category" in final_df.columns and "category" not in final_df.columns:
    final_df = final_df.withColumnRenamed("final_category", "category")

# Architecture doc uses "anomaly_flags" (array) — rename from internal "anomalies"
if "anomalies" in final_df.columns:
    final_df = final_df.withColumnRenamed("anomalies", "anomaly_flags")

# Architecture doc output includes "components_used" — derive from processing_path + match_type
# Gives the Reviewer Agent a human-readable description of which pipeline components fired.
final_df = final_df.withColumn("components_used",
    F.when(F.col("processing_path") == "FAST_DETERMINISTIC",  F.lit("rules_expert(det)→tax_lookup→tax_writer→qa"))
     .when(F.col("processing_path") == "FAST_RAG",            F.lit("rules_expert(rag)→tax_lookup→tax_writer→qa"))
     .when(F.col("processing_path") == "STANDARD_CLASSIFIER", F.lit("rules_expert→xgboost+ai_classify→tax_lookup→tax_writer→qa"))
     .when(F.col("processing_path") == "COMPLEX_LLM_RESOLVED",F.lit("rules_expert→xgboost+ai_classify→gpt4o_resolve→tax_lookup→tax_writer→qa"))
     .when(F.col("processing_path") == "ESCALATION_REROUTED", F.lit("rules_expert→classifier→qa_flag→reroute(x2)→tax_lookup→tax_writer→qa"))
     .otherwise(F.lit("unknown"))
)

# ── Null-fill columns absent on the rules path ────────────────────────────────
for _col, _dtype in [
    ("xgboost_pred",        StringType()),
    ("xgboost_conf",        FloatType()),
    ("llm_pred",            StringType()),
    ("llm_conf",            FloatType()),
    ("agreement",           BooleanType()),
    ("ensemble_confidence", FloatType()),
    ("resolution_method",   StringType()),
    ("reroute_iteration",   IntegerType()),
]:
    if _col not in final_df.columns:
        final_df = final_df.withColumn(_col, F.lit(None).cast(_dtype))

    "invoice_id", "inv_line_number",

    "final_description", "vendor_name", "vendor_id", "extc", "mic_cln", "account_code",
    "state", "amount", "use_tax_amount", "entity_code",

    "category", "taxability", "Reverse_UseTax", "confidence", "ensemble_confidence",

    # Write-up
    "short_justification", "full_writeup", "reviewer_note", "audit_ready",
    # QA  — "anomaly_flags" matches architecture doc output spec
    "anomaly_score", "anomaly_flags", "review_priority", "override_suggested", "recommendation",
    # Metadata
    "reviewed_by", "pipeline_run_month", "pipeline_run_ts",
]

# Only select columns that actually exist (guards against optional columns)
output_cols = [c for c in CANONICAL_COLS if c in final_df.columns]
final_df    = final_df.select(output_cols)

# ── Quality gate ──────────────────────────────────────────────────────────────
final_count = final_df.count()
if final_count != TOTAL_RECORDS:

final_df.cache()
flagged_review    = final_df.filter(F.col("review_priority").isin("HIGH", "CRITICAL")).count()

path_dist = {

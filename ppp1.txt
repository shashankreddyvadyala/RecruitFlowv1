def _assign(df, cond, cluster, label):
    """
    Apply one rule. Only sets cluster/rule_matched when still NULL.
    First match wins — never overwrites an already-set cluster.
    """
    return (
        df.withColumn("cluster",
                      when(col("cluster").isNull() & cond, lit(cluster))
                      .otherwise(col("cluster")))
          .withColumn("rule_matched",
                      when(col("rule_matched").isNull() & cond, lit(label))
                      .otherwise(col("rule_matched")))
    )


# Description columns searched by _contains() — all three are checked
# so a keyword match in ANY of them fires the rule
_DESC_COLS = [
    "final_description",          # coalesced: LINE_DESCRIPTION + DESCRIPTION combined
    "DESCRIPTION",                # raw AP invoice description field
    "DIST_ALL_DESCRIPTION",       # distribution-level description
]


def _contains(*kws):
    """
    True if ANY of the description columns contains ANY of the given keywords.
    Searches final_description, DESCRIPTION, and DIST_ALL_DESCRIPTION together
    so a match in any column fires the rule — no content is missed.
    Case-insensitive.
    """
    if not kws:
        return lit(False)
    expr = lit(False)
    for kw in kws:
        for dc in _DESC_COLS:
            expr = expr | lower(col(dc)).like(f"%{kw.lower()}%")
    return expr


def _not_contains(*kws):
    """True if NONE of the description columns contain any of the given keywords."""
    return ~_contains(*kws)

        # Returns NULL if column is missing, is a filler, or is null
        if c not in df.columns:

        return when(trim(col(c)).isin(filler) | col(c).isNull(), lit(None)).otherwise(col(c))

    df = _assign(df, (col(E) == "879") | (col(A) == "6621.99"),
        "Conferencing & Resale Services", "extc=879/acct=6621")

    # ── Professional & Consulting Services ───────────────────────────────────
    # 6722.99 = consulting/professional services accruals

    df = _assign(df, col(A) == "6728.99",

    # 6231.xxx = OSP engineering design (project mgmt accounts)

        "Outside Plant Construction", "acct=623x.x")

    # ── Facilities & Water ────────────────────────────────────────────────────
    # EXTC=543 = facility/water (QUENCH = water cooler, CBRE = building services)
    # 6123.91 = water/beverage service at facilities
    df = _assign(df, (col(E) == "543") | (col(A) == "6123.91"),
        "Facilities Services", "extc=543/acct=6123.91")

    # ── EXTC 520 + non-maintenance accounts → field safety/equipment ──────────
    # 520 + 65xx = field tools/safety equipment via EXTC 520
    df = _assign(df,
        (col(E) == "520") & (col(A).like("65%") | col(A).like("651%")),
        "Tools & Testing Equipment", "extc=520+acct=65x")

    # ── Miscellaneous uncovered EXTC codes → catch-all by account prefix ──────
    # These fire AFTER all specific rules — broad fallback for remaining 2003.x
    df = _assign(df, col(A).like("2003.%"),
        "Outside Plant Construction", "acct=2003.x-catchall")
    # 6121.xxx = operating expense accounts (telecom property tax / G&A)
    df = _assign(df, col(A).like("6121.%") & ~_contains("utility","util","electric","gas","water","trash"),
        "Operating Expense", "acct=6121.x")
    # 1330.xxx = prepaid / intangible assets
    df = _assign(df, col(A).like("1330.%"),
        "Prepaid / Intangible Asset", "acct=1330.x")
    # 4080.xxx = access / settlement charges
    df = _assign(df, col(A).like("4080.%"),
        "Access & Settlement Charges", "acct=4080.x")
    # 6998/6999 = miscellaneous expense
    df = _assign(df, col(A).like("6998.%") | col(A).like("6999.%"),
        "Miscellaneous Expense", "acct=6998/6999.x")
    # 6362.xxx = customer premises equipment maintenance

        "CPE Maintenance", "acct=6362.x")

        "Utilities", "acct=6531.x")
    # 6512.x = depreciation / amortization (catch last)

        "Maintenance", "acct=651x.x")

    # PRIORITY 3 — KEYWORD RULES  (description text — least specific, last)
    # ════════════════════════════════════════════════════════════════════════
    # keyword_rules is passed in from the caller (loaded from Excel or hardcoded).
    # dtv is always applied first — it comes after co=7M39 (which → "DTV") so
    # rows with company_code=7M39 are already set and won't be overwritten here.

    # Drop the temporary effective MIC column — not needed in output
    df = df.drop("_eff_mic")

    df = (df
          .withColumn("label_cluster",
                      when(col("cluster").isNotNull(), col("cluster"))
                      .otherwise(lit(None).cast("string")))
          .withColumn("label_source",
                      when(col("cluster").isNotNull(), lit("rule"))
                      .otherwise(lit("no_rule"))))

    cnts   = df.groupBy("label_source").agg(count("*").alias("n")).toPandas()
    total  = int(cnts["n"].sum())
    rule_n = int(cnts.loc[cnts["label_source"] == "rule", "n"].sum()) if total else 0
    if total:
        print(f"  Rule-matched: {rule_n:,} ({rule_n/total*100:.1f}%)  "
              f"→ ML: {total - rule_n:,}")
    return df

# ── Shared utility ── must be defined before _build_features and train() ──────
def _to_str(x):
    """Convert any value to a clean string for LabelEncoder. Handles None/NaN/array."""
    if x is None:                                  return "UNKNOWN"
    if isinstance(x, float) and np.isnan(x):      return "UNKNOWN"
    if isinstance(x, (list, np.ndarray)):
        return str(x[0]).strip() if len(x) > 0 else "UNKNOWN"
    return str(x).strip()


def _build_features(pdf, tfidf, tfidf_vendor, cat_enc):
    """
    Build a sparse feature matrix.
    Accepts fitted artifacts as explicit arguments so it is
    safely serialisable inside mapInPandas on any executor.
    """
    texts  = pdf["final_description"].fillna("").astype(str).tolist()
    X_text = tfidf.transform(texts)
    vendor_texts = pdf[COLS["vendor_name"]].fillna("").astype(str).tolist()
    X_vendor = tfidf_vendor.transform(vendor_texts)

    parts = []
    for c in CAT_COLS:
        enc   = cat_enc[c]
        known = set(enc.classes_)
        vals  = np.array([v if (v := _to_str(u)) in known else "UNKNOWN"
                          for u in pdf[c].values])
        parts.append(enc.transform(vals).reshape(-1, 1).astype(np.float32))

    vh = (pdf[COLS["vendor_name"]].fillna("")
          .apply(lambda n: int(hashlib.md5(str(n).encode("utf-8")).hexdigest(), 16) % 2000)
          .values.reshape(-1, 1).astype(np.float32))

    am = (np.log1p(np.abs(
          pd.to_numeric(pdf[COLS["tran_amount"]], errors="coerce").fillna(0).values))
          .reshape(-1, 1).astype(np.float32))

    acct_h = (pdf[COLS["account_code"]].fillna("")
              .apply(lambda a: int(hashlib.md5(str(a).encode("utf-8")).hexdigest(), 16) % 500)
              .values.reshape(-1, 1).astype(np.float32))
    return sp_hstack([X_text, X_vendor, csr_matrix(np.hstack(parts + [vh, am, acct_h]))], format="csr")


print("Features OK")

# CELL 6 — TRAIN  (Spark for sampling, driver for XGBoost)
# ================================================================================

def train(df_hist, df_curr_eval=None):
    """
    1. Spark side  — filter rule rows, sampleBy (no shuffle, no full sort)
    2. Driver side — toPandas on the small sample → fit TF-IDF + LabelEncoders
                     → train XGBoost
    Returns: (model, tfidf, cat_enc, label_enc)  — broadcast by the caller.
    """
    TRAIN_COLS = ["final_description", "label_cluster"] + CAT_COLS + \
                 [COLS["vendor_name"], COLS["tran_amount"], COLS["account_code"]]
    avail = [c for c in TRAIN_COLS if c in df_hist.columns]

    # Filter to rule-matched rows with a known cluster
    df_lbl = (df_hist.filter(col("label_source") == "rule")
                     .filter(col("label_cluster").isNotNull())
                     .select(avail))

    # Count per cluster to build sampleBy fractions
    cnts = (df_lbl.groupBy("label_cluster")
                  .agg(count("*").alias("n"))
                  .toPandas())
    valid = cnts.loc[cnts["n"] >= MIN_CLASS_SAMPLES, "label_cluster"].tolist()
    print(f"  {int(cnts['n'].sum()):,} labeled rows  |  {len(valid)} valid clusters")

    if len(valid) == 0:
        raise ValueError("No clusters meet MIN_CLASS_SAMPLES — check rules output.")

    # ── sampleBy: approximate stratified sample, NO shuffle, very fast ───────
    n_by_cluster = cnts.set_index("label_cluster")["n"].to_dict()
    fractions    = {
        c: min(1.0, MAX_ROWS_PER_CLASS / max(1, n_by_cluster[c]))
        for c in valid
    }
    pdf = (df_lbl.filter(col("label_cluster").isin(valid))
                 .sampleBy("label_cluster", fractions=fractions, seed=42)
                 .toPandas())
    print(f"  Training sample: {len(pdf):,} rows pulled to driver")

    # ── Fit TF-IDF + LabelEncoders on training sample ─────────────────────────
    # min_df=2 requires each term to appear in ≥2 docs; use 1 on small training sets
    _min_df = 2 if len(pdf) > 500 else 1
    # Description TF-IDF: larger vocab + trigrams for better phrase capture
    tfidf = TfidfVectorizer(max_features=8_000, ngram_range=(1, 3),
                            sublinear_tf=True, min_df=_min_df,
                            analyzer="word", token_pattern=r"[a-zA-Z0-9]{2,}")

    texts  = pdf["final_description"].fillna("").astype(str).tolist()
    X_text = tfidf.fit_transform(texts)

    # Vendor name TF-IDF: vendor names carry strong cluster signal
    _min_df_v = 2 if len(pdf) > 300 else 1
    tfidf_vendor = TfidfVectorizer(max_features=1_000, analyzer="char_wb",
                       ngram_range=(3, 5), sublinear_tf=True,
                       min_df=_min_df_v)
    vendor_texts = pdf[COLS["vendor_name"]].fillna("").astype(str).tolist()
    X_vendor = tfidf_vendor.fit_transform(vendor_texts)

    cat_enc = {}
    parts   = []
    for c in CAT_COLS:
        vals = np.array([_to_str(v) for v in pdf[c].values])
        enc  = LabelEncoder()
        enc.fit(np.append(vals, "UNKNOWN"))
        cat_enc[c] = enc
        parts.append(enc.transform(vals).reshape(-1, 1).astype(np.float32))

    vh = (pdf[COLS["vendor_name"]].fillna("")
          .apply(lambda n: int(hashlib.md5(str(n).encode("utf-8")).hexdigest(), 16) % 2000)
          .values.reshape(-1, 1).astype(np.float32))
    am = (np.log1p(np.abs(
          pd.to_numeric(pdf[COLS["tran_amount"]], errors="coerce").fillna(0).values))
          .reshape(-1, 1).astype(np.float32))

    # Account code hash — ACCOUNT_CODE has 120+ unique values, hash avoids one-hot
    acct_h = (pdf[COLS["account_code"]].fillna("")
              .apply(lambda a: int(hashlib.md5(str(a).encode("utf-8")).hexdigest(), 16) % 500)
              .values.reshape(-1, 1).astype(np.float32))
    X = sp_hstack([X_text, X_vendor, csr_matrix(np.hstack(parts + [vh, am, acct_h]))], format="csr")
    print(f"  Feature matrix: {X.shape[0]:,} × {X.shape[1]:,}")

    label_enc = LabelEncoder()
    y = label_enc.fit_transform(pdf["label_cluster"].astype(str))
    del pdf

    # ── Eval set ──────────────────────────────────────────────────────────────
    pdf_ev = pd.DataFrame()
    if df_curr_eval is not None:
        avail_ev = [c for c in TRAIN_COLS if c in df_curr_eval.columns]
        pdf_ev   = (df_curr_eval.filter(col("label_source") == "rule")
                                .filter(col("label_cluster").isNotNull())
                                .select(avail_ev).dropna(subset=["label_cluster"])
                                .limit(500 * len(valid))
                                .toPandas())
        known  = set(label_enc.classes_)
        pdf_ev = pdf_ev[pdf_ev["label_cluster"].astype(str).isin(known)]

    if len(pdf_ev) == 0:
        n   = X.shape[0]
        if n < 10:
            raise ValueError(
                f"Only {n} training rows after sampling — need at least 10. "
                "Lower MIN_CLASS_SAMPLES or check rules output."
            )
        idx = np.random.RandomState(42).permutation(n)
        cut = max(1, int(n * 0.9))   # max(1,...) prevents empty train set
        X, X_ev = X[idx[:cut]], X[idx[cut:]]
        y, y_ev = y[idx[:cut]], y[idx[cut:]]
        eval_lbl = "10% historical holdout"
    else:
        X_ev     = _build_features(pdf_ev, tfidf, tfidf_vendor, cat_enc)
        y_ev     = label_enc.transform(pdf_ev["label_cluster"].astype(str).values)
        eval_lbl = f"current-month out-of-time ({len(y_ev):,} rows)"
        del pdf_ev

    n_classes = len(label_enc.classes_)
    print(f"  Eval: {eval_lbl}  |  {n_classes} clusters")
    print("  Training XGBoost …")

    model = XGBClassifier(
        n_estimators=500,           # more trees — early stopping finds optimal
        max_depth=8,                # deeper trees capture complex rule-like patterns
        learning_rate=0.03,         # lower LR + more trees = better generalisation
        subsample=0.85,
        colsample_bytree=0.7,       # feature subsampling reduces overfitting
        colsample_bylevel=0.8,
        min_child_weight=3,         # prevents splits on very rare clusters
        gamma=0.1,                  # minimum loss reduction for a split
        reg_alpha=0.1,              # L1 regularisation
        reg_lambda=1.5,             # L2 regularisation
        eval_metric="mlogloss",
        early_stopping_rounds=30,   # more patience for complex multi-class
        tree_method="hist",
        grow_policy="lossguide",
        max_leaves=64,              # limit leaf count for regularisation
        objective="multi:softprob",
        num_class=n_classes,        # required by XGBoost 2.x
        random_state=42,
        n_jobs=-1,
    )
    model.fit(X, y, eval_set=[(X_ev, y_ev)], verbose=False)
    best_n = getattr(model, 'best_iteration', model.n_estimators)
    print(f"  Best iteration : {best_n} / {model.n_estimators} trees")

    y_pred_ev = np.argmax(model.predict_proba(X_ev), 1)
    acc = accuracy_score(y_ev, y_pred_ev)
    f1  = f1_score(y_ev, y_pred_ev, average="weighted", zero_division=0)
    print(f"  Eval accuracy : {acc:.4f}")
    print(f"  Weighted F1   : {f1:.4f}")
    return model, tfidf, tfidf_vendor, cat_enc, label_enc


print("Train OK")

# CELL 7 — SCORE  (fully distributed via mapInPandas — no toPandas on full data)
# ================================================================================

def score(df, model_bc, tfidf_bc, tfidf_vendor_bc, cat_enc_bc, label_enc_bc):
    """
    Classify every row distributed across the cluster.

    Strategy:
      - Rule rows     → predicted_cluster = cluster column (already known)
      - No-rule rows  → XGBoost prediction via mapInPandas Pandas UDF
                        Model is broadcast → loaded ONCE per partition, not per row.

    Returns a Spark DataFrame (stays distributed — no driver bottleneck).
    """

    # Only select the columns we need — avoid pulling unused data to executors
    sel = list(dict.fromkeys([
        COLS["invoice_id"],
        COLS["accounting_date"],          # keep for downstream date filtering
        "INV_LINE_NUMBER",                # composite unique key part 2
        "INVOICE_DIST_LINE_NUM",          # composite unique key part 3
        "INVOICE_NUM",                    # AP reference number
        COLS["vendor_name"],
        COLS["account_code"],
        COLS["tran_amount"],
        "final_description",
        "label_source",
        "label_cluster",
        "cluster",
        "rule_matched",
    ] + CAT_COLS))
    avail = [c for c in sel if c in df.columns]

    # Output schema: original columns + 4 prediction columns
    input_schema = df.select(avail).schema
    output_schema = StructType(list(input_schema.fields) + [
        StructField("predicted_cluster", StringType(), True),
        StructField("confidence",        FloatType(),  True),
        StructField("decision",          StringType(), True),
        StructField("reported_date",     DateType(),   True),
    ])

    # ── mapInPandas function — runs on each executor partition ───────────────
    def _score_partition(iterator):
        """
        Called once per Spark partition.
        Broadcast values are loaded from the driver once per partition —
        NOT once per row. This is the key efficiency gain.
        """
        model        = model_bc.value
        tfidf        = tfidf_bc.value
        tfidf_vendor = tfidf_vendor_bc.value
        cat_enc      = cat_enc_bc.value
        label_enc    = label_enc_bc.value

        for pdf in iterator:
            if pdf.empty:
                yield pdf.assign(
                    predicted_cluster=pd.Series(dtype="object"),
                    confidence=pd.Series(dtype="float32"),
                    decision=pd.Series(dtype="object"),
                    reported_date=pd.Series(dtype="object"),
                )
                continue

            rule_mask = pdf["label_source"] == "rule"
            ml_mask   = ~rule_mask

            pdf = pdf.copy()
            pdf["predicted_cluster"] = None
            pdf["confidence"]        = None
            pdf["decision"]          = None
            pdf["reported_date"]     = None

            # Rule rows — cluster already known, confidence = 1.0
            if rule_mask.sum() > 0:
                pdf.loc[rule_mask, "predicted_cluster"] = pdf.loc[rule_mask, "cluster"]
                pdf.loc[rule_mask, "confidence"]        = np.float32(1.0)
                pdf.loc[rule_mask, "decision"]          = "rule"

            # ML rows — score with XGBoost
            if ml_mask.sum() > 0:
                pdf_ml = pdf.loc[ml_mask].copy()
                X      = _build_features(pdf_ml, tfidf, tfidf_vendor, cat_enc)
                prob   = model.predict_proba(X)
                idx    = np.argmax(prob, 1)
                conf   = prob[np.arange(len(prob)), idx].astype(np.float32)
                preds  = label_enc.inverse_transform(idx)

                pdf.loc[ml_mask, "predicted_cluster"] = list(preds)
                pdf.loc[ml_mask, "confidence"]        = list(conf)
                # CONFIDENCE_THRESHOLD captured from module scope (set in CONFIG)
                pdf.loc[ml_mask, "decision"]          = np.where(
                    conf >= CONFIDENCE_THRESHOLD, "ml", "need_review"
                ).tolist()

            # reported_date = the date this pipeline was run (today on the executor)
            pdf["reported_date"] = _dt.date.today()

            yield pdf

    # Repartition before scoring for even workload across executors
    df_scored = (df.select(avail)
                   .repartition(SCORE_PARTITIONS)
                   .mapInPandas(_score_partition, schema=output_schema))

    return df_scored


print("Score OK")

def _load_keyword_rules_from_excel(excel_path):
    """
    Load keyword rules from the Excel file and return a MERGED list:
      - Hardcoded baseline rules are the foundation (always present).
      - Excel rules OVERRIDE any hardcoded rule with the same first keyword.
      - New keywords found only in the Excel are APPENDED.
      - Hardcoded rules not updated in Excel are preserved unchanged.

    This ensures:
      • AP team can update rules in Excel without touching the code.
      • Code-only rules (new rules added by developers) are never silently lost.
      • The two sources stay in sync — Excel wins on conflicts.
    """
    import subprocess, os, re as _re

    # ── Copy from ADLS to driver /tmp ────────────────────────────────────────
    local_path = "/tmp/_cluster_rules_tmp.xlsx"
    try:
        dbutils.fs.cp(excel_path, f"file:{local_path}", recurse=False)
    except NameError:
        subprocess.run(
            ["hdfs", "dfs", "-get", "-f", excel_path, local_path], check=True
        )

    # ── Cluster name normalisation ────────────────────────────────────────────
    _CLUSTER_FIXES = {
        "cable-metallic":              "Cable - Metallic",
        "cable - nonmetalic":          "Cable - Nonmetallic",
        "cable nonmetalic":            "Cable - Nonmetallic",
        "security service":            "Security Service",
        "tools and testing equipment": "Tools & Testing Equipment",
    }
    def _fix_cluster(name):
        n = str(name).strip()
        return _CLUSTER_FIXES.get(n.lower(), n)

    # ── Rule text normalisation ────────────────────────────────────────────────
    _RULE_FIXES = {"pannel": "panel", "cabel": "cable", "ectc": "extc"}
    def _fix_rule(text):
        for wrong, right in _RULE_FIXES.items():
            text = text.replace(wrong, right)
        return text

    # ── Keywords to skip (too broad, artifacts, or single letters) ────────────
    _SKIP = {"data", "cabel", "cabel but", "cable but", "fiber and",
             "but", "or", "and", "not"}

    # ── Parse Excel into a flat list of (kws, cluster) pairs ─────────────────
    excel_rules_raw = []
    seen_kw0 = set()

    try:
        xl = pd.ExcelFile(local_path)
    except Exception as ex:
        os.remove(local_path)
        raise RuntimeError(f"Cannot open Excel file at {excel_path}: {ex}")

    # Sheet 1: Keyword Rules
    try:
        df_kw = xl.parse("Keyword Rules")
    except Exception:
        df_kw = xl.parse(0)
        print("  Note: 'Keyword Rules' sheet not found — using first sheet")

    for _, row in df_kw.dropna(subset=["RULE", "CLUSTER"]).iterrows():
        raw  = _fix_rule(str(row["RULE"]).strip().lower())
        clus = _fix_cluster(row["CLUSTER"])
        if "but not"   in raw: raw = raw.split("but not")[0].strip().strip("()")
        if "should not" in raw: raw = raw.split("should not")[0].strip().strip("()")
        if len(raw) > 80:
            parts = [p.strip() for p in raw.replace(";",",").split(",")
                     if 2 < len(p.strip()) < 30]
            raw = parts[0] if parts else ""
        if not raw or raw in _SKIP:
            continue
        kws = tuple(k.strip() for k in raw.replace("/",",").split(",")
                    if k.strip() and len(k.strip()) > 1)
        if not kws or kws[0] in seen_kw0:
            continue
        seen_kw0.add(kws[0])
        excel_rules_raw.append((kws, clus))

    # Sheet 2: Generic Rules — extract description keyword patterns only
    try:
        df_gen = xl.parse("Generic Rules")
        _bad_kw = {"base service","landscape gravel","fiber","fiber and",
                   "account_code","extc","mic","vendor_name"}
        for _, row in df_gen.dropna(subset=["RULE","CLUSTER"]).iterrows():
            raw  = str(row["RULE"]).strip().lower()
            clus = _fix_cluster(row["CLUSTER"])
            m = _re.search(r"description\s*=\s*%?([a-zA-Z0-9 _-]{3,30})%?", raw)
            if m:
                kw = m.group(1).strip()
                if (kw and kw not in _bad_kw and "account" not in kw
                        and len(kw) > 3 and kw not in seen_kw0):
                    seen_kw0.add(kw)
                    excel_rules_raw.append(((kw,), clus))
    except Exception as ex:
        print(f"  Note: Generic Rules sheet parse issue: {ex}")

    os.remove(local_path)

    # ── MERGE: Excel overrides hardcoded, new Excel rules appended ────────────
    merged   = list(_KEYWORD_RULES_BASE)          # start with full hardcoded base
    hc_index = {r[0][0]: i for i, r in enumerate(merged)}  # index by first keyword

    overrides  = []
    additions  = []
    for ex_kws, ex_cls in excel_rules_raw:
        kw0 = ex_kws[0]
        lbl = f"kw_xl={kw0[:20]}"
        if kw0 in hc_index:
            # Excel rule updates an existing hardcoded rule — override in-place
            idx = hc_index[kw0]
            old_cls = merged[idx][1]
            merged[idx] = (ex_kws, ex_cls, lbl)
            if old_cls != ex_cls:
                overrides.append(f"'{kw0}': '{old_cls}' → '{ex_cls}'")
        else:
            # New keyword only in Excel — append after hardcoded rules
            merged.append((ex_kws, ex_cls, lbl))
            hc_index[kw0] = len(merged) - 1
            additions.append(f"'{kw0}' → '{ex_cls}'")

    # ── Print merge summary ───────────────────────────────────────────────────
    print(f"  Excel merge complete:")
    print(f"    Base (hardcoded) : {len(_KEYWORD_RULES_BASE)} rules")
    print(f"    From Excel       : {len(excel_rules_raw)} rules")
    print(f"    Overrides        : {len(overrides)}")
    print(f"    New additions    : {len(additions)}")
    print(f"    Final total      : {len(merged)} rules")
    if overrides:
        for o in overrides:
            print(f"      OVERRIDE  {o}")
    if additions:
        for a in additions[:5]:
            print(f"      NEW       {a}")
        if len(additions) > 5:
            print(f"      ... and {len(additions)-5} more")

    return merged

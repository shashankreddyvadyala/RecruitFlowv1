def load_rules(path=None):
    if not path:
        print("No rules file — all invoices go to ML")
        empty = pd.DataFrame(columns=["rule","cluster"])
        return {"keyword": empty, "parker_volpe": pd.DataFrame()}
    dbutils.fs.cp(path, "file:/tmp/rules.xlsx", recurse=False)
    kw = pd.read_excel("/tmp/rules.xlsx", sheet_name="Keyword Rules")
    pv = pd.read_excel("/tmp/rules.xlsx", sheet_name="Parker Volpe - Rules")
    kw.columns = ["rule","cluster","generic_satisfied"]
    rules = {"keyword": kw.dropna(subset=["rule","cluster"]), "parker_volpe": _parse_pv(pv)}
    print(f"  {len(rules['keyword'])} keyword rules, {len(rules['parker_volpe'])} Parker Volpe")
    return rules

def _parse_pv(raw):
    recs = [{"extc":"69A","cluster":"Operating Lease","rule_type":"extc_only"}]
    for _, r in raw.iloc[6:18,:2].dropna(how="all").iterrows():
        if pd.notna(r.iloc[0]) and pd.notna(r.iloc[1]):
            recs.append({"vendor_name":str(r.iloc[0]).strip(),"cluster":str(r.iloc[1]).strip(),"rule_type":"vendor_only"})
    for _, r in raw.iloc[21:29,:3].dropna(how="all").iterrows():
        if pd.notna(r.iloc[0]) and pd.notna(r.iloc[2]):
            recs.append({"vendor_name":str(r.iloc[0]).strip(),"extc":str(r.iloc[1]).strip(),"cluster":str(r.iloc[2]).strip(),"rule_type":"vendor_extc"})
    for _, r in raw.iloc[32:42,:3].dropna(how="all").iterrows():
        if pd.notna(r.iloc[0]) and pd.notna(r.iloc[2]):
            recs.append({"vendor_name":str(r.iloc[0]).strip(),"desc_kw":str(r.iloc[1]).strip() if pd.notna(r.iloc[1]) else "","cluster":str(r.iloc[2]).strip(),"rule_type":"vendor_desc"})
    return pd.DataFrame(recs)

def _assign(df, cond, cluster, label):
    return (df.withColumn("cluster",     when(cond, lit(cluster)).otherwise(col("cluster")))
              .withColumn("rule_matched",when(cond, lit(label)).otherwise(col("rule_matched"))))

def _contains(*kws):
    if not kws: return lit(False)
    c = lower(col("final_description")).like(f"%{kws[0].lower()}%")
    for k in kws[1:]: c = c | lower(col("final_description")).like(f"%{k.lower()}%")
    return c

def build_labels(df, rules, spark):
    # Build description
    filler = ["SCM Purchase","SCM PURCHASE","scm purchase"]
    def _clean(c): return when(col(c).isin(filler)|col(c).isNull(), lit(None)).otherwise(col(c))
    df = (df.withColumn("_l",_clean(COLS["line_description"]))
            .withColumn("_d",_clean(COLS["description"]))
            .withColumn("_x",_clean(COLS["dist_all_description"])))
    df = df.withColumn("final_description", coalesce(
        when(col("_l").isNotNull()&col("_d").isNotNull(), concat_ws(" ",col("_l"),col("_d"))),
        col("_l"), col("_d"), col("_x"), lit("")
    )).drop("_l","_d","_x")

    df = df.withColumn("cluster",     lit(None).cast("string"))
    df = df.withColumn("rule_matched",lit(None).cast("string"))

    # Keyword rules
    kw = rules["keyword"]
    if not kw.empty:
        for _, rule in kw.iterrows():
            raw = str(rule["rule"]).strip()
            if "but not" in raw.lower():
                base, excl = raw.lower().split("but not",1)
                cond = lower(col("final_description")).like(f"%{base.strip()}%")
                for e in excl.replace("(","").replace(")","").split(","):
                    cond = cond & ~lower(col("final_description")).like(f"%{e.strip()}%")
            elif "," in raw:
                cond = _contains(*[k.strip() for k in raw.split(",")])
            else:
                cond = lower(col("final_description")).like(f"%{raw.lower()}%")
            df = _assign(df, cond, rule["cluster"], f"kw:{raw[:30]}")

    # Generic field rules
    M=COLS["mic"]; E=COLS["extc"]; A=COLS["account_code"]; V=COLS["vendor_name"]; CO=COLS["company_code"]
    df = _assign(df, col(M)=="SW",                             "ERTU",                              "mic=SW")
    df = _assign(df, col(M).isin(["CPEMIS","CPERTR"]),         "CPE",                               "mic=CPE")
    df = _assign(df, col(M)=="CEQ",                            "Computer Equipments",               "mic=CEQ")
    df = _assign(df, col(M).isin(["COE10K","CHROR1K"]),        "Miscellaneous Materials - Telecom", "mic=COE10K")
    df = _assign(df, col(M).like("SEQ%"),                      "Network Equipment",                 "mic=SEQ")
    df = _assign(df, col(M).like("NEQ%"),                      "Network Equipment",                 "mic=NEQ")
    df = _assign(df, col(M).like("TOWER%"),                    "Tower Materials",                   "mic=TOWER")
    df = _assign(df, col(M).like("CEQ%"),                      "Telecom Equipment",                 "mic=CEQ%")
    df = _assign(df, col(M).like("FBR%")&col(M).like("%K"),   "Telecom Equipment",                 "mic=FBR%K")
    df = _assign(df, col(M).like("ANT%")&col(M).like("%K"),   "Communications Equipment",          "mic=ANT%K")
    df = _assign(df, col(E)=="48C",                            "Freight - Common Carrier Charges",  "extc=48C")
    df = _assign(df, col(E)=="89Q",                            "Research and Development",          "extc=89Q")
    df = _assign(df, col(E).like("894%"),                      "Permits",                           "extc=894%")
    df = _assign(df, (col(E)=="520")&col(M).isin(["COE1K","COEOTHER"]), "Miscellaneous Materials - Telecom","extc=520+mic")
    df = _assign(df, (col(M)=="COEOTHER")&(col(E)=="450"),     "Contract plant labor",              "mic=COE+extc=450")
    df = _assign(df, (col(M)=="COEOTHER")&(col(E)=="460"),     "Vendor Engineering",                "mic=COE+extc=460")
    df = _assign(df, col(M).like("EFI%")&(col(E)=="450"),      "Contract plant labor",              "mic=EFI+extc=450")
    df = _assign(df, (col(M)=="CPEMIS")&(col(E)=="48C"),       "Freight",                           "mic=CPEMIS+extc=48C")
    df = _assign(df, (col(E)=="520")&col(A).like("%M"),        "Maintenance",                       "extc=520+acct=%M")
    df = _assign(df, col(A).like("1130%"),                     "Nontaxable",                        "acct=1130%")
    df = _assign(df, col(A).isin(["1220.121","1220.147"]),     "Inventory",                         "acct=1220.x")
    df = _assign(df, col(A).like("4010%")&~_contains("freight charges"), "Inventory",              "acct=4010%")
    df = _assign(df, (col(A)=="6124.570M")&(col(E)=="6A9"),    "Operating Lease",                   "acct+extc=6A9")
    df = _assign(df, (col(A)=="399C")&(col(E)=="61J"),         "Software - ERTU",                   "acct=399C")
    df = _assign(df, col(A)=="6114.1",                         "Tools and Testing Equipment",       "acct=6114.1")
    df = _assign(df, (col(E)=="450")&(col(A)=="2003.85C"),     "Outside Plant Construction",        "extc=450+acct=2003")
    df = _assign(df, (col(E)=="460")&(col(A)=="2003.85C"),     "Outside Plant Construction",        "extc=460+acct=2003")
    df = _assign(df, col(CO)=="7M39",                          "DTV",                               "co=7M39")

        "IRON MOUNTAIN":"Storage Services","DIVERSIFIED WIRE & CABLE":"Cable - Nonmetalic",
        "CISCO SYSTEMS CAPITAL":"Capital Lease","TEKSYSTEMS INC":"Data Processing",
        "HIGHTOWERS PETROLEUM":"Propane / Fuels","AMERICAN SAFETY UTILITY":"Safety Equipment",
        "SAF T GARD INTERNATIONAL":"Safety Equipment","STANLEY CONVERGENT":"Security Service",
        "CANTEEN ONE":"Food/Beverage","ARAMARK REFRESHMENT":"Food/Beverage",
    }.items():
        df = _assign(df, col(V).like(f"%{vendor}%"), cluster, f"vendor={vendor[:20]}")

    # Parker Volpe overrides
    pv = rules["parker_volpe"]
    if not pv.empty and "rule_type" in pv.columns:
        for _, r in pv[pv["rule_type"]=="extc_only"].iterrows():
            df = _assign(df, col(E)==r["extc"], r["cluster"], f"pv:extc={r['extc']}")

        for _, r in pv[pv["rule_type"]=="vendor_desc"].iterrows():
            df = _assign(df, (col(V)==r["vendor_name"])&_contains(*[k.strip() for k in str(r.get("desc_kw","")).split("OR")]), r["cluster"], "pv:v+d")

    # Map cluster → account code
    map_path = PATHS.get("cluster_account_map")
    if map_path:
        cmap = spark.read.csv(map_path, header=True).withColumnRenamed("cluster","_c")
        df = df.join(F.broadcast(cmap), col("cluster")==col("_c"), "left").drop("_c")
        df = df.withColumn("label_account_code",
                 when(col("cluster").isNotNull()&col("account_code_mapped").isNotNull(), col("account_code_mapped"))

        df = df.withColumn("account_code_mapped", lit(None).cast("string"))

    df = df.withColumn("label_source",
             when(col("cluster").isNotNull(), lit("rule")).otherwise(lit("no_rule")))

    cnts  = df.groupBy("label_source").agg(count("*").alias("n")).toPandas()
    total = int(cnts["n"].sum())
    rule_n = int(cnts.loc[cnts["label_source"]=="rule","n"].sum()) if total else 0
    if total:
        print(f"  Rule-matched: {rule_n:,} ({rule_n/total*100:.1f}%)   To ML: {total-rule_n:,}")
    return df

CAT_COLS   = ["EXTC_CODE","MIC","COMPANY_CODE","STATE","INVOICE_SOURCE"]
_tfidf     = None
_cat_enc   = None
_label_enc = None

def _features(pdf, fit=False):
    global _tfidf, _cat_enc
    texts = pdf["final_description"].fillna("").astype(str).tolist()
    if fit:
        _tfidf = TfidfVectorizer(max_features=5_000, ngram_range=(1,2), sublinear_tf=True, min_df=2)
        X_text = _tfidf.fit_transform(texts)
        _cat_enc = {}
    else:
        X_text = _tfidf.transform(texts)

    parts = []
    for c in CAT_COLS:
        vals = pdf[c].fillna("UNKNOWN").astype(str).values
        if fit:
            enc = LabelEncoder()
            enc.fit(np.append(vals, "UNKNOWN"))
            _cat_enc[c] = enc
        else:
            enc  = _cat_enc[c]
            known = set(enc.classes_)
            vals  = np.array([v if v in known else "UNKNOWN" for v in vals])
        parts.append(enc.transform(vals).reshape(-1,1).astype(np.float32))

    vh = pdf[COLS["vendor_name"]].fillna("").apply(
         lambda n: int(hashlib.md5(str(n).encode()).hexdigest(),16)%2000
    ).values.reshape(-1,1).astype(np.float32)
    am = np.log1p(np.abs(pdf[COLS["tran_amount"]].fillna(0).astype(float).values)).reshape(-1,1).astype(np.float32)
    return sp_hstack([X_text, csr_matrix(np.hstack(parts+[vh,am]))], format="csr")


def train_models(df_hist, df_curr=None):
    global _label_enc
    sel   = ["final_description","label_account_code"] + CAT_COLS + [COLS["vendor_name"],COLS["tran_amount"]]
    avail = [c for c in sel if c in df_hist.columns]

    df_lbl = df_hist.filter(col("label_source")=="rule").select(avail).dropna(subset=["label_account_code"])

    # Count classes, drop rare
    cnts = df_lbl.groupBy("label_account_code").agg(count("*").alias("n")).toPandas()
    valid = cnts.loc[cnts["n"]>=MIN_CLASS_SAMPLES,"label_account_code"].tolist()
    print(f"  {int(cnts['n'].sum()):,} labeled rows  |  {len(valid)} classes")

    # Stratified cap — Window row_number (single Spark job, no union chain)
    w = Window.partitionBy("label_account_code").orderBy(lit(1))
    pdf = (df_lbl.filter(col("label_account_code").isin(valid))
                 .withColumn("_r", row_number().over(w))
                 .filter(col("_r") <= MAX_ROWS_PER_CLASS)
                 .drop("_r")
                 .toPandas())
    print(f"  Training on {len(pdf):,} rows")

    _label_enc = LabelEncoder()
    y = _label_enc.fit_transform(pdf["label_account_code"].astype(str))
    X = _features(pdf, fit=True)
    print(f"  Feature matrix: {X.shape[0]:,} × {X.shape[1]:,}")
    del pdf

    # Eval set: current month rule rows (out-of-time), capped to 500/class
    if df_curr is not None:
        avail_c = [c for c in sel if c in df_curr.columns]
        w2 = Window.partitionBy("label_account_code").orderBy(lit(1))
        pdf_ev = (df_curr.filter(col("label_source")=="rule")
                         .select(avail_c).dropna(subset=["label_account_code"])
                         .withColumn("_r", row_number().over(w2))
                         .filter(col("_r") <= 500)
                         .drop("_r")
                         .toPandas())
        if len(pdf_ev) > 0:
            known = set(_label_enc.classes_)
            pdf_ev = pdf_ev[pdf_ev["label_account_code"].astype(str).isin(known)]
        if len(pdf_ev) == 0:
            df_curr = None

    if df_curr is not None:
        X_ev = _features(pdf_ev, fit=False)
        y_ev = _label_enc.transform(pdf_ev["label_account_code"].astype(str).values)
        del pdf_ev
        eval_name = "current month (out-of-time)"
    else:
        # 10% holdout from historical
        n = X.shape[0]
        idx = np.random.RandomState(42).permutation(n)
        cut = int(n*0.9)
        X, X_ev = X[idx[:cut]], X[idx[cut:]]
        y, y_ev = y[idx[:cut]], y[idx[cut:]]
        eval_name = "10% historical holdout"

    n_classes = len(_label_enc.classes_)
    print(f"  Eval: {eval_name}  ({X_ev.shape[0]:,} rows)  |  {n_classes} classes")

    # XGBoost only — no LightGBM to avoid Databricks version conflicts
    print("  Training XGBoost...")
    model = XGBClassifier(
        n_estimators=300,
        max_depth=6,
        learning_rate=0.1,
        subsample=0.8,
        colsample_bytree=0.8,
        eval_metric="mlogloss",
        tree_method="hist",
        grow_policy="lossguide",
        objective="multi:softprob",
        num_class=n_classes,
        random_state=42,
        n_jobs=-1,
    )
    model.fit(X, y, eval_set=[(X_ev, y_ev)], verbose=50,
              early_stopping_rounds=20)
    proba = model.predict_proba(X_ev)
    acc   = accuracy_score(y_ev, np.argmax(proba,1))
    print(f"  XGBoost → acc={acc:.4f}  best_round={model.best_iteration}")
    if df_curr is not None:
        print(f"  ↑ Real-world accuracy on current month data")
    return model

print("Features + training OK")

def classify(df, model, spark):
    id_col = COLS["invoice_id"]

    cnts      = df.groupBy("label_source").agg(count("*").alias("n")).toPandas()
    rule_n    = int(cnts.loc[cnts["label_source"]=="rule","n"].sum())
    no_rule_n = int(cnts.loc[cnts["label_source"]=="no_rule","n"].sum())
    print(f"  Rule: {rule_n:,}   ML: {no_rule_n:,}")

    df_rule    = df.filter(col("label_source")=="rule")
    df_no_rule = df.filter(col("label_source")=="no_rule")

    df_rule = (df_rule
        .withColumn("final_pred",    col("label_account_code"))
        .withColumn("confidence",    lit(1.0).cast(FloatType()))
        .withColumn("decision",      lit("rule"))
        .withColumn("ml_pred",       lit(None).cast(StringType()))
        .withColumn("ml_conf",       lit(None).cast(FloatType())))

    if no_rule_n == 0:
        return df_rule

    sel   = [id_col,"final_description"] + CAT_COLS + [COLS["vendor_name"],COLS["tran_amount"]]
    avail = [c for c in sel if c in df_no_rule.columns]

    _m = model; _le = _label_enc; _thr = CONFIDENCE_THRESHOLD
    _empty = [id_col,"ml_pred","ml_conf","final_pred","confidence","decision","label_source"]

    def _score(iterator):
        for pdf in iterator:
            if pdf.empty:
                yield pd.DataFrame(columns=_empty)
                continue
            X    = _features(pdf, fit=False)
            prob = _m.predict_proba(X)
            idx  = np.argmax(prob,1)
            conf = prob[np.arange(len(prob)),idx]
            acc  = conf >= _thr
            yield pd.DataFrame({
                id_col:       pdf[id_col].values,
                "ml_pred":    _le.inverse_transform(idx),
                "ml_conf":    conf.astype(float),
                "final_pred": _le.inverse_transform(idx),
                "confidence": conf.astype(float),
                "decision":   np.where(acc,"ml","need_review"),
                "label_source":np.where(acc,"ml","need_review"),
            })

    schema = StructType([
        StructField(id_col,        df.schema[id_col].dataType, True),
        StructField("ml_pred",     StringType(), True),
        StructField("ml_conf",     FloatType(),  True),
        StructField("final_pred",  StringType(), True),
        StructField("confidence",  FloatType(),  True),
        StructField("decision",    StringType(), True),
        StructField("label_source",StringType(), True),
    ])

    preds      = df_no_rule.select(avail).mapInPandas(_score, schema=schema)
    df_no_rule = df_no_rule.drop("label_source").join(preds, on=id_col, how="left")
    return df_rule.unionByName(df_no_rule, allowMissingColumns=True)

print("Inference OK")

def report(df, month):
    ts   = datetime.now().strftime("%Y%m%d_%H%M")
    path = f"/tmp/AP_Tax_{month}_{ts}.xlsx"

    cnts  = df.groupBy("label_source").agg(count("*").alias("n")).toPandas()
    total = int(cnts["n"].sum())
    def _n(s): return int(cnts.loc[cnts["label_source"]==s,"n"].sum()) if s in cnts["label_source"].values else 0
    rule_n=_n("rule"); ml_n=_n("ml"); ntr_n=_n("need_review")
    if total == 0:
        print("  No rows — skipping report"); return path

    summary = pd.DataFrame([
        {"Metric":"Total Invoices",       "Value":total,       "Pct":"100%"},
        {"Metric":"Classified by Rules",  "Value":rule_n,      "Pct":f"{rule_n/total*100:.1f}%"},
        {"Metric":"Classified by ML",     "Value":ml_n,        "Pct":f"{ml_n/total*100:.1f}%"},
        {"Metric":"Need Human Review",    "Value":ntr_n,       "Pct":f"{ntr_n/total*100:.1f}%"},
        {"Metric":"Auto-classified",      "Value":rule_n+ml_n, "Pct":f"{(rule_n+ml_n)/total*100:.1f}%"},
    ])

    want   = [COLS["invoice_id"],COLS["invoice_num"],"final_description",
              COLS["vendor_name"],COLS["extc"],COLS["mic"],COLS["state"],
              COLS["tran_amount"],COLS["account_code"],
              "ml_pred","ml_conf","final_pred","confidence","decision"]
    ntr_pdf = (df.filter(col("label_source")=="need_review")
                 .select([c for c in want if c in df.columns])
                 .limit(50_000).toPandas())

    with pd.ExcelWriter(path, engine="openpyxl") as w:
        summary.to_excel(w, sheet_name="Summary", index=False)
        ntr_pdf.to_excel(w, sheet_name="Need Review", index=False)

    if PATHS.get("reports_output"):
        dst = f"{PATHS['reports_output']}AP_Tax_{month}_{ts}.xlsx"
        dbutils.fs.cp(f"file:{path}", dst)
        print(f"  Report → {dst}")
        return dst

    print(f"  Report saved to /tmp/ (download from Databricks UI → Cluster → Driver Logs)")
    print(f"  Auto-classified: {rule_n+ml_n:,}/{total:,} ({(rule_n+ml_n)/total*100:.1f}%)")
    print(f"  Need review:     {ntr_n:,}/{total:,} ({ntr_n/total*100:.1f}%)")
    return path

print("Report OK")

    _start   = f"{INVOICE_MONTH}-01"
    _end     = f"{(_dt.date(_yr,_mo,1)+_dt.timedelta(days=32)).replace(day=1).strftime('%Y-%m')}-01"
else:
    _start, _end = None, None

# [1] Historical data for training (everything before current month)
print(f"\n[1/4] Loading historical data...")
hist_path = PATHS.get("historical_labeled") or PATHS["daily_invoices"]
df_h = spark.read.format("delta").load(hist_path)
if _start and not PATHS.get("historical_labeled"):
    df_h = df_h.filter(col(COLS["accounting_date"]) < _start)
df_h = build_labels(df_h, rules, spark)
df_h.cache(); df_h.count()

# [2] Current month for eval + classification
print(f"\n[2/4] Loading current month {INVOICE_MONTH or '(all)'}...")
if _start:
    df_c = (spark.read.format("delta").load(PATHS["daily_invoices"])
                 .filter((col(COLS["accounting_date"])>=_start)&(col(COLS["accounting_date"])<_end)))
else:
    df_c = spark.read.format("delta").load(PATHS["daily_invoices"])
df_c = build_labels(df_c, rules, spark)
df_c.cache(); df_c.count()

# [3] Train
print(f"\n[3/4] Training...")
model = train_models(df_h, df_curr=df_c)
df_h.unpersist()

# [4] Classify + checkpoint + report
print(f"\n[4/4] Classifying + report...")
df_scored = classify(df_c, model, spark)
df_c.unpersist()

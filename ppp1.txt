def _assign(df, cond, cluster, label):
    return (df.withColumn("cluster",      when(cond, lit(cluster)).otherwise(col("cluster")))
              .withColumn("rule_matched", when(cond, lit(label)).otherwise(col("rule_matched"))))


def _desc_contains(*kws):
    if not kws: return lit(False)
    c = lower(col("final_description")).like(f"%{kws[0].lower()}%")
    for k in kws[1:]: c = c | lower(col("final_description")).like(f"%{k.lower()}%")
    return c


def build_final_description(df):
    filler = ["SCM Purchase", "SCM PURCHASE", "scm purchase"]
    def _clean(c): return when(col(c).isin(filler) | col(c).isNull(), lit(None)).otherwise(col(c))
    df = (df.withColumn("_l", _clean(COLS["line_description"]))
            .withColumn("_d", _clean(COLS["description"]))
            .withColumn("_x", _clean(COLS["dist_all_description"])))
    return df.withColumn("final_description", coalesce(
        when(col("_l").isNotNull() & col("_d").isNotNull(), concat_ws(" ", col("_l"), col("_d"))),
        col("_l"), col("_d"), col("_x"), lit(""),
    )).drop("_l", "_d", "_x")


def _apply_keyword_rules(df, rules):
    if rules.empty: return df
    for _, rule in rules.iterrows():
        raw = str(rule["rule"]).strip()
        if "but not" in raw.lower():
            base, excl = raw.lower().split("but not", 1)
            cond = lower(col("final_description")).like(f"%{base.strip()}%")
            for e in excl.replace("(","").replace(")","").split(","):
                cond = cond & ~lower(col("final_description")).like(f"%{e.strip()}%")
        elif "," in raw:
            cond = _desc_contains(*[k.strip() for k in raw.split(",")])
        else:
            cond = lower(col("final_description")).like(f"%{raw.lower()}%")
        df = _assign(df, cond, rule["cluster"], f"kw:{raw[:40]}")
    return df

def _apply_parker_volpe_rules(df, pv):
    if pv.empty or "rule_type" not in pv.columns: return df
    EXTC=COLS["extc"]; VENDOR=COLS["vendor_name"]
    for _, r in pv[pv["rule_type"]=="extc_only"].iterrows():
        df = _assign(df, col(EXTC)==r["extc"], r["cluster"], f"pv:extc={r['extc']}")
    for _, r in pv[pv["rule_type"]=="vendor_only"].iterrows():
        df = _assign(df, col(VENDOR)==r["vendor_name"], r["cluster"], f"pv:vendor={r['vendor_name']}")
    for _, r in pv[pv["rule_type"]=="vendor_extc"].iterrows():
        df = _assign(df, (col(VENDOR)==r["vendor_name"]) & col(EXTC).isin([e.strip() for e in str(r.get("extc","")).split("OR")]), r["cluster"], "pv:vendor+extc")
    for _, r in pv[pv["rule_type"]=="vendor_desc"].iterrows():
        df = _assign(df, (col(VENDOR)==r["vendor_name"]) & _desc_contains(*[k.strip() for k in str(r.get("description_keyword","")).split("OR")]), r["cluster"], "pv:vendor+desc")
    for _, r in pv[pv["rule_type"]=="vendor_desc_extc"].iterrows():
        df = _assign(df, (col(VENDOR)==r["vendor_name"]) & _desc_contains(*[k.strip() for k in str(r.get("description_keyword","")).split("OR")]) & col(EXTC).isin([e.strip() for e in str(r.get("extc","")).split("OR")]), r["cluster"], "pv:vendor+desc+extc")
    return df


def build_labels(df, rules, spark):
    df = build_final_description(df)
    df = df.withColumn("cluster", lit(None).cast("string")).withColumn("rule_matched", lit(None).cast("string"))
    df = _apply_keyword_rules(df, rules["keyword"])
    df = _apply_generic_rules(df)
    df = _apply_parker_volpe_rules(df, rules["parker_volpe"])
    map_path = PATHS.get("cluster_account_map")
    if map_path:
        cmap = spark.read.csv(map_path, header=True).withColumnRenamed("cluster", "_c")
        df = df.join(F.broadcast(cmap), col("cluster")==col("_c"), "left").drop("_c")
        df = df.withColumn("label_account_code",
                 when(col("cluster").isNotNull() & col("account_code_mapped").isNotNull(), col("account_code_mapped"))
                 .otherwise(col(COLS["account_code"])))
    else:
        df = df.withColumn("account_code_mapped", lit(None).cast("string"))
        df = df.withColumn("label_account_code", col(COLS["account_code"]))
    df = df.withColumn("label_source", when(col("cluster").isNotNull(), lit("rule")).otherwise(lit("no_rule")))
    cnts = df.groupBy("label_source").agg(count("*").alias("n")).toPandas()
    total = int(cnts["n"].sum())
    rule_n = int(cnts.loc[cnts["label_source"]=="rule","n"].sum())
    if total > 0:
        print(f"   Rule-matched: {rule_n:,} ({rule_n/total*100:.1f}%)  To ML: {total-rule_n:,}")
    else:
        print("   Warning: 0 rows after build_labels — check your data path")
    return df

CAT_COLS   = ["EXTC_CODE", "MIC", "COMPANY_CODE", "STATE", "INVOICE_SOURCE"]
_tfidf_vec = None


def _hash_vendor(name, bins=5000):
    return int(hashlib.md5(str(name).encode()).hexdigest(), 16) % bins


def _fit_tfidf(texts):
    global _tfidf_vec
    _tfidf_vec = TfidfVectorizer(max_features=MODEL_CONFIG["tfidf_max_features"],
                                 ngram_range=(1,2), sublinear_tf=True, min_df=2)
    return _tfidf_vec.fit_transform(texts)   # returns sparse CSR


def _transform_tfidf(texts):
    return _tfidf_vec.transform(texts)       # returns sparse CSR


def build_feature_matrix(pdf, cat_encoders=None, fit=False):
    from scipy.sparse import csr_matrix as _csr
    X_text = _fit_tfidf(pdf["final_description"].fillna("").astype(str).tolist()) if fit \
             else _transform_tfidf(pdf["final_description"].fillna("").astype(str).tolist())
    cat_parts = []; ce = cat_encoders or {}
    if fit: ce = {}
    for c in CAT_COLS:
        vals = pdf[c].fillna("UNKNOWN").astype(str).values
        if fit:
            enc = LabelEncoder(); enc.fit(np.append(vals, "UNKNOWN")); ce[c] = enc
        else:
            enc = ce[c]; known = set(enc.classes_)
            vals = np.array([v if v in known else "UNKNOWN" for v in vals])
        cat_parts.append(enc.transform(vals).reshape(-1,1).astype(np.float32))
    vh = pdf[COLS["vendor_name"]].fillna("").apply(_hash_vendor).values.reshape(-1,1).astype(np.float32)
    am = np.log1p(np.abs(pdf[COLS["tran_amount"]].fillna(0).astype(float).values)).reshape(-1,1).astype(np.float32)
    X = sparse_hstack([X_text, _csr(np.hstack(cat_parts + [vh, am]))], format="csr")
    return X, ce

def train_models(df_hist, df_current=None):
    """
    df_hist    — historical Spark DF (already labelled by build_labels).
                 Rule-matched rows from this DF are used for training.
    df_current — current-month Spark DF (already labelled by build_labels).
                 Rule-matched rows from this DF are used as the REAL test set
                 (out-of-time evaluation — the model has never seen this month).
                 If None, a 10% holdout from df_hist is used for evaluation.
    """
    cfg = MODEL_CONFIG
    print("   Pulling historical rule-labeled records...")
    sel   = ["final_description","label_account_code"] + CAT_COLS + [COLS["vendor_name"],COLS["tran_amount"]]
    avail = [c for c in sel if c in df_hist.columns]

    df_labeled = df_hist.filter(col("label_source")=="rule").select(avail).dropna(subset=["label_account_code"])

    # Class counts — single Spark job
    counts_pdf = df_labeled.groupBy("label_account_code").agg(count("*").alias("n")).toPandas()
    valid_cls  = counts_pdf.loc[counts_pdf["n"] >= cfg["min_class_samples"], "label_account_code"].tolist()
    dropped    = int(counts_pdf.loc[counts_pdf["n"] <  cfg["min_class_samples"], "n"].sum())
    total      = int(counts_pdf["n"].sum())
    print(f"   Historical: {total:,} rows  |  {len(valid_cls)} classes  |  {dropped:,} dropped (rare)")

    # Stratified per-class cap — single Spark job using Window row_number.
    # Avoids loop+union which creates a deeply nested query plan (gRPC overflow).
    from pyspark.sql.window import Window
    from pyspark.sql.functions import row_number
    max_n = cfg["max_rows_per_class"]
    w = Window.partitionBy("label_account_code").orderBy(lit(1))
    df_train_spark = (df_labeled
                      .filter(col("label_account_code").isin(valid_cls))
                      .withColumn("_rn", row_number().over(w))
                      .filter(col("_rn") <= max_n)
                      .drop("_rn"))
    pdf_train = df_train_spark.toPandas()
    print(f"   Training on {len(pdf_train):,} records ({len(valid_cls)} classes, max {max_n:,}/class)")

    # Fit TF-IDF and encoders on training data only
    label_enc = LabelEncoder()
    y_train_all = label_enc.fit_transform(pdf_train["label_account_code"].astype(str))
    X_train_all, cat_encoders = build_feature_matrix(pdf_train, fit=True)
    print(f"   Feature matrix: {X_train_all.shape[0]:,} × {X_train_all.shape[1]:,}")
    del pdf_train

    # ── Evaluation set ────────────────────────────────────────────────────────
    # PREFERRED: current month rule-matched rows → true out-of-time test
    # FALLBACK:  10% holdout from historical (used for LGB early stopping only)
    if df_current is not None:
        print("   Building current-month eval set (out-of-time test)...")
        avail_c = [c for c in sel if c in df_current.columns]
        # Cap eval to 1000 rows/class — more than enough for reliable metrics
        from pyspark.sql.window import Window
        from pyspark.sql.functions import row_number as _rn
        _w = Window.partitionBy("label_account_code").orderBy(lit(1))
        pdf_eval = (df_current
                    .filter(col("label_source")=="rule")
                    .select(avail_c)
                    .dropna(subset=["label_account_code"])
                    .withColumn("_r", _rn().over(_w))
                    .filter(col("_r") <= 1_000)
                    .drop("_r")
                    .toPandas())
        if len(pdf_eval) == 0:
            print("   Warning: no rule-matched rows in current month — falling back to 10% holdout")
            df_current = None   # fall through to holdout

    if df_current is None:
        # 10% holdout from historical — only for early stopping, not real evaluation
        from sklearn.model_selection import train_test_split as _tts
        idx = np.arange(X_train_all.shape[0])
        idx_tr, idx_va = _tts(idx, test_size=0.10, random_state=cfg["random_seed"],
                               stratify=y_train_all if np.bincount(y_train_all).min() >= 2 else None)
        X_train_all, X_eval = X_train_all[idx_tr], X_train_all[idx_va]
        y_train_all, y_eval_raw = y_train_all[idx_tr], y_train_all[idx_va]
        print(f"   Using 10% historical holdout for early stopping ({len(y_eval_raw):,} rows)")
        eval_label_str = "Historical holdout"
        eval_from_current = False
    else:
        # Transform current month eval through already-fitted encoders
        X_eval_raw, _ = build_feature_matrix(pdf_eval, cat_encoders=cat_encoders, fit=False)
        y_eval_raw    = label_enc.transform(
            pdf_eval["label_account_code"].astype(str).apply(
                lambda v: v if v in label_enc.classes_ else label_enc.classes_[0]  # unseen → closest
            ).values
        )
        # Keep only classes the model was trained on
        known_mask = np.isin(y_eval_raw, np.arange(len(label_enc.classes_)))
        X_eval     = X_eval_raw[known_mask]
        y_eval_raw = y_eval_raw[known_mask]
        eval_label_str = f"Current month ({len(y_eval_raw):,} rule-matched rows)"
        eval_from_current = True
        del pdf_eval

    # Re-encode train labels to contiguous 0..N-1
    train_enc = LabelEncoder().fit(y_train_all)
    y_train   = train_enc.transform(y_train_all)
    X_train   = X_train_all
    del X_train_all, y_train_all

    # Align eval set to same label space
    eval_mask = np.isin(y_eval_raw, train_enc.classes_)
    X_eval    = X_eval[eval_mask]
    y_eval    = train_enc.transform(y_eval_raw[eval_mask])
    label_enc.classes_ = label_enc.classes_[train_enc.classes_]
    n_classes = len(label_enc.classes_)
    print(f"   Classes: {n_classes}  |  Eval ({eval_label_str}): {X_eval.shape[0]:,} rows")

    # ── Train XGBoost (early stopping — typically stops at 80-150 rounds) ────
    print("   Training XGBoost...")
    xgb_p = {k:v for k,v in cfg["xgb_params"].items() if k != "early_stopping_rounds"}
    xgb = XGBClassifier(**xgb_p, objective="multi:softprob", num_class=n_classes)
    xgb.fit(X_train, y_train, eval_set=[(X_eval, y_eval)], verbose=50,
            early_stopping_rounds=cfg["xgb_params"]["early_stopping_rounds"])
    xgb_proba = xgb.predict_proba(X_eval)          # one call, reused below
    xgb_pred  = np.argmax(xgb_proba, 1)
    print(f"   XGBoost  → acc={accuracy_score(y_eval,xgb_pred):.4f}  "
          f"f1={f1_score(y_eval,xgb_pred,average='weighted',zero_division=0):.4f}  "
          f"trees={xgb.best_iteration}")

    # ── Train LightGBM (early stopping — typically stops at 80-120 rounds) ───
    print("   Training LightGBM...")
    num_rounds = cfg["lgb_params"]["n_estimators"]
    lgb_p = {**{k:v for k,v in cfg["lgb_params"].items() if k not in ("n_estimators","verbose")},
              "objective":"multiclass", "num_class":n_classes, "verbose":-1}
    lgb_tr = lgb_lib.Dataset(X_train, label=y_train, free_raw_data=False)
    lgb_va = lgb_lib.Dataset(X_eval,  label=y_eval,  reference=lgb_tr, free_raw_data=False)
    lgb = lgb_lib.train(lgb_p, lgb_tr, num_boost_round=num_rounds, valid_sets=[lgb_va],
                        callbacks=[lgb_lib.early_stopping(20,verbose=False), lgb_lib.log_evaluation(50)])
    lgb_proba = lgb.predict(X_eval)                # one call, reused below
    lgb_pred  = np.argmax(lgb_proba, 1)
    print(f"   LightGBM → acc={accuracy_score(y_eval,lgb_pred):.4f}  "
          f"f1={f1_score(y_eval,lgb_pred,average='weighted',zero_division=0):.4f}  "
          f"trees={lgb.best_iteration}")

    # ── Ensemble (reuse already-computed probas — no extra predict calls) ────
    ens      = 0.5*xgb_proba + 0.5*lgb_proba
    ens_pred = np.argmax(ens, 1)
    print(f"   Ensemble → acc={accuracy_score(y_eval,ens_pred):.4f}  "
          f"f1={f1_score(y_eval,ens_pred,average='weighted',zero_division=0):.4f}")
    if eval_from_current:
        print(f"   ↑ Accuracy measured on CURRENT MONTH data — this is your real-world performance.")

    return xgb, lgb, label_enc, cat_encoders

print("✓ Model training OK")

def _score_batch(pdf, xgb, lgb, label_enc, cat_enc, id_col, w_xgb, w_lgb, thr):
    X, _ = build_feature_matrix(pdf, cat_encoders=cat_enc, fit=False)
    xp = xgb.predict_proba(X); lp = lgb.predict(X)
    ep = w_xgb*xp + w_lgb*lp
    xi = np.argmax(xp,1); li = np.argmax(lp,1); ei = np.argmax(ep,1)
    ec = ep[np.arange(len(ep)), ei]; acc = ec >= thr
    return pd.DataFrame({
        id_col:          pdf[id_col].values,
        "xgb_pred":      label_enc.inverse_transform(xi),
        "xgb_conf":      xp[np.arange(len(xp)), xi].astype(float),
        "lgb_pred":      label_enc.inverse_transform(li),
        "lgb_conf":      lp[np.arange(len(lp)), li].astype(float),
        "final_pred":    label_enc.inverse_transform(ei),
        "final_conf":    ec.astype(float),
        "decision_path": np.where(acc, "ml", "need_to_review"),
        "label_source":  np.where(acc, "ml", "need_to_review"),
    })


def classify_invoices(df, xgb, lgb, label_enc, cat_encoders, spark):
    id_col = COLS["invoice_id"]

    cnts      = df.groupBy("label_source").agg(count("*").alias("n")).toPandas()
    rule_n    = int(cnts.loc[cnts["label_source"]=="rule",    "n"].sum())
    no_rule_n = int(cnts.loc[cnts["label_source"]=="no_rule", "n"].sum())
    print(f"   Rule-matched: {rule_n:,}  |  Going to ML: {no_rule_n:,}")

    df_rule    = df.filter(col("label_source")=="rule")
    df_no_rule = df.filter(col("label_source")=="no_rule")

    df_rule = (df_rule
        .withColumn("final_pred",    col("label_account_code"))
        .withColumn("final_conf",    lit(1.0).cast(FloatType()))
        .withColumn("decision_path", lit("rule"))
        .withColumn("xgb_pred",      lit(None).cast(StringType()))
        .withColumn("xgb_conf",      lit(None).cast(FloatType()))
        .withColumn("lgb_pred",      lit(None).cast(StringType()))
        .withColumn("lgb_conf",      lit(None).cast(FloatType())))

    if no_rule_n == 0:
        return df_rule.withColumn("discovery_cluster_id", lit(None).cast("int"))

    avail = [c for c in [id_col,"final_description"]+CAT_COLS+[COLS["vendor_name"],COLS["tran_amount"]]
             if c in df_no_rule.columns]

    # Close over model objects for cloudpickle serialisation to executors
    _xgb=xgb; _lgb=lgb; _le=label_enc; _ce=cat_encoders
    _id=id_col; _wx=MODEL_CONFIG["ensemble_weights"]["xgb"]
    _wl=MODEL_CONFIG["ensemble_weights"]["lgb"]; _thr=THRESHOLDS["accept_conf"]
    _empty = [_id,"xgb_pred","xgb_conf","lgb_pred","lgb_conf","final_pred","final_conf","decision_path","label_source"]

    def _score_partition(iterator):
        for pdf in iterator:
            yield pd.DataFrame(columns=_empty) if pdf.empty \
                  else _score_batch(pdf, _xgb, _lgb, _le, _ce, _id, _wx, _wl, _thr)

    schema = StructType([
        StructField(id_col,          df.schema[id_col].dataType, True),
        StructField("xgb_pred",      StringType(), True),
        StructField("xgb_conf",      FloatType(),  True),
        StructField("lgb_pred",      StringType(), True),
        StructField("lgb_conf",      FloatType(),  True),
        StructField("final_pred",    StringType(), True),
        StructField("final_conf",    FloatType(),  True),
        StructField("decision_path", StringType(), True),
        StructField("label_source",  StringType(), True),
    ])

    preds = df_no_rule.select(avail).mapInPandas(_score_partition, schema=schema)
    df_no_rule = df_no_rule.drop("label_source").join(preds, on=id_col, how="left")
    return df_rule.unionByName(df_no_rule, allowMissingColumns=True)

print("✓ Inference OK")

def discover_new_clusters(df, spark):
    id_col = COLS["invoice_id"]
    ntr_n  = df.filter(col("label_source")=="need_to_review").count()
    print(f"   Need-to-review: {ntr_n:,}")

    df_out = df.withColumn("discovery_cluster_id", lit(None).cast("int"))
    if ntr_n < DISCOVERY_CONFIG["min_cluster_size"]:
        print("   Too few — skipping discovery.")
        return df_out, pd.DataFrame()

    ctx = [id_col,"final_description",COLS["vendor_name"],COLS["extc"],COLS["mic"],COLS["state"],COLS["tran_amount"]]
    avail = [c for c in ctx if c in df.columns]
    pdf   = df.filter(col("label_source")=="need_to_review").select(avail).limit(50_000).toPandas()

    X_emb = _transform_tfidf(pdf["final_description"].fillna("").astype(str).tolist()).toarray()
    # MiniBatchKMeans: ~10x faster than KMeans on large NTR sets
    # Silhouette check capped at 6 clusters — 9 full fits is overkill
    max_k = max(2, min(6, len(pdf)//20, len(pdf)-1))
    best_k, best_sc = 2, -1.0
    for k in range(2, max_k+1):
        lbl = MiniBatchKMeans(n_clusters=k, random_state=42, n_init=3, batch_size=1024).fit_predict(X_emb)
        sc  = silhouette_score(X_emb, lbl, sample_size=min(500, len(pdf)))
        if sc > best_sc: best_k, best_sc = k, sc
    print(f"   k={best_k}  silhouette={best_sc:.3f}")

    pdf["discovery_cluster_id"] = MiniBatchKMeans(n_clusters=best_k, random_state=42, n_init=5, batch_size=2048).fit_predict(X_emb)
    sizes = pdf["discovery_cluster_id"].value_counts()
    valid = sizes[sizes >= DISCOVERY_CONFIG["min_cluster_size"]].index
    pdf["discovery_cluster_id"] = pdf["discovery_cluster_id"].apply(lambda c: int(c) if c in valid else -1)

    rows = []
    for cid in sorted(valid):
        sub = pdf[pdf["discovery_cluster_id"]==cid]
        rows.append({
            "cluster_id": int(cid), "record_count": len(sub),
            "proposed_cluster_name": "", "proposed_account_code": "",
            "top_vendors":      ", ".join(sub[COLS["vendor_name"]].value_counts().head(3).index.tolist()),
            "top_extc_codes":   ", ".join(sub[COLS["extc"]].value_counts().head(3).index.tolist()),
            "top_states":       ", ".join(sub[COLS["state"]].value_counts().head(3).index.tolist()),
            "avg_tran_amount":  round(sub[COLS["tran_amount"]].astype(float).mean(), 2),
            "sample_descriptions": " | ".join(sub["final_description"].dropna().head(DISCOVERY_CONFIG["samples_per_cluster"]).tolist()),
        })
    cluster_summary = pd.DataFrame(rows)

    cid_spark = spark.createDataFrame(pdf[[id_col,"discovery_cluster_id"]])
    df_out = df.drop("discovery_cluster_id").join(cid_spark, on=id_col, how="left")
    df_out = df_out.withColumn("discovery_cluster_id", coalesce(col("discovery_cluster_id"), lit(None).cast("int")))
    print(f"   Valid clusters: {len(valid)}")
    return df_out, cluster_summary

print("✓ Cluster discovery OK")

def generate_report(df, cluster_summary, report_month):
    ts   = datetime.now().strftime("%Y%m%d_%H%M")
    path = f"/tmp/NeedToReview_{report_month}_{ts}.xlsx"

    cnts   = df.groupBy("label_source").agg(count("*").alias("n"), avg("final_conf").alias("avg_conf")).toPandas()
    total  = int(cnts["n"].sum())
    def _n(s):   return int(cnts.loc[cnts["label_source"]==s,"n"].sum()) if s in cnts["label_source"].values else 0
    def _avg(s):
        v = cnts.loc[cnts["label_source"]==s,"avg_conf"]
        return round(float(v.iloc[0]),4) if len(v)>0 and pd.notna(v.iloc[0]) else 0.0

    rule_n=_n("rule"); ml_n=_n("ml"); ntr_n=_n("need_to_review"); avg_c=_avg("ml")
    if total == 0:
        print("   Warning: 0 rows in scored DataFrame — no report generated.")
        return path
    summary = pd.DataFrame([
        {"Metric":"Total Invoices",        "Value":total,       "Pct":"100%"},
        {"Metric":"Classified by Rules",   "Value":rule_n,      "Pct":f"{rule_n/total*100:.1f}%"},
        {"Metric":"Classified by ML",      "Value":ml_n,        "Pct":f"{ml_n/total*100:.1f}%"},
        {"Metric":"Need Human Review",     "Value":ntr_n,       "Pct":f"{ntr_n/total*100:.1f}%"},
        {"Metric":"Total Auto-classified", "Value":rule_n+ml_n, "Pct":f"{(rule_n+ml_n)/total*100:.1f}%"},
        {"Metric":"Avg ML Confidence",     "Value":avg_c,       "Pct":""},
    ])

    desired = [COLS["invoice_id"],COLS["invoice_num"],"final_description",
               COLS["vendor_name"],COLS["extc"],COLS["mic"],COLS["state"],
               COLS["tran_amount"],COLS["account_code"],
               "xgb_pred","xgb_conf","lgb_pred","lgb_conf",
               "final_pred","final_conf","decision_path","discovery_cluster_id"]
    ntr_pdf = (df.filter(col("label_source")=="need_to_review")
                 .select([c for c in desired if c in df.columns])
                 .orderBy("discovery_cluster_id").limit(50_000).toPandas())

    with pd.ExcelWriter(path, engine="openpyxl") as w:
        summary.to_excel(w, sheet_name="Executive Summary", index=False)
        if not cluster_summary.empty:
            cluster_summary.to_excel(w, sheet_name="Discovered Clusters", index=False)
            try:
                from openpyxl.styles import PatternFill
                ws = w.sheets["Discovered Clusters"]
                yel = PatternFill(start_color="FFFF00", end_color="FFFF00", fill_type="solid")
                hdrs = [c.value for c in ws[1]]
                for fc in ["proposed_cluster_name","proposed_account_code"]:
                    if fc in hdrs:
                        ci = hdrs.index(fc)+1
                        for row in ws.iter_rows(min_row=2, min_col=ci, max_col=ci):
                            for cell in row: cell.fill = yel
            except Exception: pass
        else:
            pd.DataFrame({"message":["No new clusters this month."]}).to_excel(w, sheet_name="Discovered Clusters", index=False)
        ntr_pdf.to_excel(w, sheet_name="Raw NTR Records", index=False)

    if PATHS.get("reports_output"):
        dst = f"{PATHS['reports_output']}NeedToReview_{report_month}_{ts}.xlsx"
        dbutils.fs.cp(f"file:{path}", dst)
        print(f"   Report → {dst}")
        return dst
    print(f"   Report → /tmp (download from Driver Logs in Databricks UI)")
    print(f"   Auto-classified: {rule_n+ml_n:,}/{total:,} ({(rule_n+ml_n)/total*100:.1f}%)")
    print(f"   Need review:     {ntr_n:,}/{total:,} ({ntr_n/total*100:.1f}%)")
    return path

print("✓ Report OK")

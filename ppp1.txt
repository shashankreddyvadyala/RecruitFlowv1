# ── 2.3 GTE Large embeddings via mapPartitions ────────────────
print(f"[2.3] Generating {EMBED_DIM}-dim GTE embeddings...")

_EMBED_ENDPOINT_CLOSURE = EMBED_ENDPOINT
_EMBED_BATCH_CLOSURE    = EMBED_BATCH

def _embed_partition(rows):
    """Batch-embed one Spark partition via the GTE endpoint."""
    from mlflow.deployments import get_deploy_client
    client = get_deploy_client("databricks")
    ids, texts = [], []

    def _flush():
        if not texts:
            return
        resp = client.predict(_EMBED_ENDPOINT_CLOSURE, {"input": texts})
        for rid, item in zip(ids, resp["data"]):
            yield (rid, item["embedding"])
        ids.clear(); texts.clear()

    for row in rows:
        ids.append(row["INVOICE_ID"])
        texts.append(row["clean_description"])
        if len(texts) >= _EMBED_BATCH_CLOSURE:
            yield from _flush()
    yield from _flush()

_embed_schema = StructType([
    StructField("INVOICE_ID",    StringType(),           False),
    StructField("gte_embedding", ArrayType(FloatType()), False),
])

embeddings_df = (df.select("INVOICE_ID","clean_description")
                  .repartition(64)
                  .rdd.mapPartitions(_embed_partition)
                  .toDF(_embed_schema))

df = df.join(embeddings_df, on="INVOICE_ID", how="left")

# [5] PHASE 2 — XGBOOST TRAINING (mode=train only)
# ─────────────────────────────────────────────────────────────
if MODE == "train":
    print(f"\n{'='*60}\n  PHASE 2 — XGBOOST TRAINING\n{'='*60}")

    labels_df = (spark.read.format("delta").table(TBL["labels"])
                 .select("INVOICE_ID","INV_LINE_NUMBER","cluster_name")
                 .withColumnRenamed("cluster_name","label"))

    train_df = (df.join(labels_df, on=["INVOICE_ID","INV_LINE_NUMBER"], how="inner")
                  .filter(col("label").isNotNull())
                  .filter(col("gte_embedding").isNotNull()))

    # Remove rare classes
    cls_counts  = train_df.groupBy("label").agg(count("*").alias("n"))
    valid_labels = [r["label"] for r in cls_counts.filter(col("n") >= 10).collect()]
    train_df     = train_df.filter(col("label").isin(valid_labels))
    n_train      = train_df.count()
    n_classes    = len(valid_labels)
    print(f"   Training rows : {n_train:,}  classes : {n_classes}")

    # Class weights
    cls_freq      = {r["label"]: r["n"] for r in cls_counts.collect()}
    total_labeled = sum(cls_freq.values())
    cls_weights   = {l: total_labeled / (n_classes * f) for l, f in cls_freq.items()}

    pdf    = train_df.select("label","gte_embedding").toPandas()
    le     = LabelEncoder()
    y      = le.fit_transform(pdf["label"])
    X      = np.array(pdf["gte_embedding"].tolist(), dtype=np.float32)
    sw     = np.array([cls_weights.get(l, 1.0) for l in pdf["label"]])

    # Stratified K-Fold CV
    skf = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)
    cv_acc, cv_f1 = [], []
    for _, (tr, va) in enumerate(skf.split(X, y)):
        clf = XGBClassifier(**XGB_PARAMS, num_class=n_classes)
        clf.fit(X[tr], y[tr], sample_weight=sw[tr],
                eval_set=[(X[va], y[va])], verbose=False)
        p = clf.predict(X[va])
        cv_acc.append(accuracy_score(y[va], p))
        cv_f1.append(f1_score(y[va], p, average="weighted", zero_division=0))
    print(f"   CV acc={np.mean(cv_acc):.4f}±{np.std(cv_acc):.4f}  f1={np.mean(cv_f1):.4f}±{np.std(cv_f1):.4f}")

    # Final model
    final_xgb = XGBClassifier(**XGB_PARAMS, num_class=n_classes)
    final_xgb.fit(X, y, sample_weight=sw, verbose=False)

    # Log + register in MLflow UC
    mlflow.set_registry_uri("databricks-uc")
    mlflow.set_experiment("/atims/xgboost_non_norad_training")
    with mlflow.start_run(run_name=f"xgb_train_{RUN_MONTH_STR}") as run:
        mlflow.log_params({"run_month": RUN_MONTH_STR, "n_classes": n_classes,
                           "n_train": n_train, "embedding_model": EMBED_ENDPOINT})
        mlflow.log_metrics({"cv_acc_mean": float(np.mean(cv_acc)),
                             "cv_f1_mean":  float(np.mean(cv_f1))})
        le_path = "/tmp/label_encoder.pkl"
        with open(le_path,"wb") as fh: pickle.dump(le, fh)
        mlflow.log_artifact(le_path, artifact_path="encoders")
        mlflow.xgboost.log_model(final_xgb, artifact_path="model",
                                  registered_model_name=TBL["model_uc"])

    # Promote alias
    mlflow_client = MlflowClient()
    versions = mlflow_client.search_model_versions(f"name='{TBL['model_uc']}'")
    latest_v = sorted(versions, key=lambda v: int(v.version))[-1]
    mlflow_client.set_registered_model_alias(TBL["model_uc"], "production", latest_v.version)
    print(f"   Model v{latest_v.version} → alias 'production'")

    # Refresh serving endpoint
    from databricks.sdk import WorkspaceClient
    from databricks.sdk.service.serving import ServedEntityInput, EndpointCoreConfigInput
    wc = WorkspaceClient()
    try:
        wc.serving_endpoints.update_config(
            name=XGB_ENDPOINT,
            served_entities=[ServedEntityInput(entity_name=TBL["model_uc"],
                entity_version=latest_v.version, workload_size="Small",
                scale_to_zero_enabled=False)])
        print(f"   Endpoint '{XGB_ENDPOINT}' updated → v{latest_v.version}")
    except Exception:
        wc.serving_endpoints.create(name=XGB_ENDPOINT,
            config=EndpointCoreConfigInput(served_entities=[ServedEntityInput(
                entity_name=TBL["model_uc"], entity_version=latest_v.version,
                workload_size="Small", scale_to_zero_enabled=False)]))
        print(f"   Endpoint '{XGB_ENDPOINT}' created.")
else:
    print("[2.4-2.6] MODE=inference — using existing endpoint.")

             .select("rule_id","extc_code","mic","account_code","category","priority")
             .withColumnRenamed("extc_code",    "r_extc")

df_ranked = (pr
    .join(F.broadcast(det_rules),
          ((col("EXTC_CODE")==col("r_extc")) & (col("MIC")==col("r_mic")) & (col("ACCOUNT_CODE")==col("r_account"))) |
          ((col("EXTC_CODE")==col("r_extc")) & (col("MIC")==col("r_mic")) & col("r_account").isNull()) |
          ((col("EXTC_CODE")==col("r_extc")) & col("r_mic").isNull()      & col("r_account").isNull()),
          how="left")
    .withColumn("_rank", F.rank().over(
        Window.partitionBy("INVOICE_ID","INV_LINE_NUMBER")
              .orderBy(col("priority").asc_nulls_last())))
    .filter((col("_rank")==1) | col("r_rule_id").isNull())
    .drop("_rank","priority")
)

det_matched_df = (df_ranked.filter(col("r_rule_id").isNotNull())

    .withColumn("category",   col("r_category")).withColumn("confidence", lit(1.0).cast(FloatType()))
    .withColumn("match_type", lit("det"))
    .drop("r_extc","r_mic","r_account","r_category","r_rule_id"))

no_rules_df = (df_ranked.filter(col("r_rule_id").isNull())

vsc = VectorSearchClient()

def _batch_vs(pdf, index_name, threshold, k):
    """Batch Vector Search over pandas DataFrame. Returns pandas DataFrame."""
    idx    = vsc.get_index(endpoint_name=VS_ENDPOINT, index_name=index_name)
    ids    = pdf["INVOICE_ID"].tolist()
    vecs   = pdf["gte_embedding"].tolist()
    out    = []
    for i in range(0, len(ids), 100):
        b_ids  = ids[i:i+100]; b_vecs = vecs[i:i+100]
        resp   = idx.similarity_search(
            query_vectors=b_vecs if len(b_vecs)>1 else None,
            query_vector =b_vecs[0] if len(b_vecs)==1 else None,
            columns=["rule_id","category","score"], num_results=k)
        data = resp.get("result",{}).get("data_array",[])
        for j, inv_id in enumerate(b_ids):
            top = data[j][0] if j < len(data) and data[j] else None
            out.append({"INVOICE_ID": inv_id,
                        "vs_category": top[1] if top else None,
                        "vs_rule_id":  top[0] if top else None,
                        "vs_score":    float(top[2]) if top else 0.0})
    return pd.DataFrame(out)

# ── 3.4c Parse JSON response ──────────────────────────────────
from pyspark.sql.functions import udf
from pyspark.sql.types import StructType as ST, StructField as SF

_WU_SCHEMA = ST([SF("short_justification", StringType(), True),
                  SF("full_writeup",        StringType(), True),
                  SF("audit_ready",         BooleanType(), True)])

@udf(returnType=_WU_SCHEMA)
def _parse_writeup(raw):
    if not raw: return (None, None, False)
    try:
        clean = raw.strip()
        if clean.startswith("```"): clean = "\n".join(clean.split("\n")[1:-1])
        d = json.loads(clean)
        return (str(d.get("short_justification",""))[:500],
                str(d.get("full_writeup",""))[:2000],
                bool(d.get("audit_ready", False)))
    except Exception:
        return (raw[:200], raw[:2000], False)

            .withColumn("flags",        when(col("_f"), array_union(col("flags"), F.array(lit("TAXABILITY_SANITY_FAIL")))).otherwise(col("flags")))
            .withColumn("anomaly_score",when(col("_f"), col("anomaly_score")+4).otherwise(col("anomaly_score")))
            .drop("sr_state","sr_cat","expected_taxability","_f"))
except Exception as e:
    print(f"   [WARN] state_rules unavailable: {e}")

# Check 6 — batch-level outlier (top 0.5% anomaly_score)
print("[3.5f] Check 6: batch-level outlier...")
p995 = qa.approxQuantile("anomaly_score",[0.995],0.01)[0]
qa   = (qa
    .withColumn("_f", col("anomaly_score") > p995)
    .withColumn("flags",        when(col("_f"), array_union(col("flags"), F.array(lit("BATCH_OUTLIER")))).otherwise(col("flags")))
    .withColumn("anomaly_score",when(col("_f"), col("anomaly_score")+1).otherwise(col("anomaly_score")))
    .drop("_f")
)

# Check 7 — null taxability direct escalation
print("[3.5g] Check 7: null taxability escalation...")
qa = (qa
    .withColumn("flags",        when(col("taxability").isNull() & (col("tax_source")=="no_match"),
                                     array_union(col("flags"), F.array(lit("NO_TAXABILITY_MATCH")))).otherwise(col("flags")))
    .withColumn("anomaly_score",when(col("taxability").isNull(), col("anomaly_score")+5).otherwise(col("anomaly_score")))
)

# Assign review_priority + override_suggested
qa = (qa
    .withColumn("review_priority",
        when(F.array_contains(col("flags"),"NO_TAXABILITY_MATCH") | (col("ensemble_confidence") < CONF_CRITICAL), lit("CRITICAL"))
        .when(F.array_contains(col("flags"),"TAXABILITY_SANITY_FAIL") |
              F.array_contains(col("flags"),"RULE_CLASSIFIER_CONFLICT") |
              F.array_contains(col("flags"),"HISTORICAL_RATE_DEVIATION"),                                          lit("HIGH"))
        .when(F.array_contains(col("flags"),"AMOUNT_OUTLIER") |
              F.array_contains(col("flags"),"VENDOR_CATEGORY_INCONSISTENCY"),                                      lit("MEDIUM"))
        .otherwise(lit("LOW")))
    .withColumn("override_suggested",
        col("review_priority").isin("CRITICAL","HIGH") & (col("ensemble_confidence") < 0.70))
)

    .option("overwriteSchema","true")
    .option("replaceWhere", f"ACCOUNTING_DATE like '{RUN_MONTH_STR}%'")
    .saveAsTable(TBL["qa_out"]))

qa.groupBy("review_priority").count().orderBy("review_priority").show()

if override_n > 0:
    def _reclassify(rdf, pass_num):
        """Re-run XGBoost + ai_classify with QA-flag context injected."""
        # XGBoost re-predict
        rpdf   = rdf.select("INVOICE_ID","gte_embedding").toPandas()
        parts  = []
        for s in range(0, len(rpdf), XGB_CHUNK):
            chunk = rpdf.iloc[s:s+XGB_CHUNK]
            resp  = deploy_client.predict(XGB_ENDPOINT, {"inputs": chunk["gte_embedding"].tolist()})
            prb   = np.array(resp["predictions"])
            idx   = np.argmax(prb, axis=1)
            parts.append(pd.DataFrame({
                "INVOICE_ID":   chunk["INVOICE_ID"].values,
                "new_xgb_pred": [CATEGORIES[i] if i < len(CATEGORIES) else "Unknown" for i in idx],
                "new_xgb_conf": prb[np.arange(len(prb)), idx].astype(float),
            }))
        xgb_re = spark.createDataFrame(pd.concat(parts, ignore_index=True),
                     "INVOICE_ID STRING, new_xgb_pred STRING, new_xgb_conf FLOAT")
        rdf = rdf.join(xgb_re, on="INVOICE_ID", how="left")

        # ai_classify with flag context
        rdf.createOrReplaceTempView("reroute_temp")
        llm_re = spark.sql(f"""
            SELECT INVOICE_ID,
                   ai_classify(
                       CONCAT(clean_description,' [QA:',CAST(flags AS STRING),
                              '][Prior:',COALESCE(final_category,'?'),']'),
                       ARRAY({CATS_STR})
                   ) AS new_llm_cat
            FROM reroute_temp
        """).withColumn("new_llm_conf", lit(0.75).cast(FloatType()))

        rdf = (rdf.join(llm_re, on="INVOICE_ID", how="left")
            .withColumn("_agree", col("new_xgb_pred")==col("new_llm_cat"))
            .withColumn("_ens",   ((col("new_xgb_conf")+col("new_llm_conf"))/2.0).cast(FloatType()))
        )

        # GPT-4o on remaining disagreements
        dis = rdf.filter(col("_agree")==False)
        if dis.count() > 0:
            dis.createOrReplaceTempView("reroute_dis_temp")
            res = spark.sql(f"""
                SELECT INVOICE_ID,
                       TRIM(ai_query('{GPT4O_ENDPOINT}',
                           CONCAT('QA flags: ',CAST(flags AS STRING),
                                  ' | XGB: ',new_xgb_pred,' | LLM: ',new_llm_cat,
                                  ' | Prior: ',COALESCE(final_category,'?'),
                                  ' | Desc: ',clean_description,
                                  '. ONE category from: {CATS_STR}')
                       )) AS rr_resolved
                FROM reroute_dis_temp
            """)
            rdf = rdf.join(res, on="INVOICE_ID", how="left")
        else:
            rdf = rdf.withColumn("rr_resolved", lit(None).cast(StringType()))

        rdf = (rdf
            .withColumn("final_category",
                when(col("_agree"),                                col("new_xgb_pred"))
                .when(col("rr_resolved").isNotNull(),              col("rr_resolved"))
                .otherwise(col("new_xgb_pred")))
            .withColumn("ensemble_confidence",
                when(col("_agree"), col("_ens"))
                .otherwise(lit(0.70).cast(FloatType())))
            .drop("_agree","_ens","new_xgb_pred","new_xgb_conf","new_llm_cat","new_llm_conf","rr_resolved")
        )
        return rdf

    def _retax(rdf):
        """Re-apply tax lookup + use tax after reclassification."""
        for c in ["taxability","tax_source","Reverse_UseTax"]:
            if c in rdf.columns: rdf = rdf.drop(c)
        rdf = (rdf.join(F.broadcast(tax_m),
                        (col("final_category")==col("tm_cluster")) & (col("STATE")==col("tm_state")), "left")
                  .drop("tm_cluster","tm_state")
                  .withColumn("tax_source",
                      when(col("taxability").isNotNull(), lit("matrix")).otherwise(lit("no_match")))
                  .withColumn("Reverse_UseTax",
                      when(col("USE_TAX_AMOUNT").isNull(),                                         lit(None).cast(StringType()))
                      .when(col("taxability").isNull(),                                            lit(None).cast(StringType()))
                      .when(col("USE_TAX_AMOUNT")==0,                                              lit("No"))
                      .when((col("USE_TAX_AMOUNT")>0) & (col("taxability")=="Exempt"),             lit("Yes"))
                      .when((col("USE_TAX_AMOUNT")>0) & (col("taxability")=="Taxable"),            lit("No"))
                      .otherwise(lit(None).cast(StringType())))
        )
        return rdf

    def _reqa(rdf):
        """Recompute QA flags after reclassification (simplified)."""
        rdf = (rdf.withColumn("flags",         F.array().cast(ArrayType(StringType())))
                  .withColumn("anomaly_score",  lit(0).cast(IntegerType())))
        rdf = (rdf
            .withColumn("flags", when(col("taxability").isNull(),
                array_union(col("flags"), F.array(lit("NO_TAXABILITY_MATCH")))).otherwise(col("flags")))
            .withColumn("anomaly_score", when(col("taxability").isNull(), col("anomaly_score")+5).otherwise(col("anomaly_score")))
            .withColumn("_lc", col("ensemble_confidence") < CONF_MEDIUM)
            .withColumn("flags", when(col("_lc"), array_union(col("flags"), F.array(lit("LOW_CONFIDENCE_AFTER_REROUTE")))).otherwise(col("flags")))
            .withColumn("anomaly_score", when(col("_lc"), col("anomaly_score")+2).otherwise(col("anomaly_score")))
            .drop("_lc")
        )
        rdf = (rdf
            .withColumn("review_priority",
                when(col("taxability").isNull() | (col("ensemble_confidence") < CONF_CRITICAL), lit("CRITICAL"))
                .when(F.array_contains(col("flags"),"LOW_CONFIDENCE_AFTER_REROUTE"),             lit("HIGH"))
                .otherwise(lit("MEDIUM")))
            .withColumn("override_suggested",
                col("review_priority").isin("CRITICAL","HIGH") & (col("ensemble_confidence") < 0.70))
        )
        return rdf

    current = override_df
    for pass_num in range(1, MAX_REROUTE + 1):
        print(f"\n  ── Re-route pass {pass_num} ──")

        remaining = current.filter(col("override_suggested")==True)

        print(f"     Resolved : {r_n:,}   Still flagged : {rem_n:,}")

        if rem_n == 0: break
        if pass_num == MAX_REROUTE:
            escalated = remaining.withColumn("escalated", lit(True)).withColumn("review_priority", lit("CRITICAL"))
            final_resolved = final_resolved.unionByName(escalated, allowMissingColumns=True)
            print(f"     Escalated to human review : {rem_n:,}")
        else:
            current = remaining

escalated_n = final_resolved.filter(col("escalated")==True).count()
